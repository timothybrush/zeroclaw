26/// Build a single shared SopEngine + SopAuditLogger pair.
27///
28/// This is the sole construction site for SOP state within a daemon.
29/// Callers receive `Arc<Mutex<SopEngine>>` and `Arc<SopAuditLogger>`
30/// handles — never call `SopEngine::new` or `SopAuditLogger::new`
31/// directly outside this module.
32pub fn build_sop_engine(
33    config: SopConfig,
34    workspace_dir: &Path,
35    audit_memory: Arc<dyn Memory>,
36) -> (Arc<Mutex<SopEngine>>, Arc<SopAuditLogger>) {
37    let mut engine = SopEngine::new(config);
38    engine.reload(workspace_dir);
39    let engine = Arc::new(Mutex::new(engine));
40    let audit = Arc::new(SopAuditLogger::new(audit_memory));
41    (engine, audit)
42}

44/// Parse an execution mode string into `SopExecutionMode`, falling back to
45/// `Supervised` for unknown values.
46pub fn parse_execution_mode(s: &str) -> SopExecutionMode {
47    match s.trim().to_lowercase().as_str() {
48        "auto" => SopExecutionMode::Auto,
49        "step_by_step" => SopExecutionMode::StepByStep,
50        "priority_based" => SopExecutionMode::PriorityBased,
51        "deterministic" => SopExecutionMode::Deterministic,
52        // "supervised" and any unknown value
53        _ => SopExecutionMode::Supervised,
54    }
55}

59/// Return the default SOPs directory: `<workspace>/sops`.
60fn sops_dir(workspace_dir: &Path) -> PathBuf {
61    workspace_dir.join("sops")
62}

64/// Resolve the SOPs directory from config, falling back to workspace default.
65pub fn resolve_sops_dir(workspace_dir: &Path, config_dir: Option<&str>) -> PathBuf {
66    match config_dir {
67        Some(dir) if !dir.is_empty() => {
68            let expanded = shellexpand::tilde(dir);
69            PathBuf::from(expanded.as_ref())
70        }
71        _ => sops_dir(workspace_dir),
72    }
73}

87/// Load SOPs from a specific directory. Each subdirectory may contain
88/// `SOP.toml` (metadata + triggers) and `SOP.md` (procedure steps).
89pub fn load_sops_from_directory(
90    sops_dir: &Path,
91    default_execution_mode: SopExecutionMode,
92) -> Vec<Sop> {
93    if !sops_dir.exists() {
94        return Vec::new();
95    }
96
97    let mut sops = Vec::new();
98
99    let Ok(entries) = std::fs::read_dir(sops_dir) else {
100        return sops;
101    };
102
103    for entry in entries.flatten() {
104        let path = entry.path();
105        if !path.is_dir() {
106            continue;
107        }
108
109        let toml_path = path.join("SOP.toml");
110        if !toml_path.exists() {
111            continue;
112        }
113
114        match load_sop(&path, default_execution_mode) {
115            Ok(sop) => sops.push(sop),
116            Err(e) => {
117                ::zeroclaw_log::record!(
118                    WARN,
119                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
120                        .with_outcome(::zeroclaw_log::EventOutcome::Unknown)
121                        .with_attrs(::serde_json::json!({"error": format!("{}", e)})),
122                    &format!("Failed to load SOP from {}", path.display().to_string())
123                );
124            }
125        }
126    }
127
128    sops.sort_by(|a, b| a.name.cmp(&b.name));
129    sops
130}

132/// Load a single SOP from a directory containing SOP.toml and optionally SOP.md.
133fn load_sop(sop_dir: &Path, default_execution_mode: SopExecutionMode) -> Result<Sop> {
134    let toml_path = sop_dir.join("SOP.toml");
135    let toml_content = std::fs::read_to_string(&toml_path)?;
136    let manifest: SopManifest = toml::from_str(&toml_content)?;
137
138    let md_path = sop_dir.join("SOP.md");
139    let steps = if md_path.exists() {
140        let md_content = std::fs::read_to_string(&md_path)?;
141        parse_steps(&md_content)
142    } else {
143        Vec::new()
144    };
145
146    let SopMeta {
147        name,
148        description,
149        version,
150        priority,
151        execution_mode,
152        cooldown_secs,
153        max_concurrent,
154        deterministic,
155    } = manifest.sop;
156
157    // When deterministic=true, override execution_mode to Deterministic
158    let effective_mode = if deterministic {
159        SopExecutionMode::Deterministic
160    } else {
161        execution_mode.unwrap_or(default_execution_mode)
162    };
163
164    Ok(Sop {
165        name,
166        description,
167        version,
168        priority,
169        execution_mode: effective_mode,
170        triggers: manifest.triggers,
171        steps,
172        cooldown_secs,
173        max_concurrent,
174        location: Some(sop_dir.to_path_buf()),
175        deterministic,
176    })
177}

181/// Parse procedure steps from SOP.md content.
182///
183/// Expects a `## Steps` heading followed by numbered items (`1.`, `2.`, …).
184/// Each item's first bold text (`**...**`) is the step title; the rest is body.
185/// Sub-bullets `- tools:` and `- requires_confirmation: true` are parsed.
186pub fn parse_steps(md: &str) -> Vec<SopStep> {
187    let mut steps = Vec::new();
188    let mut in_steps_section = false;
189    let mut current_number: Option<u32> = None;
190    let mut current_title = String::new();
191    let mut current_body = String::new();
192    let mut current_tools: Vec<String> = Vec::new();
193    let mut current_requires_confirmation = false;
194    let mut current_kind = SopStepKind::Execute;
195
196    for line in md.lines() {
197        let trimmed = line.trim();
198
199        // Detect ## Steps heading
200        if trimmed.starts_with("## ") {
201            if trimmed.eq_ignore_ascii_case("## steps") || trimmed.eq_ignore_ascii_case("## Steps")
202            {
203                in_steps_section = true;
204                continue;
205            }
206            // Any other ## heading ends the steps section
207            if in_steps_section {
208                // Flush pending step
209                flush_step(
210                    &mut steps,
211                    &mut current_number,
212                    &mut current_title,
213                    &mut current_body,
214                    &mut current_tools,
215                    &mut current_requires_confirmation,
216                    &mut current_kind,
217                );
218                in_steps_section = false;
219            }
220            continue;
221        }
222
223        if !in_steps_section {
224            continue;
225        }
226
227        // Check for numbered item: `1.`, `2.`, etc.
228        if let Some(rest) = parse_numbered_item(trimmed) {
229            // Flush previous step
230            flush_step(
231                &mut steps,
232                &mut current_number,
233                &mut current_title,
234                &mut current_body,
235                &mut current_tools,
236                &mut current_requires_confirmation,
237                &mut current_kind,
238            );
239
240            let step_num = u32::try_from(steps.len())
241                .unwrap_or(u32::MAX)
242                .saturating_add(1);
243            current_number = Some(step_num);
244
245            // Extract title from bold text: **title** — body
246            if let Some((title, body)) = extract_bold_title(rest) {
247                current_title = title;
248                current_body = body;
249            } else {
250                current_title = rest.to_string();
251                current_body = String::new();
252            }
253            current_tools = Vec::new();
254            current_requires_confirmation = false;
255            continue;
256        }
257
258        // Sub-bullet parsing (only when inside a step)
259        if current_number.is_some() && trimmed.starts_with("- ") {
260            let bullet = trimmed.trim_start_matches("- ").trim();
261            if let Some(tools_str) = bullet.strip_prefix("tools:") {
262                current_tools = tools_str
263                    .split(',')
264                    .map(|t| t.trim().to_string())
265                    .filter(|t| !t.is_empty())
266                    .collect();
267            } else if bullet.starts_with("requires_confirmation:") {
268                if let Some(val) = bullet.strip_prefix("requires_confirmation:") {
269                    current_requires_confirmation = val.trim().eq_ignore_ascii_case("true");
270                }
271            } else if bullet.starts_with("kind:") {
272                if let Some(val) = bullet.strip_prefix("kind:") {
273                    let val = val.trim();
274                    if val.eq_ignore_ascii_case("checkpoint") {
275                        current_kind = SopStepKind::Checkpoint;
276                    } else {
277                        current_kind = SopStepKind::Execute;
278                    }
279                }
280            } else {
281                // Continuation body line
282                if !current_body.is_empty() {
283                    current_body.push('\n');
284                }
285                current_body.push_str(trimmed);
286            }
287            continue;
288        }
289
290        // Continuation line for step body
291        if current_number.is_some() && !trimmed.is_empty() {
292            if !current_body.is_empty() {
293                current_body.push('\n');
294            }
295            current_body.push_str(trimmed);
296        }
297    }
298
299    // Flush final step
300    flush_step(
301        &mut steps,
302        &mut current_number,
303        &mut current_title,
304        &mut current_body,
305        &mut current_tools,
306        &mut current_requires_confirmation,
307        &mut current_kind,
308    );
309
310    steps
311}

313/// Flush accumulated step state into the steps vector.
314fn flush_step(
315    steps: &mut Vec<SopStep>,
316    number: &mut Option<u32>,
317    title: &mut String,
318    body: &mut String,
319    tools: &mut Vec<String>,
320    requires_confirmation: &mut bool,
321    kind: &mut SopStepKind,
322) {
323    if let Some(n) = number.take() {
324        steps.push(SopStep {
325            number: n,
326            title: std::mem::take(title),
327            body: body.trim().to_string(),
328            suggested_tools: std::mem::take(tools),
329            requires_confirmation: *requires_confirmation,
330            kind: *kind,
331            schema: None,
332        });
333        *body = String::new();
334        *requires_confirmation = false;
335        *kind = SopStepKind::Execute;
336    }
337}

339/// Try to parse `N. rest` from a line, returning `rest` if successful.
340fn parse_numbered_item(line: &str) -> Option<&str> {
341    let dot_pos = line.find(". ")?;
342    let prefix = &line[..dot_pos];
343    if prefix.chars().all(|c| c.is_ascii_digit()) && !prefix.is_empty() {
344        Some(line[dot_pos + 2..].trim())
345    } else {
346        None
347    }
348}

350/// Extract `**title**` from the beginning of text, returning (title, rest).
351pub fn extract_bold_title(text: &str) -> Option<(String, String)> {
352    let start = text.find("**")?;
353    let after_start = start + 2;
354    let end = text[after_start..].find("**")?;
355    let title = text[after_start..after_start + end].to_string();
356
357    // Rest is everything after the closing ** and any separator (— or -)
358    let rest_start = after_start + end + 2;
359    let rest = text[rest_start..].trim();
360    let rest = rest
361        .strip_prefix("—")
362        .or_else(|| rest.strip_prefix("–"))
363        .or_else(|| rest.strip_prefix("-"))
364        .unwrap_or(rest)
365        .trim();
366
367    Some((title, rest.to_string()))
368}

372/// Validate a loaded SOP and return a list of warnings.
373pub fn validate_sop(sop: &Sop) -> Vec<String> {
374    let mut warnings = Vec::new();
375
376    if sop.name.is_empty() {
377        warnings.push("SOP name is empty".into());
378    }
379    if sop.description.is_empty() {
380        warnings.push("SOP description is empty".into());
381    }
382    if sop.triggers.is_empty() {
383        warnings.push("SOP has no triggers defined".into());
384    }
385    if sop.steps.is_empty() {
386        warnings.push("SOP has no steps (missing or empty SOP.md)".into());
387    }
388
389    // Check step numbering continuity
390    for (i, step) in sop.steps.iter().enumerate() {
391        let expected = u32::try_from(i).unwrap_or(u32::MAX).saturating_add(1);
392        if step.number != expected {
393            warnings.push(format!(
394                "Step numbering gap: expected {expected}, got {}",
395                step.number
396            ));
397        }
398        if step.title.is_empty() {
399            warnings.push(format!("Step {} has an empty title", step.number));
400        }
401    }
402
403    warnings
