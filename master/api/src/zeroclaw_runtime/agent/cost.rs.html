9/// Per-provider pricing snapshot consumed by the cost tracker.
10///
11/// Outer key: model provider alias (e.g. `openrouter`, `anthropic`,
12/// `azure-openai`). Inner key: user-defined model identifier, optionally
13/// suffixed with `.input` / `.output` to encode pricing dimension. Values
14/// are USD per 1M tokens.
15pub type ModelProviderPricing = HashMap<String, HashMap<String, f64>>;

17/// Per-scope token/cost accumulator. Records pushed by
18/// `record_tool_loop_cost_usage` alongside the shared `CostTracker` so the
19/// wrapping code can read out the total for *this* call after the scope
20/// exits, without racing concurrent requests sharing the same tracker.
21#[derive(Default, Clone, Copy, Debug)]
22pub struct TurnUsage {
23    pub input_tokens: u64,
24    pub output_tokens: u64,
25    pub cost_usd: f64,
26}

28/// Context for cost tracking within the tool call loop.
29/// Scoped via `tokio::task_local!` at call sites (channels, gateway).
30#[derive(Clone)]
31pub struct ToolLoopCostTrackingContext {
32    /// Shared cost tracker. `None` for usage-only contexts that accumulate
33    /// per-turn token totals without persisting cost records or enforcing
34    /// budgets (see [`Self::usage_only`]).
35    pub tracker: Option<Arc<CostTracker>>,
36    pub model_provider_pricing: Arc<ModelProviderPricing>,
37    pub turn_usage: Arc<Mutex<TurnUsage>>,
38    /// Alias of the agent driving this turn. Stamped onto persisted
39    /// `CostRecord`s so `/api/cost?agent=<alias>` can attribute spend.
40    pub agent_alias: Option<String>,
41}
42
43impl ToolLoopCostTrackingContext {
44    pub fn new(
45        tracker: Arc<CostTracker>,
46        model_provider_pricing: Arc<ModelProviderPricing>,
47    ) -> Self {
48        Self {
49            tracker: Some(tracker),
50            model_provider_pricing,
51            turn_usage: Arc::new(Mutex::new(TurnUsage::default())),
52            agent_alias: None,
53        }
54    }
55
56    /// Accumulation-only context: snapshots per-turn token usage without a
57    /// backing tracker. `record_tool_loop_cost_usage` skips persistence and
58    /// the missing-pricing warning (there is no cost enforcement to be
59    /// silently inert); `check_tool_loop_budget` reports no budget. Lets
60    /// wrappers that never tracked costs (e.g. `Agent::turn`) read summed
61    /// token totals out of the loop for observer events.
62    pub fn usage_only() -> Self {
63        Self {
64            tracker: None,
65            model_provider_pricing: Arc::new(ModelProviderPricing::new()),
66            turn_usage: Arc::new(Mutex::new(TurnUsage::default())),
67            agent_alias: None,
68        }
69    }
70
71    /// Attach an agent alias to this context so subsequent
72    /// `record_tool_loop_cost_usage` calls stamp records with it.
73    #[must_use]
74    pub fn with_agent_alias(mut self, agent_alias: impl Into<String>) -> Self {
75        self.agent_alias = Some(agent_alias.into());
76        self
77    }
78
79    /// Snapshot the per-scope usage. Wrapping code calls this after the
80    /// scoped future completes to populate observer-event annotations.
81    pub fn snapshot_turn_usage(&self) -> TurnUsage {
82        *self.turn_usage.lock()
83    }
84}

90/// Resolve `(input, output, cached_input)` per-1M-token rates for a given
91/// model on a model provider's pricing map. Lookup order:
92///
93/// 1. Dimension-specific keys: `{model}.input` / `{model}.output` /
94///    `{model}.cached_input`.
95/// 2. Bare model key as a flat fallback applied to whichever dimension
96///    didn't match in step 1.
97/// 3. The model alias path's last segment (`.../suffix`) tried under the
98///    same rules.
99///
100/// Returns `(0.0, 0.0, 0.0)` if no entry matches; the caller logs a
101/// one-shot warn in that case. A zero `cached_input` rate means "no
102/// discount" — the per-token caller bills the cached subset at the
103/// standard input rate.
104fn resolve_rates(pricing: &HashMap<String, f64>, model: &str) -> (f64, f64, f64) {
105    let try_lookup = |key: &str| -> Option<(Option<f64>, Option<f64>, Option<f64>)> {
106        let input = pricing.get(&format!("{key}.input")).copied();
107        let output = pricing.get(&format!("{key}.output")).copied();
108        let cached = pricing.get(&format!("{key}.cached_input")).copied();
109        let flat = pricing.get(key).copied();
110        if input.is_none() && output.is_none() && cached.is_none() && flat.is_none() {
111            None
112        } else {
113            Some((input.or(flat), output.or(flat), cached))
114        }
115    };
116
117    if let Some((input, output, cached)) = try_lookup(model) {
118        return (
119            input.unwrap_or(0.0),
120            output.unwrap_or(0.0),
121            cached.unwrap_or(0.0),
122        );
123    }
124    if let Some((_, suffix)) = model.rsplit_once('/')
125        && let Some((input, output, cached)) = try_lookup(suffix)
126    {
127        return (
128            input.unwrap_or(0.0),
129            output.unwrap_or(0.0),
130            cached.unwrap_or(0.0),
131        );
132    }
133    (0.0, 0.0, 0.0)
134}

136/// Resolve the per-model pricing map for a provider reference.
137///
138/// `model_provider_name` always arrives as the composite `<type>.<alias>`
139/// (see `agent_provider_composite`), but the outer pricing map may be keyed
140/// either way depending on which builder populated it: the CLI / cron / web
141/// agent loop keys by the composite alias, while the channel orchestrator keys
142/// by the bare provider `<type>` (rates are per provider type, not per alias).
143/// Try the composite verbatim first, then fall back to the bare type prefix so
144/// cost tracking resolves regardless of the builder — and so the type-keyed
145/// `cost.rates` sheet is honored on the alias paths too.
146fn provider_pricing<'a>(
147    map: &'a ModelProviderPricing,
148    model_provider_name: &str,
149) -> Option<&'a HashMap<String, f64>> {
150    map.get(model_provider_name).or_else(|| {
151        model_provider_name
152            .split_once('.')
153            .and_then(|(provider_type, _alias)| map.get(provider_type))
154    })
155}

157/// Record token usage from an LLM response via the task-local cost tracker.
158/// Returns `(total_tokens, cost_usd)` on success, `None` when not scoped or no usage.
159pub fn record_tool_loop_cost_usage(
160    model_provider_name: &str,
161    model: &str,
162    usage: &zeroclaw_providers::traits::TokenUsage,
163) -> Option<(u64, f64)> {
164    let input_tokens = usage.input_tokens.unwrap_or(0);
165    let output_tokens = usage.output_tokens.unwrap_or(0);
166    let cached_input_tokens = usage.cached_input_tokens.unwrap_or(0);
167    let total_tokens = input_tokens.saturating_add(output_tokens);
168    if total_tokens == 0 {
169        return None;
170    }
171
172    let ctx = TOOL_LOOP_COST_TRACKING_CONTEXT
173        .try_with(Clone::clone)
174        .ok()
175        .flatten()?;
176    let pricing = provider_pricing(&ctx.model_provider_pricing, model_provider_name);
177    let (input_rate, output_rate, cached_rate) = pricing
178        .map(|map| resolve_rates(map, model))
179        .unwrap_or((0.0, 0.0, 0.0));
180
181    let cost_usage = CostTokenUsage::new(
182        model,
183        input_tokens,
184        output_tokens,
185        cached_input_tokens,
186        input_rate,
187        output_rate,
188        cached_rate,
189    );
190
191    // Promote first sighting of (model_provider, model) without pricing to a WARN
192    // so operators notice the silent zero-cost record before they need to
193    // grep DEBUG logs. Subsequent sightings stay at DEBUG so the warn
194    // stream doesn't get spammy. Missing pricing means either the
195    // model_provider has no pricing map at all, or the map exists but
196    // produced zero rates for this model.
197    if ctx.tracker.is_some() && (pricing.is_none() || (input_rate == 0.0 && output_rate == 0.0)) {
198        warn_once_missing_pricing(model_provider_name, model);
199    }
200
201    if let Some(tracker) = &ctx.tracker
202        && let Err(error) =
203            tracker.record_usage_with_agent(cost_usage.clone(), ctx.agent_alias.as_deref())
204    {
205        ::zeroclaw_log::record!(WARN, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_outcome(::zeroclaw_log::EventOutcome::Unknown).with_attrs(::serde_json::json!({"model_provider": model_provider_name, "model": model, "error": format!("{}", error)})), "Failed to record cost tracking usage: ");
206    }
207
208    {
209        let mut usage = ctx.turn_usage.lock();
210        usage.input_tokens = usage.input_tokens.saturating_add(input_tokens);
211        usage.output_tokens = usage.output_tokens.saturating_add(output_tokens);
212        usage.cost_usd += cost_usage.cost_usd;
213    }
214
215    Some((cost_usage.total_tokens, cost_usage.cost_usd))
216}

218/// Insert `(model_provider, model)` into `seen`. Returns `true` on first sighting,
219/// `false` thereafter. Split out from `warn_once_missing_pricing` so the
220/// dedup contract can be unit-tested with a caller-owned set instead of the
221/// process-static one.
222fn missing_pricing_first_sighting(
223    seen: &Mutex<HashSet<(String, String)>>,
224    model_provider: &str,
225    model: &str,
226) -> bool {
227    seen.lock()
228        .insert((model_provider.to_string(), model.to_string()))
229}
230
231/// First-time WARN, subsequent DEBUG, per `(model_provider, model)` pair.
232///
233/// The default pricing catalog has no entries for most non-OpenAI/Anthropic/
234/// Google models. Operators only realize their cost-tracking surface is
235/// reporting zero when they happen to enable DEBUG logging — a pure-DEBUG
236/// signal is too quiet for "your cost enforcement is silently inert" to
237/// register. Promote the first sighting per-pair to WARN with a config-path
238/// pointer; all subsequent same-pair occurrences stay at DEBUG so the warn
239/// stream doesn't get spammy.
240fn warn_once_missing_pricing(model_provider: &str, model: &str) {
241    static SEEN: OnceLock<Mutex<HashSet<(String, String)>>> = OnceLock::new();
242    let seen = SEEN.get_or_init(|| Mutex::new(HashSet::new()));
243    if missing_pricing_first_sighting(seen, model_provider, model) {
244        ::zeroclaw_log::record!(
245            WARN,
246            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
247                .with_outcome(::zeroclaw_log::EventOutcome::Unknown)
248                .with_attrs(
249                    ::serde_json::json!({"model_provider": model_provider, "model": model})
250                ),
251            "Cost tracking: no pricing entry found for {model_provider}/{model} — \
252             token usage will be recorded with zero cost and budget enforcement \
253             is inert for this model. Add a `pricing` table to the model provider \
254             entry in config.toml (under `[providers.models.\"{model_provider}\"]`) \
255             with `\"{model}.input\"` and `\"{model}.output\"` keys (USD per 1M tokens). \
256             This warning fires once per (model_provider, model) pair per process."
257        );
258    } else {
259        ::zeroclaw_log::record!(
260            DEBUG,
261            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_attrs(
262                ::serde_json::json!({"model_provider": model_provider, "model": model})
263            ),
264            "Cost tracking recorded token usage with zero pricing (no pricing entry found)"
265        );
266    }
267}

269/// Check budget before an LLM call. Returns `None` when no cost tracking
270/// context is scoped (tests, delegate, CLI without cost config).
271pub fn check_tool_loop_budget() -> Option<BudgetCheck> {
272    TOOL_LOOP_COST_TRACKING_CONTEXT
273        .try_with(Clone::clone)
274        .ok()
275        .flatten()
276        .and_then(|ctx| {
277            ctx.tracker
278                .map(|tracker| tracker.check_budget(0.0).unwrap_or(BudgetCheck::Allowed))
279        })
280}

331        model_rates.insert("glm-5.1.output".to_string(), 4.4);
332
333        // CLI / agent-loop builder keys by the composite `<type>.<alias>`.
334        let mut composite_keyed: ModelProviderPricing = HashMap::new();
335        composite_keyed.insert("glm.default".to_string(), model_rates.clone());
336        assert!(
337            provider_pricing(&composite_keyed, "glm.default").is_some(),
338            "composite-keyed map must resolve via the verbatim composite lookup"
339        );
340
341        // Channel orchestrator builder keys by the bare provider `<type>`, yet
342        // the lookup still arrives as the composite alias — must fall back.
343        let mut type_keyed: ModelProviderPricing = HashMap::new();
344        type_keyed.insert("glm".to_string(), model_rates.clone());
345        assert!(
346            provider_pricing(&type_keyed, "glm.default").is_some(),
347            "type-keyed map must resolve the composite alias via the bare-type fallback"
348        );
349
350        // An unrelated provider must not accidentally match.
351        assert!(
352            provider_pricing(&type_keyed, "openai.default").is_none(),
353            "fallback must not resolve a provider type absent from the map"
354        );
355    }
