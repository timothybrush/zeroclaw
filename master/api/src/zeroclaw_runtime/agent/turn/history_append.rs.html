1//! History append for one tool round: the assistant message plus per-call
2//! `role=tool` messages (native) or a `[Tool results]` user message (prompt
3//! mode).
4
5use zeroclaw_providers::{ChatMessage, ToolCall};
6
7/// Append this round's assistant message and tool results to history
8/// (upstream loop body, history-append section).
9///
10/// Native mode uses JSON-structured messages so `convert_messages()` can
11/// reconstruct proper OpenAI-format `tool_calls` and tool result messages;
12/// prompt mode uses the XML-based text format as before.
13pub(crate) fn append_tool_round_to_history(
14    history: &mut Vec<ChatMessage>,
15    assistant_history_content: String,
16    native_tool_calls: &[ToolCall],
17    individual_results: &[(Option<String>, String)],
18    tool_results: &str,
19    use_native_tools: bool,
20) {
21    history.push(ChatMessage::assistant(assistant_history_content));
22    if native_tool_calls.is_empty() {
23        let all_results_have_ids = use_native_tools
24            && !individual_results.is_empty()
25            && individual_results
26                .iter()
27                .all(|(tool_call_id, _)| tool_call_id.is_some());
28        if all_results_have_ids {
29            for (tool_call_id, result) in individual_results {
30                let tool_msg = serde_json::json!({
31                    "tool_call_id": tool_call_id,
32                    "content": result,
33                });
34                history.push(ChatMessage::tool(tool_msg.to_string()));
35            }
36        } else {
37            history.push(ChatMessage::user(format!("[Tool results]\n{tool_results}")));
38        }
39    } else {
40        // `zip` would drop trailing results on any length divergence,
41        // leaving a native tool_use id with no matching tool_result.
42        // Pair on each result's own id instead.
43        for (idx, (tool_call_id, result)) in individual_results.iter().enumerate() {
44            let resolved_id = tool_call_id
45                .clone()
46                .or_else(|| native_tool_calls.get(idx).map(|call| call.id.clone()));
47            let tool_msg = serde_json::json!({
48                "tool_call_id": resolved_id,
49                "content": result,
50            });
51            history.push(ChatMessage::tool(tool_msg.to_string()));
52        }
53    }
54}
