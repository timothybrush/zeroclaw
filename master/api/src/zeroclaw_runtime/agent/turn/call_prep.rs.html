1//! The per-call preparation loop: `before_tool_call` hook, delivery defaults,
2//! the approval gate, the duplicate-call gate, and start logging — producing
3//! the executable subset of this round's tool calls.
4
5use super::approval_gate::{ApprovalGateOutcome, gate_tool_approval};
6use super::context::TurnCtx;
7use super::delivery_defaults::maybe_inject_channel_delivery_defaults;
8use super::events::{StreamDelta, emit_tool_call_pair};
9use super::redact::scrub_credentials;
10use crate::agent::tool_execution::ToolExecutionOutcome;
11use crate::util::truncate_with_ellipsis;
12use std::collections::HashSet;
13use std::time::Duration;
14use zeroclaw_tool_call_parser::{ParsedToolCall, canonicalize_json_for_tool_signature};
15
16/// The prepared subset of one round's tool calls.
17///
18/// `ordered_results` has one slot per incoming call; prep fills the slots for
19/// calls that never execute (hook-cancelled, denied, replaced, deduplicated)
20/// and post-exec fills the rest, so result ordering always matches the
21/// model's call ordering. `seen_tool_signatures` stays owned by the loop
22/// skeleton (reset per iteration) and is threaded in as `&mut`.
23pub(crate) struct PreparedToolCalls {
24    pub(crate) ordered_results: Vec<Option<(String, Option<String>, ToolExecutionOutcome)>>,
25    pub(crate) executable_indices: Vec<usize>,
26    pub(crate) executable_calls: Vec<ParsedToolCall>,
27}
28
29/// Run per-call preparation over this round's parsed tool calls (upstream
30/// loop body, per-call prep loop).
31pub(crate) async fn prepare_tool_calls(
32    ctx: &TurnCtx<'_>,
33    tool_calls: &[ParsedToolCall],
34    seen_tool_signatures: &mut HashSet<(String, String)>,
35    iteration: usize,
36    dedup_enabled: bool,
37) -> PreparedToolCalls {
38    let mut ordered_results: Vec<Option<(String, Option<String>, ToolExecutionOutcome)>> =
39        (0..tool_calls.len()).map(|_| None).collect();
40    let mut executable_indices: Vec<usize> = Vec::new();
41    let mut executable_calls: Vec<ParsedToolCall> = Vec::new();
42
43    for (idx, call) in tool_calls.iter().enumerate() {
44        // ── Hook: before_tool_call (modifying) ──────────
45        let mut tool_name = call.name.clone();
46        let mut tool_args = call.arguments.clone();
47        if let Some(hooks) = ctx.hooks {
48            match hooks
49                .run_before_tool_call(tool_name.clone(), tool_args.clone())
50                .await
51            {
52                crate::hooks::HookResult::Cancel(reason) => {
53                    ::zeroclaw_log::record!(INFO, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_attrs(::serde_json::json!({"tool": call.name, "reason": reason.to_string()})), "tool call cancelled by hook");
54                    let cancelled = format!("Cancelled by hook: {reason}");
55                    ::zeroclaw_log::record!(
56                        WARN,
57                        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Cancel)
58                            .with_outcome(::zeroclaw_log::EventOutcome::Failure)
59                            .with_attrs(::serde_json::json!({
60                                "model": ctx.model,
61                                "iteration": iteration + 1,
62                                "tool": call.name,
63                                "arguments": scrub_credentials(&tool_args.to_string()),
64                                "result": cancelled,
65                                "trace_id": ctx.turn_id,
66                            })),
67                        "tool_call_result"
68                    );
69                    if let Some(tx) = ctx.on_delta {
70                        let _ = tx
71                            .send(StreamDelta::Status(format!(
72                                "\u{274c} {}: {}\n",
73                                call.name,
74                                truncate_with_ellipsis(&scrub_credentials(&cancelled), 200)
75                            )))
76                            .await;
77                    }
78                    let outcome = ToolExecutionOutcome {
79                        output: cancelled,
80                        success: false,
81                        error_reason: Some(scrub_credentials(&reason)),
82                        duration: Duration::ZERO,
83                        receipt: None,
84                    };
85                    // Streaming consumers still see the call and its
86                    // hook-cancel outcome as a ToolCall/ToolResult pair,
87                    // as the direct execution path always emitted.
88                    if let Some(tx) = ctx.event_tx {
89                        emit_tool_call_pair(tx, call, &outcome).await;
90                    }
91                    ordered_results[idx] =
92                        Some((call.name.clone(), call.tool_call_id.clone(), outcome));
93                    continue;
94                }
95                crate::hooks::HookResult::Continue((name, args)) => {
96                    tool_name = name;
97                    tool_args = args;
98                }
99            }
100        }
101
102        maybe_inject_channel_delivery_defaults(
103            &tool_name,
104            &mut tool_args,
105            ctx.channel_name,
106            ctx.channel_reply_target,
107        );
108
109        crate::agent::set_runtime_approved_arg(&tool_name, &mut tool_args, false);
110
111        // ── Approval hook ────────────────────────────────
112        let approved = match gate_tool_approval(ctx, &tool_name, &tool_args, iteration).await {
113            ApprovalGateOutcome::Proceed { approved } => approved,
114            ApprovalGateOutcome::Deny(outcome) | ApprovalGateOutcome::Replace(outcome) => {
115                // Streaming consumers see the denied/replaced call and its
116                // synthesized result (e.g. a DenyWithEdit replacement) as a
117                // ToolCall/ToolResult pair, as the direct path always did.
118                if let Some(tx) = ctx.event_tx {
119                    emit_tool_call_pair(tx, call, &outcome).await;
120                }
121                ordered_results[idx] =
122                    Some((tool_name.clone(), call.tool_call_id.clone(), outcome));
123                continue;
124            }
125        };
126        crate::agent::set_runtime_approved_arg(&tool_name, &mut tool_args, approved);
127
128        let signature = {
129            let canonical_args = canonicalize_json_for_tool_signature(&tool_args);
130            let args_json =
131                serde_json::to_string(&canonical_args).unwrap_or_else(|_| "{}".to_string());
132            (tool_name.trim().to_ascii_lowercase(), args_json)
133        };
134        let dedup_exempt = ctx.dedup_exempt_tools.iter().any(|e| e == &tool_name)
135            || crate::tools::REENTRANT_AGENT_TOOLS.contains(&tool_name.as_str());
136        if dedup_enabled && !dedup_exempt && !seen_tool_signatures.insert(signature) {
137            let duplicate = format!(
138                "Skipped duplicate tool call '{tool_name}' with identical arguments in this turn."
139            );
140            ::zeroclaw_log::record!(
141                INFO,
142                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Skip)
143                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
144                    .with_attrs(::serde_json::json!({
145                        "model": ctx.model,
146                        "iteration": iteration + 1,
147                        "tool": tool_name.clone(),
148                        "arguments": scrub_credentials(&tool_args.to_string()),
149                        "result": duplicate,
150                        "deduplicated": true,
151                        "trace_id": ctx.turn_id,
152                    })),
153                "tool_call_result"
154            );
155            if let Some(tx) = ctx.on_delta {
156                let _ = tx
157                    .send(StreamDelta::Status(format!(
158                        "\u{274c} {}: {}\n",
159                        tool_name, duplicate
160                    )))
161                    .await;
162            }
163            ordered_results[idx] = Some((
164                tool_name.clone(),
165                call.tool_call_id.clone(),
166                ToolExecutionOutcome {
167                    output: duplicate.clone(),
168                    success: false,
169                    error_reason: Some(duplicate),
170                    duration: Duration::ZERO,
171                    receipt: None,
172                },
173            ));
174            continue;
175        }
176
177        ::zeroclaw_log::record!(
178            INFO,
179            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Start).with_attrs(
180                ::serde_json::json!({
181                    "model": ctx.model,
182                    "iteration": iteration + 1,
183                    "tool": tool_name.clone(),
184                    "arguments": scrub_credentials(&tool_args.to_string()),
185                    "trace_id": ctx.turn_id,
186                })
187            ),
188            "tool_call_start"
189        );
190
191        // ── Progress: tool start ────────────────────────────
192        if let Some(tx) = ctx.on_delta {
193            let hint = {
194                let raw = match tool_name.as_str() {
195                    "shell" => tool_args.get("command").and_then(|v| v.as_str()),
196                    "file_read" | "file_write" => tool_args.get("path").and_then(|v| v.as_str()),
197                    _ => tool_args
198                        .get("action")
199                        .and_then(|v| v.as_str())
200                        .or_else(|| tool_args.get("query").and_then(|v| v.as_str())),
201                };
202                match raw {
203                    Some(s) => truncate_with_ellipsis(s, 60),
204                    None => String::new(),
205                }
206            };
207            let progress = if hint.is_empty() {
208                format!("\u{23f3} {}\n", tool_name)
209            } else {
210                format!("\u{23f3} {}: {hint}\n", tool_name)
211            };
212            ::zeroclaw_log::record!(
213                DEBUG,
214                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
215                    .with_attrs(::serde_json::json!({"tool": tool_name})),
216                "Sending progress start to draft"
217            );
218            let _ = tx.send(StreamDelta::Status(progress)).await;
219        }
220
221        executable_indices.push(idx);
222        executable_calls.push(ParsedToolCall {
223            name: tool_name,
224            arguments: tool_args,
225            tool_call_id: call.tool_call_id.clone(),
226        });
227    }
228
229    PreparedToolCalls {
230        ordered_results,
231        executable_indices,
232        executable_calls,
233    }
234}
