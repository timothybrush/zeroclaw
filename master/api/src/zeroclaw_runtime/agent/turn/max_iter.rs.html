1//! The max-iteration exit: when the loop exhausts its iterations, ask the
2//! LLM for a tools-free final summary (with step timeout + cancel select)
3//! and return it appended to the accumulated display text, or bail.
4
5use super::knobs::{LoopKnobs, MaxIterationBehavior};
6use super::outcome::ToolLoopCancelled;
7use anyhow::Result;
8use std::time::Duration;
9use tokio_util::sync::CancellationToken;
10use zeroclaw_config::schema::PacingConfig;
11use zeroclaw_providers::{ChatMessage, ModelProvider};
12
13/// Graceful shutdown after the loop exhausts `max_iterations` (upstream loop
14/// body, max-iteration exit): log exhaustion, push a summary-request user
15/// message, make a tools-free `chat` call honoring `pacing.step_timeout_secs`
16/// and the cancellation token, and return `Ok(accumulated + summary)` — or
17/// bail with "exceeded maximum tool iterations" when the summary is empty or
18/// the call fails.
19#[allow(clippy::too_many_arguments)]
20pub(crate) async fn finish_after_max_iterations(
21    model_provider: &dyn ModelProvider,
22    history: &mut Vec<ChatMessage>,
23    provider_name: &str,
24    model: &str,
25    temperature: Option<f64>,
26    pacing: &PacingConfig,
27    cancellation_token: Option<&CancellationToken>,
28    max_iterations: usize,
29    mut accumulated_display_text: String,
30    turn_id: &str,
31    knobs: &LoopKnobs,
32    mut new_messages_out: Option<&mut Vec<ChatMessage>>,
33) -> Result<String> {
34    ::zeroclaw_log::record!(
35        WARN,
36        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
37            .with_outcome(::zeroclaw_log::EventOutcome::Failure)
38            .with_attrs(::serde_json::json!({
39                "model": model,
40                "max_iterations": max_iterations,
41                "trace_id": turn_id,
42            })),
43        "tool_loop_exhausted"
44    );
45
46    // ErrorAtCap callers (embedders driving Agent::turn) treat the cap as a
47    // control signal: bail instead of spending another LLM call on a summary.
48    if knobs.max_iteration_behavior == MaxIterationBehavior::ErrorAtCap {
49        anyhow::bail!("Agent exceeded maximum tool iterations ({max_iterations})")
50    }
51
52    // Graceful shutdown: ask the LLM for a final summary without tools
53    ::zeroclaw_log::record!(
54        WARN,
55        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
56            .with_outcome(::zeroclaw_log::EventOutcome::Unknown)
57            .with_attrs(::serde_json::json!({"max_iterations": max_iterations})),
58        "Max iterations reached, requesting final summary"
59    );
60    let summary_prompt = ChatMessage::user(
61        "You have reached the maximum number of tool iterations. \
62         Please provide your best answer based on the work completed so far. \
63         Summarize what you accomplished and what remains to be done."
64            .to_string(),
65    );
66    // Pushed into history for the request below, but mirrored into the
67    // append-log (and kept in history) only when the summary call SUCCEEDS:
68    // a failed/cancelled/timed-out/empty summary must not persist an
69    // unanswered synthetic prompt into wrapper transcripts — every failure
70    // exit pops it back off.
71    let summary_prompt_mirror = summary_prompt.clone();
72    history.push(summary_prompt);
73
74    enum SummaryCall {
75        Cancelled,
76        TimedOut(u64),
77        Done(Result<zeroclaw_providers::ChatResponse>),
78    }
79    let summary_call = {
80        let summary_request = zeroclaw_providers::ChatRequest {
81            messages: history,
82            tools: None, // No tools — force a text response
83            thinking: zeroclaw_api::NATIVE_THINKING_OVERRIDE
84                .try_with(Clone::clone)
85                .ok()
86                .flatten(),
87        };
88        let summary_future = model_provider.chat(summary_request, model, temperature);
89        match pacing.step_timeout_secs {
90            Some(step_secs) if step_secs > 0 => {
91                let step_timeout = Duration::from_secs(step_secs);
92                if let Some(token) = cancellation_token {
93                    tokio::select! {
94                        () = token.cancelled() => SummaryCall::Cancelled,
95                        result = tokio::time::timeout(step_timeout, summary_future) => match result {
96                            Ok(inner) => SummaryCall::Done(inner),
97                            Err(_) => SummaryCall::TimedOut(step_secs),
98                        },
99                    }
100                } else {
101                    match tokio::time::timeout(step_timeout, summary_future).await {
102                        Ok(inner) => SummaryCall::Done(inner),
103                        Err(_) => SummaryCall::TimedOut(step_secs),
104                    }
105                }
106            }
107            _ => {
108                if let Some(token) = cancellation_token {
109                    tokio::select! {
110                        () = token.cancelled() => SummaryCall::Cancelled,
111                        result = summary_future => SummaryCall::Done(result),
112                    }
113                } else {
114                    SummaryCall::Done(summary_future.await)
115                }
116            }
117        }
118    };
119
120    let resp = match summary_call {
121        SummaryCall::Cancelled => {
122            history.pop();
123            return Err(ToolLoopCancelled.into());
124        }
125        SummaryCall::TimedOut(step_secs) => {
126            history.pop();
127            anyhow::bail!("Final summary LLM call timed out after {step_secs}s (step_timeout_secs)")
128        }
129        SummaryCall::Done(Err(e)) => {
130            ::zeroclaw_log::record!(
131                ERROR,
132                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
133                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
134                    .with_attrs(::serde_json::json!({
135                        "model": model,
136                        "provider": provider_name,
137                        "max_iterations": max_iterations,
138                        "trace_id": turn_id,
139                        "error": format!("{e}"),
140                    })),
141                "final summary LLM call failed after iteration exhaustion; bailing"
142            );
143            history.pop();
144            anyhow::bail!("Agent exceeded maximum tool iterations ({max_iterations})")
145        }
146        SummaryCall::Done(Ok(resp)) => resp,
147    };
148
149    let text = resp.text.unwrap_or_default();
150    if text.is_empty() {
151        history.pop();
152        anyhow::bail!("Agent exceeded maximum tool iterations ({max_iterations})")
153    }
154    // Persist the answered prompt + summary like every other final assistant
155    // response: without the summary message, persistent-history callers (the
156    // streamed wrapper's replay, new_messages consumers) store a transcript
157    // ending on the synthetic user prompt with no answer — the delivered
158    // summary would be absent and the model re-answers the synthetic prompt
159    // next turn.
160    let summary_msg = ChatMessage::assistant(text.clone());
161    if let Some(out) = &mut new_messages_out {
162        out.push(summary_prompt_mirror);
163        out.push(summary_msg.clone());
164    }
165    history.push(summary_msg);
166    accumulated_display_text.push_str(&text);
167    Ok(accumulated_display_text)
168}
