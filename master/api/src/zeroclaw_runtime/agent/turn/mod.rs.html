1//! The agent turn engine, decomposed into single-purpose step modules.
2//!
3//! Public paths are unchanged: external code keeps importing via
4//! `crate::agent::loop_::*` (re-export block there). Full contract manifest:
5//! `/opt/notes/work/zeroclaw/unification_modular/RUN_SHEET.md` (condensed
6//! copy below) during the #7415 migration.
7//!
8//! # Run sheet (condensed)
9//!
10//! [`run_tool_call_loop`] is the orchestrator: loop control only, no step
11//! logic. [`TurnCtx`] carries shared refs ONLY; every `&mut` the loop owns
12//! (history, loop detector, `seen_tool_signatures` — reset per iteration —
13//! identical-output counters, accumulated display text, malformed-retry
14//! counter) stays a loop local passed as an explicit argument.
15//!
16//! Per-iteration step sequence:
17//!
18//! ```text
19//! preflight_history_maintenance(&mut history)            history_window.rs
20//! → [model-switch check — inline]
21//! → build_iteration_tool_specs(..)                       tool_specs.rs
22//! → resolve_vision_provider(..)                          vision_route.rs
23//! → [active triple derivation — inline]
24//! → prepare_messages_for_iteration(..)                   vision_route.rs
25//! → announce_llm_request(ctx, ..) -> Instant             provider_call.rs
26//! → enforce_tool_loop_budget()                           provider_call.rs
27//! → call_provider(ctx, ..) -> ProviderCallOutcome        provider_call.rs
28//!     [streaming: consume_provider_streaming_response]   stream_consume.rs
29//!     [cancel asymmetry: see ProviderCallOutcome docs]
30//! → Ok:  interpret_chat_response(ctx, ..)                parse_response.rs
31//!   Err: record_llm_failure(..);                         context_recovery.rs
32//!        try_recover_context_overflow(..) -> bool (true ⇒ continue)
33//! → resolve_display_text / [malformed-retry — inline]
34//!   / [no-tool exit — inline, stream_text_posthoc_chunks → events.rs]
35//! → prepare_tool_calls(ctx, &mut seen, ..)               call_prep.rs
36//!     [approval via gate_tool_approval]                  approval_gate.rs
37//! → [execute dispatch — inline → tool_execution::execute_tools_{parallel,sequential}]
38//! → record_executed_outcomes(ctx, .., &mut ordered)      post_exec.rs
39//! → collect_tool_results(..) -> CollectedResults         results_collect.rs
40//! → check_identical_output_abort(.., &mut counters)      results_collect.rs
41//! → append_tool_round_to_history(&mut history, ..)       history_append.rs
42//! [loop exhausted] → finish_after_max_iterations(..)     max_iter.rs
43//! ```
44//!
45//! Leaf/type modules: `context` (TurnCtx), `events` (StreamDelta/DraftEvent,
46//! pacing consts, post-hoc chunker), `outcome` (ToolLoopCancelled,
47//! ModelSwitchRequested), `redact` (scrub_credentials), `stream_guard` +
48//! `protocol_detect` (streaming protocol suppression), `delivery_defaults`.

123/// Maximum malformed internal tool-protocol retries before returning a safe fallback.
124pub(crate) const MAX_MALFORMED_TOOL_PROTOCOL_RETRIES: usize = 2;

126/// Default maximum agentic tool-use iterations per user message to prevent runaway loops.
127/// Used as a safe fallback when `max_tool_iterations` is unset or configured as zero.
128pub(crate) const DEFAULT_MAX_TOOL_ITERATIONS: usize = 10;
129
130// ── Agent Tool-Call Loop ──────────────────────────────────────────────────
131// Core agentic iteration: send conversation to the LLM, parse any tool
132// calls from the response, execute them, append results to history, and
133// repeat until the LLM produces a final text-only answer.
134//
135// Loop invariant: at the start of each iteration, `history` contains the
136// full conversation so far (system prompt + user messages + prior tool
137// results). The loop exits when:
138//   • the LLM returns no tool calls (final answer), or
139//   • max_iterations is reached (runaway safety), or
140//   • the cancellation token fires (external abort).
141
142/// Execute a single turn of the agent loop: send messages, parse tool calls,
143/// execute tools, and loop until the LLM produces a final text response.
144///
145/// `new_messages_out` is an append-log: every message the loop adds to
146/// `history` is mirrored into it at push time (a clone taken before any
147/// later in-loop history maintenance), so it is populated on **every** exit
148/// — success, error, and cancellation — and never derived from history
149/// indices, which in-loop pruning can invalidate. Loop-detection system
150/// notes are the one exception (merged into the existing system message;
151/// only reachable when pattern loop detection is enabled, which no
152/// `new_messages_out` consumer turns on).
153#[allow(clippy::too_many_arguments)]
154pub async fn run_tool_call_loop(
155    model_provider: &dyn ModelProvider,
156    history: &mut Vec<ChatMessage>,
157    tools_registry: &[Box<dyn Tool>],
158    observer: &dyn Observer,
159    provider_name: &str,
160    model: &str,
161    temperature: Option<f64>,
162    silent: bool,
163    approval: Option<&ApprovalManager>,
164    channel_name: &str,
165    channel_reply_target: Option<&str>,
166    multimodal_config: &zeroclaw_config::schema::MultimodalConfig,
167    max_tool_iterations: usize,
168    cancellation_token: Option<CancellationToken>,
169    on_delta: Option<tokio::sync::mpsc::Sender<DraftEvent>>,
170    hooks: Option<&crate::hooks::HookRunner>,
171    excluded_tools: &[String],
172    dedup_exempt_tools: &[String],
173    activated_tools: Option<&std::sync::Arc<std::sync::Mutex<crate::tools::ActivatedToolSet>>>,
174    model_switch_callback: Option<ModelSwitchCallback>,
175    pacing: &zeroclaw_config::schema::PacingConfig,
176    strict_tool_parsing: bool,
177    parallel_tools: bool,
178    max_tool_result_chars: usize,
179    context_token_budget: usize,
180    shared_budget: Option<Arc<std::sync::atomic::AtomicUsize>>,
181    channel: Option<&dyn Channel>,
182    receipt_generator: Option<&crate::agent::tool_receipts::ReceiptGenerator>,
183    collected_receipts: Option<&std::sync::Mutex<Vec<String>>>,
184    event_tx: Option<tokio::sync::mpsc::Sender<TurnEvent>>,
185    mut steering: Option<&mut tokio::sync::mpsc::Receiver<String>>,
186    mut new_messages_out: Option<&mut Vec<ChatMessage>>,
187    knobs: &LoopKnobs,
188    mut image_cache: Option<&mut zeroclaw_providers::multimodal::LocalImageCache>,
189) -> Result<String> {
190    let max_iterations = if max_tool_iterations == 0 {
191        DEFAULT_MAX_TOOL_ITERATIONS
192    } else {
193        max_tool_iterations
194    };
195
196    let turn_id = Uuid::new_v4().to_string();
197    let loop_started_at = Instant::now();
198    let loop_ignore_tools: HashSet<&str> = pacing
199        .loop_ignore_tools
200        .iter()
201        .map(String::as_str)
202        .collect();
203    let mut consecutive_identical_outputs: usize = 0;
204    let mut last_tool_output_hash: Option<u64> = None;
205
206    let mut loop_detector = crate::agent::loop_detector::LoopDetector::new(
207        crate::agent::loop_detector::LoopDetectorConfig {
208            enabled: pacing.loop_detection_enabled,
209            window_size: pacing.loop_detection_window_size,
210            max_repeats: pacing.loop_detection_max_repeats,
211        },
212    );
213
214    // Accumulated display text across all tool-loop calls.
215    let mut accumulated_display_text = String::new();
216    let mut malformed_tool_protocol_retries: usize = 0;
217
218    // Shared-ref context for the turn step functions. Every `&mut` the loop
219    // owns stays a loop local passed as an explicit argument (RUN_SHEET
220    // `turn.context.TurnCtx`).
221    let ctx = TurnCtx {
222        observer,
223        provider_name,
224        model,
225        temperature,
226        approval,
227        channel_name,
228        channel_reply_target,
229        cancellation_token: cancellation_token.as_ref(),
230        on_delta: on_delta.as_ref(),
231        event_tx: event_tx.as_ref(),
232        hooks,
233        dedup_exempt_tools,
234        pacing,
235        strict_tool_parsing,
236        channel,
237        turn_id: &turn_id,
238    };
239
240    for iteration in 0..max_iterations {
241        // Steering: fold caller-pushed mid-turn messages into history before
242        // this iteration's provider request.
243        for steering_message in drain_steering_messages(&mut steering) {
244            let msg = ChatMessage::user(steering_message);
245            if let Some(out) = new_messages_out.as_deref_mut() {
246                out.push(msg.clone());
247            }
248            history.push(msg);
249        }
250
251        let mut seen_tool_signatures: HashSet<(String, String)> = HashSet::new();
252
253        if cancellation_token
254            .as_ref()
255            .is_some_and(CancellationToken::is_cancelled)
256        {
257            return Err(ToolLoopCancelled.into());
258        }
259
260        // Shared iteration budget: parent + subagents share a global counter
261        if let Some(ref budget) = shared_budget {
262            let remaining = budget.load(std::sync::atomic::Ordering::Relaxed);
263            if remaining == 0 {
264                ::zeroclaw_log::record!(
265                    WARN,
266                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
267                        .with_outcome(::zeroclaw_log::EventOutcome::Unknown)
268                        .with_attrs(::serde_json::json!({"iteration": iteration})),
269                    "Shared iteration budget exhausted at iteration "
270                );
271                break;
272            }
273            budget.fetch_sub(1, std::sync::atomic::Ordering::Relaxed);
274        }
275
276        preflight_history_maintenance(history, context_token_budget, iteration);
277
278        // Check if model switch was requested via model_switch tool
279        if let Some(ref callback) = model_switch_callback
280            && let Ok(guard) = callback.lock()
281            && let Some((new_model_provider, new_model)) = guard.as_ref()
282            && (new_model_provider != provider_name || new_model != model)
283        {
284            ::zeroclaw_log::record!(
285                INFO,
286                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note),
287                &format!(
288                    "Model switch detected: {} {} -> {} {}",
289                    provider_name, model, new_model_provider, new_model
290                )
291            );
292            return Err(ModelSwitchRequested {
293                model_provider: new_model_provider.clone(),
294                model: new_model.clone(),
295            }
296            .into());
297        }
298
299        let iteration_tool_specs = build_iteration_tool_specs(
300            model_provider,
301            tools_registry,
302            excluded_tools,
303            activated_tools,
304        );
305        let IterationToolSpecs {
306            ref tool_specs,
307            use_native_tools,
308            ..
309        } = iteration_tool_specs;
310
311        let (vision_model_provider_box, degrade_strip_images) =
312            resolve_vision_provider(model_provider, history, multimodal_config, provider_name)?;
313
314        let (active_model_provider, active_model_provider_name, active_model): (
315            &dyn ModelProvider,
316            &str,
317            &str,
318        ) = if let Some(ref vp_box) = vision_model_provider_box {
319            let vp_name = multimodal_config
320                .vision_model_provider
321                .as_deref()
322                .unwrap_or(provider_name);
323            let vm = multimodal_config.vision_model.as_deref().unwrap_or(model);
324            (vp_box.as_ref(), vp_name, vm)
325        } else {
326            (model_provider, provider_name, model)
327        };
328
329        let prepared_messages = prepare_messages_for_iteration(
330            history,
331            multimodal_config,
332            degrade_strip_images,
333            image_cache.as_deref_mut(),
334        )
335        .await?;
336
337        let llm_started_at = announce_llm_request(
338            &ctx,
339            history,
340            active_model_provider,
341            active_model_provider_name,
342            active_model,
343            iteration,
344        )
345        .await;
346
347        enforce_tool_loop_budget()?;
348
349        // Unified path via ModelProvider::chat so provider-specific native tool logic
350        // (OpenAI/Anthropic/OpenRouter/compatible adapters) is honored.
351        let request_tools = if use_native_tools {
352            Some(tool_specs.as_slice())
353        } else {
354            None
355        };
356        let should_consume_provider_stream = (on_delta.is_some() || event_tx.is_some())
357            && model_provider.supports_streaming()
358            && (request_tools.is_none() || model_provider.supports_streaming_tool_events());
359        ::zeroclaw_log::record!(DEBUG, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_attrs(::serde_json::json!({"has_on_delta": on_delta.is_some(), "has_event_tx": event_tx.is_some(), "supports_streaming": model_provider.supports_streaming(), "should_consume_provider_stream": should_consume_provider_stream})), &format!("Streaming decision for iteration {}", iteration + 1));
360
361        let ProviderCallOutcome {
362            chat_result,
363            streamed_live_deltas,
364            streamed_protocol_suppressed,
365        } = call_provider(
366            &ctx,
367            active_model_provider,
368            active_model,
369            &prepared_messages.messages,
370            request_tools,
371            should_consume_provider_stream,
372            iteration,
373        )
374        .await?;
375
376        let (
377            response_text,
378            parsed_text,
379            tool_calls,
380            assistant_history_content,
381            native_tool_calls,
382            parse_issue_detected,
383            protocol_suppressed,
384            response_streamed_live,
385        ) = match chat_result {
386            Ok(resp) => {
387                let interpreted = interpret_chat_response(
388                    &ctx,
389                    resp,
390                    &iteration_tool_specs,
391                    streamed_protocol_suppressed,
392                    llm_started_at,
393                    iteration,
394                    knobs.detect_protocol_without_tools,
395                )
396                .await;
397                (
398                    interpreted.response_text,
399                    interpreted.parsed_text,
400                    interpreted.tool_calls,
401                    interpreted.assistant_history_content,
402                    interpreted.native_tool_calls,
403                    interpreted.parse_issue_detected,
404                    streamed_protocol_suppressed,
405                    streamed_live_deltas,
406                )
407            }
408            Err(e) => {
409                record_llm_failure(
410                    observer,
411                    provider_name,
412                    model,
413                    llm_started_at,
414                    iteration,
415                    &turn_id,
416                    &e,
417                );
418                let recovered = try_recover_context_overflow(history, &e, iteration);
419                if recovered {
420                    continue;
421                }
422                // A stream that died after caller-visible output: persist the
423                // partial with the interruption marker so wrappers/channels
424                // can commit what the consumer already saw.
425                if let Some(interrupted) = e.downcast_ref::<outcome::StreamInterruptedAfterOutput>()
426                    && !interrupted.partial_text.is_empty()
427                {
428                    let msg = ChatMessage::assistant(format!(
429                        "{}\n\n{}",
430                        interrupted.partial_text,
431                        crate::i18n::get_required_cli_string("turn-stream-interrupted")
432                    ));
433                    if let Some(out) = new_messages_out.as_deref_mut() {
434                        out.push(msg.clone());
435                    }
436                    history.push(msg);
437                }
438                // Same for a user cancel after visible streamed output —
439                // the pre-consolidation streaming engine committed the
440                // watched partial with this exact marker.
441                if let Some(cancelled) = e.downcast_ref::<outcome::StreamCancelledAfterOutput>()
442                    && !cancelled.partial_text.is_empty()
443                {
444                    let msg = ChatMessage::assistant(format!(
445                        "{}\n\n{}",
446                        cancelled.partial_text,
447                        crate::i18n::get_required_cli_string("turn-interrupted-by-user")
448                    ));
449                    if let Some(out) = new_messages_out.as_deref_mut() {
450                        out.push(msg.clone());
451                    }
452                    history.push(msg);
453                }
454                return Err(e);
455            }
456        };
457
458        let display_text = resolve_display_text(
459            &response_text,
460            &parsed_text,
461            !tool_calls.is_empty(),
462            !native_tool_calls.is_empty(),
463        );
464
465        // Native provider tool_calls are converted into parsed `tool_calls`
466        // above; if this branch is reached there is no valid native call to run.
467        if tool_calls.is_empty() && parse_issue_detected {
468            malformed_tool_protocol_retries += 1;
469            ::zeroclaw_log::record!(
470                WARN,
471                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
472                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
473                    .with_attrs(serde_json::json!({
474                        "channel": channel_name,
475                        "model_provider": provider_name,
476                        "model": model,
477                        "trace_id": turn_id,
478                        "error": "malformed internal tool protocol omitted from channel output",
479                    })),
480                "tool_call_parse_feedback"
481            );
482            ::zeroclaw_log::record!(
483                DEBUG,
484                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
485                    .with_attrs(serde_json::json!({
486                    "iteration": iteration + 1,
487                    "retry": malformed_tool_protocol_retries,
488                    "max_retries": MAX_MALFORMED_TOOL_PROTOCOL_RETRIES,
489                    "response_excerpt": truncate_with_ellipsis(
490                        &scrub_credentials(&response_text),
491                        600
492                    ),
493                    })),
494                "tool_call_parse_feedback_details"
495            );
496
497            if malformed_tool_protocol_retries <= MAX_MALFORMED_TOOL_PROTOCOL_RETRIES {
498                // This is model feedback, not a tool result: malformed protocol
499                // output has no valid tool_call_id to attach a role=tool message to.
500                let msg = ChatMessage::user(
501                    "[Tool call parse error]\n\
502                     Your previous response looked like an internal tool-call protocol payload, \
503                     but ZeroClaw could not parse it into a valid tool call. Use the supported \
504                     tool-call schema, or answer in natural language if no tool is needed."
505                        .to_string(),
506                );
507                if let Some(out) = new_messages_out.as_deref_mut() {
508                    out.push(msg.clone());
509                }
510                history.push(msg);
511                continue;
512            }
513
514            let fallback =
515                crate::i18n::get_required_cli_string("channel-runtime-malformed-tool-output");
516            accumulated_display_text.push_str(&fallback);
517            if let Some(ref tx) = on_delta {
518                let _ = tx.send(StreamDelta::Text(fallback.to_string())).await;
519            }
520            let msg = ChatMessage::assistant(fallback.to_string());
521            if let Some(out) = new_messages_out.as_deref_mut() {
522                out.push(msg.clone());
523            }
524            history.push(msg);
525            return Ok(accumulated_display_text);
526        }
527
528        // ── Progress: LLM responded ─────────────────────────────
529        if let Some(ref tx) = on_delta {
530            let llm_secs = llm_started_at.elapsed().as_secs();
531            if !tool_calls.is_empty() {
532                let _ = tx
533                    .send(StreamDelta::Status(format!(
534                        "\u{1f4ac} Got {} tool call(s) ({llm_secs}s)\n",
535                        tool_calls.len()
536                    )))
537                    .await;
538            }
539        }
540
541        if tool_calls.is_empty() {
542            ::zeroclaw_log::record!(
543                INFO,
544                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Complete)
545                    .with_outcome(::zeroclaw_log::EventOutcome::Success)
546                    .with_attrs(::serde_json::json!({
547                        "model": model,
548                        "iteration": iteration + 1,
549                        "text": scrub_credentials(&display_text),
550                        "trace_id": turn_id,
551                    })),
552                "turn_final_response"
553            );
554            // No tool calls — this is the final response.
555            accumulated_display_text.push_str(&display_text);
556
557            // If text wasn't streamed live, send it now post-hoc. Gated on
558            // event_tx independently of on_delta (never nested — §8.4).
559            if !response_streamed_live && !protocol_suppressed {
560                events::emit_posthoc_turn_chunk(event_tx.as_ref(), &display_text).await;
561            }
562
563            // If text wasn't streamed live, send it now via post-hoc chunking.
564            // When streamed live, the channel already received the deltas.
565            if let Some(ref tx) = on_delta
566                && !response_streamed_live
567                && !protocol_suppressed
568            {
569                events::stream_text_posthoc_chunks(tx, &display_text, cancellation_token.as_ref())
570                    .await?;
571            }
572
573            let msg = ChatMessage::assistant(response_text.clone());
574            if let Some(out) = new_messages_out.as_deref_mut() {
575                out.push(msg.clone());
576            }
577            history.push(msg);
578            return Ok(accumulated_display_text);
579        }
580
581        // Do not accumulate intermediate-turn display text into the final
582        // channel response. Native tool-call providers may emit narration or
583        // scratchpad-like text alongside tool calls; draft-capable channels
584        // can still see it live through `on_delta` below, but the final
585        // delivered response must only contain the final assistant turn.
586
587        // Native tool-call model_providers can return assistant text separately from
588        // the structured call payload; relay it to draft-capable channels.
589        if !display_text.is_empty() {
590            if !native_tool_calls.is_empty()
591                && let Some(ref tx) = on_delta
592            {
593                let mut narration = display_text.clone();
594                if !narration.ends_with('\n') {
595                    narration.push('\n');
596                }
597                let _ = tx.send(StreamDelta::Text(narration)).await;
598            }
599            if !silent {
600                print!("{display_text}");
601                let _ = std::io::stdout().flush();
602            }
603        }
604
605        // When multiple tool calls are present and interactive CLI approval is not needed, run
606        // tool executions concurrently for lower wall-clock latency.
607        let allow_parallel_execution =
608            parallel_tools && should_execute_tools_in_parallel(&tool_calls, approval);
609        let PreparedToolCalls {
610            mut ordered_results,
611            executable_indices,
612            executable_calls,
613        } = prepare_tool_calls(
614            &ctx,
615            &tool_calls,
616            &mut seen_tool_signatures,
617            iteration,
618            knobs.dedup_enabled,
619        )
620        .await;
621
622        let execution_result = if allow_parallel_execution && executable_calls.len() > 1 {
623            execute_tools_parallel(
624                &executable_calls,
625                tools_registry,
626                activated_tools,
627                observer,
628                cancellation_token.as_ref(),
629                receipt_generator,
630            )
631            .await
632        } else {
633            execute_tools_sequential(
634                &executable_calls,
635                tools_registry,
636                activated_tools,
637                observer,
638                cancellation_token.as_ref(),
639                receipt_generator,
640            )
641            .await
642        };
643        let executed_outcomes = match execution_result {
644            Ok(outcomes) => outcomes,
645            // Cancelled mid-batch (parallel path): no per-call outcomes
646            // survive; every call synthesizes as interrupted below.
647            Err(e) if is_tool_loop_cancelled(&e) => Vec::new(),
648            Err(e) => return Err(e),
649        };
650
651        // Cancelled mid-batch: the round still persists atomically below
652        // (assistant tool-call message + per-call results, completed
653        // outcomes kept, never-ran calls synthesized as interrupted —
654        // #1043 semantics), and the completed prefix is recorded exactly
655        // like a finished batch: those tools RAN, so their TurnEvent
656        // pairs, `after_tool_call` hooks, and result logs must fire even
657        // though the cancellation surfaces right after.
658        let cancelled_mid_batch = executed_outcomes.len() < executable_calls.len();
659
660        // Record the completed outcomes (the full set when the batch
661        // finished; the executed prefix when cancelled mid-batch — the
662        // sequential executor returns completed outcomes in call order).
663        let completed = executed_outcomes.len();
664        record_executed_outcomes(
665            &ctx,
666            &executable_indices[..completed],
667            &executable_calls[..completed],
668            executed_outcomes,
669            &mut ordered_results,
670            iteration,
671        )
672        .await;
673        if cancelled_mid_batch {
674            for (idx, call) in tool_calls.iter().enumerate() {
675                if ordered_results[idx].is_none() {
676                    ordered_results[idx] = Some((
677                        call.name.clone(),
678                        call.tool_call_id.clone(),
679                        crate::agent::tool_execution::ToolExecutionOutcome {
680                            output: crate::i18n::get_required_cli_string(
681                                "turn-tool-interrupted-before-result",
682                            ),
683                            success: false,
684                            error_reason: None,
685                            duration: std::time::Duration::ZERO,
686                            receipt: None,
687                        },
688                    ));
689                }
690            }
691        }
692
693        let CollectedResults {
694            individual_results,
695            tool_results,
696            detection_relevant_output,
697        } = collect_tool_results(
698            ordered_results,
699            &tool_calls,
700            history,
701            &mut loop_detector,
702            &loop_ignore_tools,
703            max_tool_result_chars,
704            collected_receipts,
705            model,
706            iteration,
707            &turn_id,
708        )?;
709
710        if !cancelled_mid_batch {
711            check_identical_output_abort(
712                &detection_relevant_output,
713                loop_started_at,
714                pacing,
715                &mut consecutive_identical_outputs,
716                &mut last_tool_output_hash,
717                model,
718                iteration,
719                &turn_id,
720            )?;
721        }
722
723        let appended_from = history.len();
724        append_tool_round_to_history(
725            history,
726            assistant_history_content,
727            &native_tool_calls,
728            &individual_results,
729            &tool_results,
730            use_native_tools,
731        );
732        if let Some(out) = new_messages_out.as_deref_mut() {
733            out.extend_from_slice(&history[appended_from..]);
734        }
735
736        if cancelled_mid_batch {
737            return Err(ToolLoopCancelled.into());
738        }
739    }
740
741    finish_after_max_iterations(
742        model_provider,
743        history,
744        provider_name,
745        model,
746        temperature,
747        pacing,
748        cancellation_token.as_ref(),
749        max_iterations,
750        accumulated_display_text,
751        &turn_id,
752        knobs,
753        new_messages_out,
754    )
755    .await
756}
