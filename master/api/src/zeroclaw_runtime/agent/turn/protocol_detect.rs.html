11pub(crate) fn longest_suffix_matching_prefix(text: &str, pattern: &str) -> usize {
12    (1..pattern.len())
13        .rev()
14        .find(|&len| text.ends_with(&pattern[..len]))
15        .unwrap_or(0)
16}

18pub(crate) fn find_embedded_protocol_candidate_start(text: &str) -> Option<usize> {
19    let lower = text.to_ascii_lowercase();
20    let mut earliest: Option<usize> = None;
21
22    for pattern in [
23        "<tool_call",
24        "<toolcall",
25        "<tool-call",
26        "<invoke",
27        "<function",
28        "```tool",
29        "```invoke",
30        "```json",
31    ] {
32        if let Some(idx) = lower.find(pattern) {
33            earliest = Some(earliest.map_or(idx, |current| current.min(idx)));
34        }
35    }
36
37    for key in ["\"tool_calls\"", "\"toolcalls\"", "\"function_call\""] {
38        if let Some(key_idx) = lower.find(key)
39            && let Some(json_start) = text[..key_idx].rfind(['{', '['])
40        {
41            earliest = Some(earliest.map_or(json_start, |current| current.min(json_start)));
42        }
43    }
44
45    earliest
46}

48pub(crate) fn find_incomplete_protocol_candidate_start(text: &str) -> Option<usize> {
49    let lower = text.to_ascii_lowercase();
50    let mut earliest: Option<usize> = None;
51
52    for pattern in [
53        "<tool",
54        "<invoke",
55        "<function",
56        "```tool",
57        "```invoke",
58        "```json",
59    ] {
60        if let Some(idx) = lower.rfind(pattern) {
61            earliest = Some(earliest.map_or(idx, |current| current.min(idx)));
62        }
63    }
64
65    for delimiter in ['{', '['] {
66        if let Some(idx) = text.rfind(delimiter) {
67            let tail = &lower[idx..];
68            if tail.contains("\"tool")
69                || tail.contains("\"function")
70                || tail.contains("\"call")
71                || tail.len() <= 16
72            {
73                earliest = Some(earliest.map_or(idx, |current| current.min(idx)));
74            }
75        }
76    }
77
78    earliest
79}

81pub(crate) fn starts_suspicious_protocol_prefix(text: &str) -> bool {
82    let trimmed = text.trim_start();
83    if trimmed.is_empty() {
84        return false;
85    }
86    let lower = trimmed.to_ascii_lowercase();
87    lower.starts_with('{')
88        || lower.starts_with('[')
89        || lower.starts_with("<tool")
90        || lower.starts_with("<invoke")
91        || lower.starts_with("<function")
92        || lower.starts_with("```tool")
93        || lower.starts_with("```invoke")
94        || lower.starts_with("```json")
95}

97pub(crate) fn starts_suspicious_tag_or_fence_prefix(text: &str) -> bool {
98    let lower = text.trim_start().to_ascii_lowercase();
99    lower.starts_with("<tool")
100        || lower.starts_with("<invoke")
101        || lower.starts_with("<function")
102        || lower.starts_with("```tool")
103        || lower.starts_with("```invoke")
104        || lower.starts_with("```json")
105        || lower.starts_with("[tool_call]")
106}

108pub(crate) fn complete_non_protocol_json(text: &str, known_tool_names: &HashSet<String>) -> bool {
109    let trimmed = text.trim();
110    (trimmed.starts_with('{') || trimmed.starts_with('['))
111        && serde_json::from_str::<serde_json::Value>(trimmed).is_ok()
112        && (!looks_like_tool_protocol_envelope(trimmed)
113            || !tool_protocol_envelope_mentions_known_tool(trimmed, known_tool_names))
114}

116pub(crate) fn complete_json_fence_protocol_state(
117    text: &str,
118    known_tool_names: &HashSet<String>,
119) -> Option<bool> {
120    let trimmed = text.trim();
121    let body = json_fence_body(trimmed)?;
122    Some(
123        looks_like_tool_protocol_envelope(body)
124            && tool_protocol_envelope_mentions_known_tool(body, known_tool_names),
125    )
126}

128pub(crate) fn detect_internal_protocol_without_tools(response: &str) -> Option<String> {
129    let trimmed = response.trim();
130    if trimmed.is_empty() {
131        return None;
132    }
133    if looks_like_tool_protocol_example(trimmed) {
134        return None;
135    }
136
137    (looks_like_malformed_tool_protocol_envelope(trimmed)
138        || contains_tool_protocol_tag_call(trimmed)
139        || classify_tool_protocol_envelope(trimmed)
140            .is_some_and(|kind| matches!(kind, ToolProtocolEnvelopeKind::TaggedToolCall))
141        || (classify_tool_protocol_envelope(trimmed).is_none()
142            && looks_like_tool_protocol_envelope(trimmed)))
143    .then(|| {
144        "response resembled an internal tool protocol envelope but no tools were enabled".into()
145    })
146}

148pub(crate) fn detect_tool_call_parse_issue_for_known_tools(
149    response: &str,
150    parsed_calls: &[ParsedToolCall],
151    known_tool_names: &HashSet<String>,
152) -> Option<String> {
153    if !parsed_calls.is_empty() {
154        return None;
155    }
156
157    let trimmed = response.trim();
158    if trimmed.is_empty() || looks_like_tool_protocol_example(trimmed) {
159        return None;
160    }
161
162    let message = "response resembled an internal tool protocol envelope but no valid tool call could be parsed";
163
164    if looks_like_malformed_tool_protocol_envelope_for_known_tools(trimmed, known_tool_names)
165        || contains_tool_protocol_tag_call(trimmed)
166    {
167        return Some(message.into());
168    }
169
170    if let Some(kind) = classify_tool_protocol_envelope(trimmed) {
171        return (matches!(
172            kind,
173            ToolProtocolEnvelopeKind::TaggedToolCall | ToolProtocolEnvelopeKind::ToolResult
174        ) || tool_protocol_envelope_mentions_known_tool(trimmed, known_tool_names))
175        .then(|| message.into());
176    }
177
178    looks_like_tool_protocol_envelope(trimmed).then(|| message.into())
179}

181pub(crate) fn json_fence_body(trimmed: &str) -> Option<&str> {
182    let rest = trimmed.strip_prefix("```")?;
183    let first_newline = rest.find('\n')?;
184    let language = rest[..first_newline].trim().trim_end_matches('\r');
185    if !language.eq_ignore_ascii_case("json") {
186        return None;
187    }
188
189    let body_with_close = &rest[first_newline + 1..];
190    let close_start = body_with_close.rfind("```")?;
191    if !body_with_close[close_start + 3..].trim().is_empty() {
192        return None;
193    }
194    Some(body_with_close[..close_start].trim())
195}
