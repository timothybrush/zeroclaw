1//! Streaming provider-response consumption for the turn loop.
2
3use super::events::{DraftEvent, StreamDelta};
4use super::outcome::{StreamCancelledAfterOutput, StreamInterruptedAfterOutput, ToolLoopCancelled};
5use super::stream_guard::{StreamTextGuard, StreamThinkTagStripper};
6use anyhow::Result;
7use futures_util::StreamExt;
8use tokio_util::sync::CancellationToken;
9use uuid::Uuid;
10use zeroclaw_api::agent::TurnEvent;
11use zeroclaw_api::model_provider::StreamEvent;
12use zeroclaw_providers::{ChatMessage, ChatRequest, ModelProvider, ToolCall};
13
14#[derive(Debug, Default)]
15pub(crate) struct StreamedChatOutcome {
16    pub(crate) response_text: String,
17    /// Accumulated reasoning/thinking content from streaming deltas.
18    ///
19    /// Captured separately from `response_text` so it can be threaded into
20    /// `ChatResponse.reasoning_content` and ultimately persisted on the
21    /// `AssistantToolCalls` history entry. Required for model_providers like
22    /// DeepSeek V4 that reject follow-up requests when the assistant's
23    /// prior `reasoning_content` is missing from replayed tool-call turns
24    ///.
25    pub(crate) reasoning_content: String,
26    pub(crate) tool_calls: Vec<ToolCall>,
27    pub(crate) forwarded_live_deltas: bool,
28    pub(crate) suppressed_protocol: bool,
29    pub(crate) usage: Option<zeroclaw_providers::traits::TokenUsage>,
30}
31
32pub(crate) async fn consume_provider_streaming_response(
33    model_provider: &dyn ModelProvider,
34    messages: &[ChatMessage],
35    request_tools: Option<&[crate::tools::ToolSpec]>,
36    model: &str,
37    temperature: Option<f64>,
38    cancellation_token: Option<&CancellationToken>,
39    on_delta: Option<&tokio::sync::mpsc::Sender<DraftEvent>>,
40    event_tx: Option<&tokio::sync::mpsc::Sender<TurnEvent>>,
41    strict_tool_parsing: bool,
42) -> Result<StreamedChatOutcome> {
43    let mut provider_stream = model_provider.stream_chat(
44        ChatRequest {
45            messages,
46            tools: request_tools,
47            thinking: zeroclaw_api::NATIVE_THINKING_OVERRIDE
48                .try_with(Clone::clone)
49                .ok()
50                .flatten(),
51        },
52        model,
53        temperature,
54        zeroclaw_providers::traits::StreamOptions::new(true),
55    );
56    let mut outcome = StreamedChatOutcome::default();
57    let mut delta_sender = on_delta;
58    let mut suppress_forwarding = false;
59    let mut text_guard = StreamTextGuard::new(request_tools);
60    let mut think_stripper = StreamThinkTagStripper::default();
61    // Correlates PreExecutedToolCall events with their later results so both
62    // TurnEvents share a stable id (FIFO per tool name).
63    let mut pre_executed_ids: std::collections::HashMap<
64        String,
65        std::collections::VecDeque<String>,
66    > = std::collections::HashMap::new();
67    // Tracks event_tx-visible output only (Chunk/Thinking/pre-executed tool
68    // events). Draft (`on_delta`) forwards don't count: drafts are mutable
69    // surfaces, so a non-streaming retry after a stream error overwrites
70    // rather than duplicates.
71    let mut visible_event_output = false;
72    // Exactly the text forwarded as `TurnEvent::Chunk` — what an event_tx
73    // consumer actually SAW. On interruption this (never the raw
74    // accumulated `response_text`, which includes guard-withheld and
75    // suppression-buffered text) is the partial that may be persisted as
76    // already-delivered output.
77    let mut forwarded_text = String::new();
78
79    loop {
80        let next_chunk = if let Some(token) = cancellation_token {
81            tokio::select! {
82                () = token.cancelled() => {
83                    // Cancel after visible streamed text: persist-worthy,
84                    // exactly like the pre-consolidation engine's
85                    // committed-partial-on-cancel.
86                    if forwarded_text.is_empty() {
87                        return Err(ToolLoopCancelled.into());
88                    }
89                    return Err(StreamCancelledAfterOutput::new(forwarded_text).into());
90                }
91                chunk = provider_stream.next() => chunk,
92            }
93        } else {
94            provider_stream.next().await
95        };
96
97        let Some(event_result) = next_chunk else {
98            break;
99        };
100
101        let event = match event_result {
102            Ok(event) => event,
103            Err(err) => {
104                ::zeroclaw_log::record!(
105                    WARN,
106                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
107                        .with_outcome(::zeroclaw_log::EventOutcome::Failure)
108                        .with_attrs(::serde_json::json!({"error": format!("{}", err)})),
109                    "model_provider stream emitted an error event"
110                );
111                let message = format!("model_provider stream error: {err}");
112                if visible_event_output {
113                    // Persist only what the consumer actually saw
114                    // (`forwarded_text`), never the raw accumulated text —
115                    // that includes guard-withheld protocol fragments and
116                    // suppression-buffered output nobody received.
117                    return Err(StreamInterruptedAfterOutput {
118                        partial_text: forwarded_text,
119                        message,
120                    }
121                    .into());
122                }
123                return Err(anyhow::Error::msg(message));
124            }
125        };
126        match event {
127            StreamEvent::Final => break,
128            StreamEvent::Usage(usage) => {
129                outcome.usage = Some(usage);
130            }
131            StreamEvent::ToolCall(tool_call) => {
132                outcome.tool_calls.push(tool_call);
133                suppress_forwarding = true;
134                text_guard.suppress_forwarding = true;
135            }
136            // Pre-executed tool events are for observability only: they are
137            // relayed as TurnEvents but do not affect the agent's tool
138            // dispatch loop.
139            StreamEvent::PreExecutedToolCall { name, args } => {
140                let id = Uuid::new_v4().to_string();
141                pre_executed_ids
142                    .entry(name.clone())
143                    .or_default()
144                    .push_back(id.clone());
145                if let Some(tx) = event_tx {
146                    visible_event_output = true;
147                    let _ = tx
148                        .send(TurnEvent::ToolCall {
149                            id,
150                            name,
151                            args: serde_json::from_str(&args).unwrap_or(serde_json::Value::Null),
152                        })
153                        .await;
154                }
155            }
156            StreamEvent::PreExecutedToolResult { name, output } => {
157                let id = pre_executed_ids
158                    .get_mut(&name)
159                    .and_then(|ids| ids.pop_front())
160                    .unwrap_or_else(|| Uuid::new_v4().to_string());
161                if let Some(tx) = event_tx {
162                    visible_event_output = true;
163                    let _ = tx.send(TurnEvent::ToolResult { id, name, output }).await;
164                }
165            }
166            StreamEvent::TextDelta(chunk) => {
167                // Reasoning/thinking deltas arrive on the same `TextDelta`
168                // event as plain text but populate `chunk.reasoning` instead
169                // of `chunk.delta`. They must be captured into the outcome
170                // even when `chunk.delta` is empty — otherwise model_providers
171                // that require reasoning to round-trip on subsequent turns
172                // (DeepSeek V4 thinking mode; see #6059) reject the next
173                // request with a 400. Reasoning is never forwarded as a
174                // visible response delta — it is the model's internal
175                // monologue, kept for replay only.
176                if let Some(reasoning) = chunk.reasoning.as_deref()
177                    && !reasoning.is_empty()
178                {
179                    outcome.reasoning_content.push_str(reasoning);
180                    // Thinking is surfaced as its own TurnEvent variant; it
181                    // must never reach the Chunk/draft text surfaces.
182                    if let Some(tx) = event_tx {
183                        visible_event_output = true;
184                        let _ = tx
185                            .send(TurnEvent::Thinking {
186                                delta: reasoning.to_string(),
187                            })
188                            .await;
189                    }
190                }
191
192                if chunk.delta.is_empty() {
193                    continue;
194                }
195
196                let sanitized_delta = think_stripper.push(&chunk.delta);
197                if sanitized_delta.is_empty() {
198                    continue;
199                }
200
201                outcome.response_text.push_str(&sanitized_delta);
202
203                if suppress_forwarding {
204                    continue;
205                }
206
207                if strict_tool_parsing {
208                    // Every event_tx send is gated on event_tx ALONE — never
209                    // nested under on_delta. Nesting doubles chunks when both
210                    // are Some and drops them when on_delta is None.
211                    if let Some(tx) = event_tx {
212                        outcome.forwarded_live_deltas = true;
213                        visible_event_output = true;
214                        forwarded_text.push_str(&sanitized_delta);
215                        let _ = tx
216                            .send(TurnEvent::Chunk {
217                                delta: sanitized_delta.clone(),
218                            })
219                            .await;
220                    }
221                    if let Some(tx) = delta_sender {
222                        outcome.forwarded_live_deltas = true;
223                        if tx.send(StreamDelta::Text(sanitized_delta)).await.is_err() {
224                            delta_sender = None;
225                        }
226                    }
227                    continue;
228                }
229
230                let Some(forward_text) = text_guard.push(&sanitized_delta) else {
231                    continue;
232                };
233
234                if let Some(tx) = event_tx {
235                    outcome.forwarded_live_deltas = true;
236                    visible_event_output = true;
237                    forwarded_text.push_str(&forward_text);
238                    let _ = tx
239                        .send(TurnEvent::Chunk {
240                            delta: forward_text.clone(),
241                        })
242                        .await;
243                }
244                if let Some(tx) = delta_sender {
245                    outcome.forwarded_live_deltas = true;
246                    if tx.send(StreamDelta::Text(forward_text)).await.is_err() {
247                        delta_sender = None;
248                    }
249                }
250            }
251        }
252    }
253
254    let trailing_delta = think_stripper.finish();
255    if !trailing_delta.is_empty() {
256        outcome.response_text.push_str(&trailing_delta);
257        if !suppress_forwarding {
258            if strict_tool_parsing {
259                if let Some(tx) = event_tx {
260                    outcome.forwarded_live_deltas = true;
261                    let _ = tx
262                        .send(TurnEvent::Chunk {
263                            delta: trailing_delta.clone(),
264                        })
265                        .await;
266                }
267                if let Some(tx) = delta_sender {
268                    outcome.forwarded_live_deltas = true;
269                    if tx.send(StreamDelta::Text(trailing_delta)).await.is_err() {
270                        delta_sender = None;
271                    }
272                }
273            } else if let Some(forward_text) = text_guard.push(&trailing_delta) {
274                if let Some(tx) = event_tx {
275                    outcome.forwarded_live_deltas = true;
276                    let _ = tx
277                        .send(TurnEvent::Chunk {
278                            delta: forward_text.clone(),
279                        })
280                        .await;
281                }
282                if let Some(tx) = delta_sender {
283                    outcome.forwarded_live_deltas = true;
284                    if tx.send(StreamDelta::Text(forward_text)).await.is_err() {
285                        delta_sender = None;
286                    }
287                }
288            }
289        }
290    }
291
292    if let Some(forward_text) = text_guard.finish() {
293        if let Some(tx) = event_tx {
294            outcome.forwarded_live_deltas = true;
295            let _ = tx
296                .send(TurnEvent::Chunk {
297                    delta: forward_text.clone(),
298                })
299                .await;
300        }
301        if let Some(tx) = delta_sender {
302            outcome.forwarded_live_deltas = true;
303            let _ = tx.send(StreamDelta::Text(forward_text)).await;
304        }
305    }
306    outcome.suppressed_protocol = text_guard.suppressed_protocol;
307
308    Ok(outcome)
309}
