1//! Post-execution recording: result log line, the `after_tool_call` hook, the
2//! completion Status, and filling the executed calls' `ordered_results` slots.
3
4use super::context::TurnCtx;
5use super::events::StreamDelta;
6use super::redact::scrub_credentials;
7use crate::agent::tool_execution::ToolExecutionOutcome;
8use crate::util::truncate_with_ellipsis;
9use zeroclaw_tool_call_parser::ParsedToolCall;
10
11/// Record each executed tool call's outcome (upstream loop body,
12/// post-execution section): one `tool_call_result` log line, the
13/// `after_tool_call` hook, a completion Status to the draft, and the
14/// call's slot in `ordered_results`.
15pub(crate) async fn record_executed_outcomes(
16    ctx: &TurnCtx<'_>,
17    executable_indices: &[usize],
18    executable_calls: &[ParsedToolCall],
19    executed_outcomes: Vec<ToolExecutionOutcome>,
20    ordered_results: &mut [Option<(String, Option<String>, ToolExecutionOutcome)>],
21    iteration: usize,
22) {
23    for ((idx, call), outcome) in executable_indices
24        .iter()
25        .zip(executable_calls.iter())
26        .zip(executed_outcomes)
27    {
28        if let Some(tx) = ctx.event_tx {
29            super::events::emit_tool_call_pair(tx, call, &outcome).await;
30        }
31
32        ::zeroclaw_log::record!(
33            INFO,
34            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Complete)
35                .with_outcome(if outcome.success {
36                    ::zeroclaw_log::EventOutcome::Success
37                } else {
38                    ::zeroclaw_log::EventOutcome::Failure
39                })
40                .with_duration(u64::try_from(outcome.duration.as_millis()).unwrap_or(u64::MAX))
41                .with_attrs(::serde_json::json!({
42                    "model": ctx.model,
43                    "iteration": iteration + 1,
44                    "tool": call.name.clone(),
45                    "error_reason": outcome.error_reason,
46                    "output": scrub_credentials(&outcome.output),
47                    "trace_id": ctx.turn_id,
48                })),
49            "tool_call_result"
50        );
51
52        // ── Hook: after_tool_call (void) ─────────────────
53        if let Some(hooks) = ctx.hooks {
54            let tool_result_obj = crate::tools::ToolResult {
55                success: outcome.success,
56                output: outcome.output.clone(),
57                error: None,
58            };
59            hooks
60                .fire_after_tool_call(&call.name, &tool_result_obj, outcome.duration)
61                .await;
62        }
63
64        // ── Progress: tool completion ───────────────────────
65        if let Some(tx) = ctx.on_delta {
66            let secs = outcome.duration.as_secs();
67            let progress_msg = if outcome.success {
68                format!("\u{2705} {} ({secs}s)\n", call.name)
69            } else if let Some(ref reason) = outcome.error_reason {
70                format!(
71                    "\u{274c} {} ({secs}s): {}\n",
72                    call.name,
73                    truncate_with_ellipsis(reason, 200)
74                )
75            } else {
76                format!("\u{274c} {} ({secs}s)\n", call.name)
77            };
78            ::zeroclaw_log::record!(
79                DEBUG,
80                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
81                    .with_attrs(::serde_json::json!({"tool": call.name, "secs": secs})),
82                "Sending progress complete to draft"
83            );
84            let _ = tx.send(StreamDelta::Status(progress_msg)).await;
85        }
86
87        ordered_results[*idx] = Some((call.name.clone(), call.tool_call_id.clone(), outcome));
88    }
89}
