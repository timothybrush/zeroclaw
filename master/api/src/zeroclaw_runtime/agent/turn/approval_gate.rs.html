1//! The per-tool-call approval gate: CLI prompt, channel inline approval, or
2//! auto-deny, plus decision recording.
3
4use super::context::TurnCtx;
5use super::events::StreamDelta;
6use super::redact::scrub_credentials;
7use crate::agent::tool_execution::ToolExecutionOutcome;
8use crate::approval::{ApprovalRequest, ApprovalRequirement, ApprovalResponse};
9use std::time::Duration;
10
11/// Outcome of [`gate_tool_approval`] for one tool call.
12///
13/// `Deny`/`Replace` carry the synthesized [`ToolExecutionOutcome`] the caller
14/// records into its `ordered_results` slot before skipping execution;
15/// `Proceed::approved` feeds `set_runtime_approved_arg`.
16pub(crate) enum ApprovalGateOutcome {
17    Proceed { approved: bool },
18    Deny(ToolExecutionOutcome),
19    Replace(ToolExecutionOutcome),
20}
21
22/// Run the approval flow for one tool call (upstream loop body, approval
23/// section): resolve the tool's approval requirement, prompt interactively on
24/// CLI or via the channel's inline approval on non-interactive channels
25/// (falling back to auto-deny), and record the decision.
26pub(crate) async fn gate_tool_approval(
27    ctx: &TurnCtx<'_>,
28    tool_name: &str,
29    tool_args: &serde_json::Value,
30    iteration: usize,
31) -> ApprovalGateOutcome {
32    let mut approval_requirement = ctx
33        .approval
34        .map(|mgr| mgr.approval_requirement(tool_name))
35        .unwrap_or(ApprovalRequirement::NotRequired);
36    if let Some(mgr) = ctx.approval
37        && approval_requirement == ApprovalRequirement::Prompt
38    {
39        let request = ApprovalRequest {
40            tool_name: tool_name.to_string(),
41            arguments: tool_args.clone(),
42        };
43
44        // Interactive CLI: prompt the operator.
45        // Non-interactive (channels): try the channel's inline
46        // approval (e.g. Telegram inline keyboard) before falling
47        // back to auto-deny.
48        let decision = if mgr.is_non_interactive() {
49            let channel_decision = if let Some(ch) = ctx.channel {
50                let ch_request = zeroclaw_api::channel::ChannelApprovalRequest {
51                    tool_name: request.tool_name.clone(),
52                    arguments_summary: crate::approval::summarize_args(&request.arguments),
53                    raw_arguments: Some(request.arguments.clone()),
54                };
55                let recipient = ctx.channel_reply_target.unwrap_or_default();
56                match ch.request_approval(recipient, &ch_request).await {
57                    Ok(Some(r)) => Some(r),
58                    Ok(None) => None,
59                    Err(e) => {
60                        ::zeroclaw_log::record!(
61                            WARN,
62                            ::zeroclaw_log::Event::new(
63                                module_path!(),
64                                ::zeroclaw_log::Action::Note
65                            )
66                            .with_outcome(::zeroclaw_log::EventOutcome::Unknown)
67                            .with_attrs(::serde_json::json!({"error": format!("{}", e)})),
68                            "Channel approval request failed"
69                        );
70                        None
71                    }
72                }
73            } else {
74                None
75            };
76            match channel_decision {
77                Some(zeroclaw_api::channel::ChannelApprovalResponse::Approve) => {
78                    ApprovalResponse::Yes
79                }
80                Some(zeroclaw_api::channel::ChannelApprovalResponse::AlwaysApprove) => {
81                    ApprovalResponse::Always
82                }
83                Some(zeroclaw_api::channel::ChannelApprovalResponse::Deny) => ApprovalResponse::No,
84                Some(zeroclaw_api::channel::ChannelApprovalResponse::DenyWithEdit {
85                    replacement,
86                }) => ApprovalResponse::ReplaceWith(replacement),
87                // Channel doesn't support approval — auto-deny.
88                None => ApprovalResponse::No,
89            }
90        } else {
91            mgr.prompt_cli(&request)
92        };
93
94        // The approval audit records which surface decided. On the streaming
95        // path `ctx.channel` is the approval bridge fanning out to several
96        // registered back-channels, and `ctx.channel_name` is the loop's
97        // static "cli"; prefer the back-channel that actually answered so a
98        // WS/ACP approval is attributed to WS/ACP, not "cli". Single channels
99        // and the CLI prompt path report `None` and keep `channel_name`.
100        let decision_channel = ctx
101            .channel
102            .and_then(|ch| ch.last_decision_channel())
103            .unwrap_or_else(|| ctx.channel_name.to_string());
104        mgr.record_decision(tool_name, tool_args, &decision, &decision_channel);
105
106        if decision == ApprovalResponse::No {
107            let denied = "Denied by user.".to_string();
108            ::zeroclaw_log::record!(
109                WARN,
110                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
111                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
112                    .with_attrs(::serde_json::json!({
113                        "model": ctx.model,
114                        "iteration": iteration + 1,
115                        "tool": tool_name,
116                        "arguments": scrub_credentials(&tool_args.to_string()),
117                        "result": denied,
118                        "trace_id": ctx.turn_id,
119                    })),
120                "tool_call_result"
121            );
122            if let Some(tx) = ctx.on_delta {
123                let _ = tx
124                    .send(StreamDelta::Status(format!(
125                        "\u{274c} {}: {}\n",
126                        tool_name, denied
127                    )))
128                    .await;
129            }
130            return ApprovalGateOutcome::Deny(ToolExecutionOutcome {
131                output: denied.clone(),
132                success: false,
133                error_reason: Some(denied),
134                duration: Duration::ZERO,
135                receipt: None,
136            });
137        }
138
139        if let ApprovalResponse::ReplaceWith(replacement) = &decision {
140            if let Some(tx) = ctx.on_delta {
141                let _ = tx
142                    .send(StreamDelta::Status(format!(
143                        "\u{270f} {}: replaced by user\n",
144                        tool_name
145                    )))
146                    .await;
147            }
148            ::zeroclaw_log::record!(
149                INFO,
150                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
151                    .with_outcome(::zeroclaw_log::EventOutcome::Success)
152                    .with_attrs(::serde_json::json!({
153                        "model": ctx.model,
154                        "iteration": iteration + 1,
155                        "tool": tool_name,
156                        "arguments": scrub_credentials(&tool_args.to_string()),
157                        "replaced": true,
158                        "output": scrub_credentials(replacement),
159                        "trace_id": ctx.turn_id,
160                    })),
161                "tool_call_result"
162            );
163            return ApprovalGateOutcome::Replace(ToolExecutionOutcome {
164                output: crate::approval::sanitize_tool_replacement(replacement),
165                success: true,
166                error_reason: None,
167                duration: Duration::ZERO,
168                receipt: None,
169            });
170        }
171
172        if matches!(decision, ApprovalResponse::Yes | ApprovalResponse::Always) {
173            approval_requirement = ApprovalRequirement::Approved;
174        }
175    }
176
177    ApprovalGateOutcome::Proceed {
178        approved: approval_requirement == ApprovalRequirement::Approved,
179    }
180}
