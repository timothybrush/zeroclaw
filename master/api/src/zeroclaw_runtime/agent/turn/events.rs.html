12/// Minimum characters per chunk when relaying LLM text to a streaming draft.
13pub(crate) const STREAM_CHUNK_MIN_CHARS: usize = 80;
14
15/// Minimum interval between progress sends to avoid flooding the draft channel.
16pub const PROGRESS_MIN_INTERVAL_MS: u64 = 500;
17
18/// Delta sent from the agent loop to the channel's draft updater.
19/// Append-only — no clear/reset variant exists by design.
20#[derive(Debug, Clone)]
21pub enum StreamDelta {
22    /// Response text to append to the message buffer.
23    Text(String),
24    /// Ephemeral tool progress (not part of the response body).
25    Status(String),
26}
27
28/// Backwards-compatible alias while callers are migrated.
29pub type DraftEvent = StreamDelta;
30
31/// Send `text` to the draft channel in word-aligned chunks of at least
32/// [`STREAM_CHUNK_MIN_CHARS`] (upstream loop body, no-tool-calls final exit).
33/// Used when the final response wasn't already streamed live. Honors the
34/// cancellation token between chunks; a closed receiver stops chunking
35/// silently.
36pub(crate) async fn stream_text_posthoc_chunks(
37    on_delta: &Sender<DraftEvent>,
38    text: &str,
39    cancellation_token: Option<&CancellationToken>,
40) -> Result<()> {
41    let mut chunk = String::new();
42    for word in text.split_inclusive(char::is_whitespace) {
43        if cancellation_token.is_some_and(CancellationToken::is_cancelled) {
44            return Err(ToolLoopCancelled.into());
45        }
46        chunk.push_str(word);
47        if chunk.len() >= STREAM_CHUNK_MIN_CHARS
48            && on_delta
49                .send(StreamDelta::Text(std::mem::take(&mut chunk)))
50                .await
51                .is_err()
52        {
53            break;
54        }
55    }
56    if !chunk.is_empty() {
57        let _ = on_delta.send(StreamDelta::Text(chunk)).await;
58    }
59    Ok(())
60}

62/// Emit the `TurnEvent::ToolCall`/`ToolResult` pair for one executed tool
63/// call (upstream E2 parity: per-outcome emission after execution).
64///
65/// Text-protocol parsed calls carry `tool_call_id: None`; the TurnEvent
66/// contract documents `id` as a stable correlation id shared by the pair
67/// (ACP/WS clients key on it), so a fresh UUID is synthesized per pair —
68/// the old streaming engine pre-assigned UUIDs to id-less calls for the
69/// same reason. Distinct calls in one batch must never collapse to `""`.
70pub(crate) async fn emit_tool_call_pair(
71    event_tx: &Sender<TurnEvent>,
72    call: &ParsedToolCall,
73    outcome: &ToolExecutionOutcome,
74) {
75    let call_id = call
76        .tool_call_id
77        .clone()
78        .unwrap_or_else(|| uuid::Uuid::new_v4().to_string());
79    let _ = event_tx
80        .send(TurnEvent::ToolCall {
81            id: call_id.clone(),
82            name: call.name.clone(),
83            args: call.arguments.clone(),
84        })
85        .await;
86    let _ = event_tx
87        .send(TurnEvent::ToolResult {
88            id: call_id,
89            name: call.name.clone(),
90            output: outcome.output.clone(),
91        })
92        .await;
93}
94
95/// `TurnEvent` variant of [`stream_text_posthoc_chunks`]: when the final
96/// response was not streamed live, emit it as one post-hoc `Chunk`.
97pub(crate) async fn emit_posthoc_turn_chunk(event_tx: Option<&Sender<TurnEvent>>, text: &str) {
98    if let Some(tx) = event_tx {
99        let _ = tx
100            .send(TurnEvent::Chunk {
101                delta: text.to_string(),
102            })
103            .await;
104    }
105}

107#[cfg(test)]
108mod tests {

110    use std::time::Duration;

112    fn parsed_call(id: Option<&str>) -> ParsedToolCall {
113        ParsedToolCall {
114            name: "echo".into(),
115            arguments: serde_json::json!({}),
116            tool_call_id: id.map(str::to_string),
117        }
118    }

120    fn ok_outcome() -> ToolExecutionOutcome {
121        ToolExecutionOutcome {
122            output: "out".into(),
123            success: true,
124            error_reason: None,
125            duration: Duration::ZERO,
126            receipt: None,
127        }
128    }

130    /// Text-protocol calls have no id; the pair must still correlate via a
131    /// fresh non-empty id, and two id-less calls must never share one.
132    #[tokio::test]
133    async fn idless_calls_get_distinct_synthesized_pair_ids() {
134        let (tx, mut rx) = tokio::sync::mpsc::channel(8);
135        emit_tool_call_pair(&tx, &parsed_call(None), &ok_outcome()).await;
136        emit_tool_call_pair(&tx, &parsed_call(None), &ok_outcome()).await;
137        drop(tx);
138
139        let mut ids = Vec::new();
140        while let Some(ev) = rx.recv().await {
141            match ev {
142                TurnEvent::ToolCall { id, .. } | TurnEvent::ToolResult { id, .. } => ids.push(id),
143                _ => {}
144            }
145        }
146        assert_eq!(ids.len(), 4, "two pairs = four events");
147        assert!(
148            ids.iter().all(|id| !id.is_empty()),
149            "synthesized ids must be non-empty: {ids:?}"
150        );
151        assert_eq!(
152            ids[0], ids[1],
153            "ToolCall/ToolResult of one pair must share the id"
154        );
155        assert_eq!(ids[2], ids[3], "second pair must share its id");
156        assert_ne!(ids[0], ids[2], "distinct calls must get distinct ids");
157    }
158
159    /// Parser-assigned ids pass through untouched.
160    #[tokio::test]
161    async fn existing_ids_pass_through() {
162        let (tx, mut rx) = tokio::sync::mpsc::channel(8);
163        emit_tool_call_pair(&tx, &parsed_call(Some("native-7")), &ok_outcome()).await;
164        drop(tx);
165        while let Some(ev) = rx.recv().await {
166            match ev {
167                TurnEvent::ToolCall { id, .. } | TurnEvent::ToolResult { id, .. } => {
168                    assert_eq!(id, "native-7");
169                }
170                _ => {}
171            }
172        }
173    }
