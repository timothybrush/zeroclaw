21/// Build assistant history entry in JSON format for native tool-call APIs.
22/// `convert_messages` in the OpenRouter model_provider parses this JSON to reconstruct
23/// the proper `NativeMessage` with structured `tool_calls`.
24pub(crate) fn build_native_assistant_history(
25    text: &str,
26    tool_calls: &[ToolCall],
27    reasoning_content: Option<&str>,
28) -> String {
29    let calls_json: Vec<serde_json::Value> = tool_calls
30        .iter()
31        .map(|tc| {
32            serde_json::json!({
33                "id": tc.id,
34                "name": tc.name,
35                "arguments": tc.arguments,
36            })
37        })
38        .collect();
39
40    let content = if text.trim().is_empty() {
41        serde_json::Value::Null
42    } else {
43        serde_json::Value::String(text.trim().to_string())
44    };
45
46    let mut obj = serde_json::json!({
47        "content": content,
48        "tool_calls": calls_json,
49    });
50
51    if let Some(rc) = reasoning_content {
52        obj.as_object_mut().unwrap().insert(
53            "reasoning_content".to_string(),
54            serde_json::Value::String(rc.to_string()),
55        );
56    }
57
58    obj.to_string()
59}

61pub(crate) fn resolve_display_text(
62    response_text: &str,
63    parsed_text: &str,
64    has_tool_calls: bool,
65    has_native_tool_calls: bool,
66) -> String {
67    if has_tool_calls {
68        if !parsed_text.is_empty() {
69            return parsed_text.to_string();
70        }
71        if has_native_tool_calls {
72            return response_text.to_string();
73        }
74        return String::new();
75    }
76
77    if parsed_text.is_empty() {
78        response_text.to_string()
79    } else {
80        parsed_text.to_string()
81    }
82}

94/// Interpret a successful chat response. Takes the response by value and
95/// holds no borrows of `ctx` past the call (RUN_SHEET `turn.parse_response`).
96pub(crate) async fn interpret_chat_response(
97    ctx: &TurnCtx<'_>,
98    resp: ChatResponse,
99    specs: &IterationToolSpecs,
100    streamed_protocol_suppressed: bool,
101    llm_started_at: Instant,
102    iteration: usize,
103    detect_protocol_without_tools: bool,
104) -> InterpretedResponse {
105    let (resp_input_tokens, resp_output_tokens) = resp
106        .usage
107        .as_ref()
108        .map(|u| (u.input_tokens, u.output_tokens))
109        .unwrap_or((None, None));
110
111    ctx.observer.record_event(&ObserverEvent::LlmResponse {
112        model_provider: ctx.provider_name.to_string(),
113        model: ctx.model.to_string(),
114        duration: llm_started_at.elapsed(),
115        success: true,
116        error_message: None,
117        input_tokens: resp_input_tokens,
118        output_tokens: resp_output_tokens,
119        channel: None,
120        agent_alias: None,
121        turn_id: None,
122    });
123
124    // Per-LLM-call usage event, right after the observer success event
125    // (upstream E2 parity, agent.rs Usage emission).
126    if let Some(tx) = ctx.event_tx
127        && let Some(ref usage) = resp.usage
128    {
129        let _ = tx
130            .send(TurnEvent::Usage {
131                input_tokens: usage.input_tokens,
132                cached_input_tokens: usage.cached_input_tokens,
133                output_tokens: usage.output_tokens,
134                cost_usd: None,
135            })
136            .await;
137    }
138
139    // Record cost via task-local tracker (no-op when not scoped)
140    let _ = resp
141        .usage
142        .as_ref()
143        .and_then(|usage| record_tool_loop_cost_usage(ctx.provider_name, ctx.model, usage));
144
145    let response_text = strip_think_tags(resp.text_or_empty());
146    // First try native structured tool calls (OpenAI-format).
147    // Fall back to text-based parsing (XML tags, markdown blocks,
148    // GLM format) only if the model_provider returned no native calls —
149    // this ensures we support both native and prompt-guided models.
150    let mut calls: Vec<ParsedToolCall> = if specs.tool_specs.is_empty() {
151        Vec::new()
152    } else {
153        resp.tool_calls
154            .iter()
155            .map(|call| ParsedToolCall {
156                name: call.name.clone(),
157                arguments: serde_json::from_str::<serde_json::Value>(&call.arguments)
158                    .unwrap_or_else(|_| serde_json::Value::Object(serde_json::Map::new())),
159                tool_call_id: Some(call.id.clone()),
160            })
161            .collect()
162    };
163    let mut parsed_text = String::new();
164
165    if calls.is_empty()
166        && !specs.tool_specs.is_empty()
167        && !ctx.strict_tool_parsing
168        && !looks_like_tool_protocol_example(&response_text)
169    {
170        let (fallback_text, fallback_calls) = parse_tool_calls(&response_text);
171        let filtered_calls: Vec<ParsedToolCall> = fallback_calls
172            .into_iter()
173            .filter(|call| {
174                specs
175                    .known_tool_names
176                    .contains(&call.name.to_ascii_lowercase())
177            })
178            .collect();
179        if !fallback_text.is_empty() && !filtered_calls.is_empty() {
180            parsed_text = fallback_text;
181        }
182        calls = filtered_calls;
183    }
184
185    let parse_issue = if ctx.strict_tool_parsing {
186        None
187    } else if specs.tool_specs.is_empty() {
188        // Knob-gated (embedders return model text verbatim); a live stream
189        // suppression already altered the visible text, so it is always
190        // reported regardless of the knob.
191        detect_protocol_without_tools
192            .then(|| detect_internal_protocol_without_tools(&response_text))
193            .flatten()
194            .or_else(|| {
195                streamed_protocol_suppressed.then(|| {
196                    "streaming text guard suppressed an internal tool protocol envelope".to_string()
197                })
198            })
199    } else {
200        detect_tool_call_parse_issue_for_known_tools(
201            &response_text,
202            &calls,
203            &specs.known_tool_names,
204        )
205        .or_else(|| {
206            streamed_protocol_suppressed.then(|| {
207                "streaming text guard suppressed an internal tool protocol envelope".to_string()
208            })
209        })
210    };
211    if let Some(ref issue) = parse_issue {
212        ::zeroclaw_log::record!(
213            WARN,
214            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
215                .with_outcome(::zeroclaw_log::EventOutcome::Failure)
216                .with_attrs(::serde_json::json!({
217                    "model": ctx.model,
218                    "iteration": iteration + 1,
219                    "issue": issue.as_str(),
220                    "response": scrub_credentials(&response_text),
221                    "trace_id": ctx.turn_id,
222                })),
223            "tool_call_parse_issue"
224        );
225    }
226
227    ::zeroclaw_log::record!(
228        INFO,
229        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Receive)
230            .with_outcome(::zeroclaw_log::EventOutcome::Success)
231            .with_duration(u64::try_from(llm_started_at.elapsed().as_millis()).unwrap_or(u64::MAX))
232            .with_attrs(::serde_json::json!({
233                "model": ctx.model,
234                "iteration": iteration + 1,
235                "input_tokens": resp_input_tokens,
236                "output_tokens": resp_output_tokens,
237                "raw_response": scrub_credentials(&response_text),
238                "native_tool_calls": resp.tool_calls.len(),
239                "parsed_tool_calls": calls.len(),
240                "trace_id": ctx.turn_id,
241            })),
242        "llm_response"
243    );
244
245    // Preserve native tool call IDs in assistant history so role=tool
246    // follow-up messages can reference the exact call id.
247    let reasoning_content = resp.reasoning_content.clone();
248    let assistant_history_content = if resp.tool_calls.is_empty() {
249        if specs.use_native_tools {
250            build_native_assistant_history_from_parsed_calls(
251                &response_text,
252                &calls,
253                reasoning_content.as_deref(),
254            )
255            .unwrap_or_else(|| response_text.clone())
256        } else {
257            response_text.clone()
258        }
259    } else {
260        build_native_assistant_history(
261            &response_text,
262            &resp.tool_calls,
263            reasoning_content.as_deref(),
264        )
265    };
266
267    let native_calls = resp.tool_calls;
268    InterpretedResponse {
269        response_text,
270        parsed_text,
271        tool_calls: calls,
272        assistant_history_content,
273        native_tool_calls: native_calls,
274        parse_issue_detected: parse_issue.is_some(),
275    }
276}
