1//! Channel delivery-default injection for cron_add tool calls.
2
3const AUTO_DELIVERY_DEFAULT_CHANNELS: &[&str] = &[
4    "telegram",
5    "discord",
6    "slack",
7    "mattermost",
8    "matrix",
9    "dingtalk",
10    "lark",
11    "feishu",
12];
13
14pub(crate) fn maybe_inject_channel_delivery_defaults(
15    tool_name: &str,
16    tool_args: &mut serde_json::Value,
17    channel_name: &str,
18    channel_reply_target: Option<&str>,
19) {
20    if tool_name != "cron_add" {
21        return;
22    }
23
24    if !AUTO_DELIVERY_DEFAULT_CHANNELS.contains(&channel_name) {
25        return;
26    }
27
28    let Some(reply_target) = channel_reply_target
29        .map(str::trim)
30        .filter(|value| !value.is_empty())
31    else {
32        return;
33    };
34
35    let Some(args) = tool_args.as_object_mut() else {
36        return;
37    };
38
39    let is_agent_job = args
40        .get("job_type")
41        .and_then(serde_json::Value::as_str)
42        .is_some_and(|job_type| job_type.eq_ignore_ascii_case("agent"))
43        || args
44            .get("prompt")
45            .and_then(serde_json::Value::as_str)
46            .is_some_and(|prompt| !prompt.trim().is_empty());
47    if !is_agent_job {
48        return;
49    }
50
51    let default_delivery = || {
52        serde_json::json!({
53            "mode": "announce",
54            "channel": channel_name,
55            "to": reply_target,
56        })
57    };
58
59    match args.get_mut("delivery") {
60        None => {
61            args.insert("delivery".to_string(), default_delivery());
62        }
63        Some(serde_json::Value::Null) => {
64            *args.get_mut("delivery").expect("delivery key exists") = default_delivery();
65        }
66        Some(serde_json::Value::Object(delivery)) => {
67            if delivery
68                .get("mode")
69                .and_then(serde_json::Value::as_str)
70                .is_some_and(|mode| mode.eq_ignore_ascii_case("none"))
71            {
72                return;
73            }
74
75            delivery
76                .entry("mode".to_string())
77                .or_insert_with(|| serde_json::Value::String("announce".to_string()));
78
79            let needs_channel = delivery
80                .get("channel")
81                .and_then(serde_json::Value::as_str)
82                .is_none_or(|value| value.trim().is_empty());
83            if needs_channel {
84                delivery.insert(
85                    "channel".to_string(),
86                    serde_json::Value::String(channel_name.to_string()),
87                );
88            }
89
90            let needs_target = delivery
91                .get("to")
92                .and_then(serde_json::Value::as_str)
93                .is_none_or(|value| value.trim().is_empty());
94            if needs_target {
95                delivery.insert(
96                    "to".to_string(),
97                    serde_json::Value::String(reply_target.to_string()),
98                );
99            }
100        }
101        Some(_) => {}
102    }
103}
