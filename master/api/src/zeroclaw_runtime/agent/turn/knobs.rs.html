1//! Per-caller loop behaviour knobs (#7415 consolidation).
2//!
3//! Every divergence between the historical turn engines that survives the
4//! consolidation is an explicit field here, set per caller. `Default`
5//! preserves today's channel/CLI behaviour.
6
7/// How to handle max-tool-iteration exhaustion.
8#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
9pub enum MaxIterationBehavior {
10    /// Ask the LLM for a tools-free final summary (channel/CLI behaviour).
11    #[default]
12    GracefulSummary,
13    /// Bail with "exceeded maximum tool iterations" (embedder control signal).
14    ErrorAtCap,
15}
16
17/// Explicit knobs for per-caller loop behaviour.
18#[derive(Debug, Clone)]
19pub struct LoopKnobs {
20    pub dedup_enabled: bool,
21    pub max_iteration_behavior: MaxIterationBehavior,
22    /// When `true` (channel paths), a response that resembles the internal
23    /// tool protocol while no tools are enabled is classified as a parse
24    /// issue (malformed-protocol retry, then i18n fallback) so raw protocol
25    /// text never reaches end users. Embedder wrappers set `false`: their
26    /// contract is to return the model text verbatim and let the embedder
27    /// do its own post-processing.
28    pub detect_protocol_without_tools: bool,
29}
30
31impl Default for LoopKnobs {
32    fn default() -> Self {
33        Self {
34            dedup_enabled: true,
35            max_iteration_behavior: MaxIterationBehavior::GracefulSummary,
36            detect_protocol_without_tools: true,
37        }
38    }
39}
