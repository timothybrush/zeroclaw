4use crate::agent::history::{emergency_history_trim, fast_trim_tool_results};
5use crate::observability::{Observer, ObserverEvent};
6use std::time::Instant;

9/// Record a failed provider call: observer `LlmResponse` (failure) and the
10/// `llm_response` failure log line.
11pub(crate) fn record_llm_failure(
12    observer: &dyn Observer,
13    provider_name: &str,
14    model: &str,
15    llm_started_at: Instant,
16    iteration: usize,
17    turn_id: &str,
18    e: &anyhow::Error,
19) {
20    // User cancellation gets the fixed message the streaming consumers have
21    // always seen (and pin), never a raw error string.
22    let safe_error = if is_tool_loop_cancelled(e) {
23        "request cancelled by user".to_string()
24    } else {
25        zeroclaw_providers::sanitize_api_error(&e.to_string())
26    };
27    observer.record_event(&ObserverEvent::LlmResponse {
28        model_provider: provider_name.to_string(),
29        model: model.to_string(),
30        duration: llm_started_at.elapsed(),
31        success: false,
32        error_message: Some(safe_error.clone()),
33        input_tokens: None,
34        output_tokens: None,
35        channel: None,
36        agent_alias: None,
37        turn_id: None,
38    });
39    ::zeroclaw_log::record!(
40        WARN,
41        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
42            .with_outcome(::zeroclaw_log::EventOutcome::Failure)
43            .with_duration(u64::try_from(llm_started_at.elapsed().as_millis()).unwrap_or(u64::MAX))
44            .with_attrs(::serde_json::json!({
45                "model": model,
46                "iteration": iteration + 1,
47                "error": safe_error,
48                "trace_id": turn_id,
49            })),
50        "llm_response"
51    );
52}

54/// Context overflow recovery: trim history and retry.
55///
56/// Returns `true` when the history was trimmed and the caller should
57/// `continue` the loop; the orchestrator keeps
58/// `if recovered { continue; } return Err(e);` inline.
59pub(crate) fn try_recover_context_overflow(
60    history: &mut Vec<ChatMessage>,
61    e: &anyhow::Error,
62    iteration: usize,
63) -> bool {
64    if zeroclaw_providers::reliable::is_context_window_exceeded(e) {
65        ::zeroclaw_log::record!(
66            WARN,
67            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
68                .with_outcome(::zeroclaw_log::EventOutcome::Unknown)
69                .with_attrs(::serde_json::json!({"iteration": iteration + 1})),
70            "Context window exceeded, attempting in-loop recovery"
71        );
72
73        // Step 1: fast-trim old tool results (cheap)
74        let chars_saved = fast_trim_tool_results(history, 4);
75        if chars_saved > 0 {
76            ::zeroclaw_log::record!(
77                INFO,
78                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
79                    .with_attrs(::serde_json::json!({"chars_saved": chars_saved})),
80                "Context recovery: trimmed old tool results, retrying"
81            );
82            return true;
83        }
84
85        // Step 2: emergency drop oldest non-system messages
86        let dropped = emergency_history_trim(history, 4);
87        if dropped > 0 {
88            ::zeroclaw_log::record!(
89                INFO,
90                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
91                    .with_attrs(::serde_json::json!({"dropped": dropped})),
92                "Context recovery: dropped old messages, retrying"
93            );
94            return true;
95        }
96
97        // Nothing left to trim — truly unrecoverable
98        ::zeroclaw_log::record!(
99            ERROR,
100            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
101                .with_outcome(::zeroclaw_log::EventOutcome::Failure),
102            "Context overflow unrecoverable: no trimmable messages"
103        );
104    }
105    false
106}
