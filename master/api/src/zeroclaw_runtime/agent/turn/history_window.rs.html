1//! Pre-iteration history maintenance: preemptive token-budget trimming,
2//! orphaned tool-message removal, and system-message normalization.
3
4use crate::agent::history::{
5    estimate_history_tokens, fast_trim_tool_results, normalize_system_messages,
6};
7use zeroclaw_providers::ChatMessage;
8
9pub(crate) fn preflight_history_maintenance(
10    history: &mut Vec<ChatMessage>,
11    context_token_budget: usize,
12    iteration: usize,
13) {
14    // Preemptive context management: trim history before it overflows
15    if context_token_budget > 0 {
16        let estimated = estimate_history_tokens(history);
17        if estimated > context_token_budget {
18            ::zeroclaw_log::record!(INFO, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_attrs(::serde_json::json!({"estimated": estimated, "budget": context_token_budget, "iteration": iteration + 1})), "Preemptive context trim: estimated tokens exceed budget");
19            let chars_saved = fast_trim_tool_results(history, 4);
20            if chars_saved > 0 {
21                ::zeroclaw_log::record!(
22                    INFO,
23                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
24                        .with_attrs(::serde_json::json!({"chars_saved": chars_saved})),
25                    "Preemptive fast-trim applied"
26                );
27            }
28            // If still over budget, use the history pruner for deeper cleanup
29            let recheck = estimate_history_tokens(history);
30            if recheck > context_token_budget {
31                let stats = crate::agent::history_pruner::prune_history(
32                    history,
33                    &crate::agent::history_pruner::HistoryPrunerConfig {
34                        enabled: true,
35                        max_tokens: context_token_budget,
36                        keep_recent: 4,
37                        collapse_tool_results: true,
38                    },
39                );
40                if stats.dropped_messages > 0 || stats.collapsed_pairs > 0 {
41                    ::zeroclaw_log::record!(INFO, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_attrs(::serde_json::json!({"collapsed": stats.collapsed_pairs, "dropped": stats.dropped_messages})), "Preemptive history prune applied");
42                }
43            }
44        }
45    }
46
47    // Remove orphaned tool-role messages whose assistant (tool_calls)
48    // counterpart was dropped by proactive trimming, context compression,
49    // or session history reloading.  Without this, model_providers like MiniMax
50    // reject the request with "tool result's tool id not found" (bug #5743).
51    let pruned_in_loop = crate::agent::history_pruner::remove_orphaned_tool_messages(history);
52    if !pruned_in_loop.is_empty() {
53        ::zeroclaw_log::record!(
54            WARN,
55            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
56                .with_outcome(::zeroclaw_log::EventOutcome::Unknown)
57                .with_attrs(::serde_json::json!({
58                    "removed": pruned_in_loop.removed,
59                    "orphan_tool_call_ids": pruned_in_loop.orphan_tool_call_ids,
60                })),
61            "remove_orphaned_tool_messages fired inside run_tool_call_loop: \
62             assistant tool_use blocks and/or tool_results were stripped from \
63             the live history. If this fires mid-conversation the model loses \
64             the in-flight tool work and acts like it just woke up."
65        );
66    }
67    normalize_system_messages(history);
68}
