1//! Agent-loop tool that sends a message to a configured peer on a
2//! shared channel.
3//!
4//! Validates the target against [`crate::peers::ResolvedPeers`] for
5//! the calling agent on the requested channel: peers must mutually
6//! opt in via a `[peer_groups.<name>]` block whose `agents` lists
7//! both, OR appear on the group's `external_peers` list, before this
8//! tool will deliver. Cross-channel sends from outside the resolver's
9//! authorization surface are rejected.
10//!
11//! Delivery splits by target type:
12//!
13//! - **Agent-alias targets** route in-process via
14//!   [`crate::agent::loop_::process_message`]: alpha calls
15//!   `send_message_to_peer(target = "beta", ...)` and beta's agent
16//!   loop runs the message. The two agents share the channel's bot
17//!   identity, so an outbound to the channel would loop the bot's
18//!   own handle back through inbound; the in-process path avoids
19//!   that and lets the orchestrator deliver beta's reply (if any)
20//!   through the same channel beta is configured on.
21//!
22//!   This path is fire-and-forget: the recipient runs on a detached
23//!   `zeroclaw_spawn::spawn!`, so the sender's `ToolResult.success = true`
24//!   means "accepted for processing", not "completed". Recipient
25//!   errors do NOT surface to the sender; they are emitted via
26//!   `tracing::warn!` inside the spawned task and via the recipient
27//!   agent's own observability (audit log, runtime trace, channel
28//!   reply). Observers diagnosing a missing peer message should look
29//!   at the recipient's spans, not the sender's tool output.
30//! - **External peers** (humans, external bots) route through
31//!   [`crate::cron::scheduler::deliver_announcement`] with the
32//!   external username as the platform target. The channel registry
33//!   the binary registers at startup forwards the send to the live
34//!   channel instance. This path is synchronous: the
35//!   `deliver_announcement` future resolves before the tool returns,
36//!   so a `success = false` here genuinely reflects a delivery
37//!   failure.
38

48/// Send a message to a peer on a shared channel. Bound to a single
49/// calling agent's alias; the tool validates every send against that
50/// agent's resolved peer set.
51pub struct SendMessageToPeerTool {
52    config: Arc<Config>,
53    sender_alias: String,
54    description: String,
55}
56
57impl SendMessageToPeerTool {
58    pub fn new(config: Arc<Config>, sender_alias: impl Into<String>) -> Self {
59        let sender_alias = sender_alias.into();
60        let description = build_description();
61        Self {
62            config,
63            sender_alias,
64            description,
65        }
66    }
67}

70impl Tool for SendMessageToPeerTool {
71    fn name(&self) -> &str {
72        "send_message_to_peer"
73    }
74
75    fn description(&self) -> &str {
76        &self.description
77    }
78
79    fn parameters_schema(&self) -> serde_json::Value {
80        json!({
81            "type": "object",
82            "properties": {
83                "channel": {
84                    "type": "string",
85                    "description": "Channel ref to deliver on (e.g. 'telegram.prod'). Must be one of the agent's configured channels and a channel the target peer also listens on."
86                },
87                "target": {
88                    "type": "string",
89                    "description": "Recipient identifier — a peer agent's alias or an external peer's username (e.g. '@operator')."
90                },
91                "message": {
92                    "type": "string",
93                    "description": "The message body to deliver."
94                }
95            },
96            "required": ["channel", "target", "message"]
97        })
98    }
99
100    async fn execute(&self, args: serde_json::Value) -> Result<ToolResult> {
101        let channel = args
102            .get("channel")
103            .and_then(|v| v.as_str())
104            .map(str::trim)
105            .filter(|v| !v.is_empty())
106            .ok_or_else(|| {
107                ::zeroclaw_log::record!(
108                    WARN,
109                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
110                        .with_outcome(::zeroclaw_log::EventOutcome::Failure)
111                        .with_attrs(::serde_json::json!({"param": "channel"})),
112                    "tool argument validation failed"
113                );
114
115                anyhow::Error::msg("Missing or empty 'channel' parameter")
116            })?
117            .to_string();
118        let target = args
119            .get("target")
120            .and_then(|v| v.as_str())
121            .map(str::trim)
122            .filter(|v| !v.is_empty())
123            .ok_or_else(|| {
124                ::zeroclaw_log::record!(
125                    WARN,
126                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
127                        .with_outcome(::zeroclaw_log::EventOutcome::Failure)
128                        .with_attrs(::serde_json::json!({"param": "target"})),
129                    "tool argument validation failed"
130                );
131
132                anyhow::Error::msg("Missing or empty 'target' parameter")
133            })?
134            .to_string();
135        let message = args
136            .get("message")
137            .and_then(|v| v.as_str())
138            .map(str::trim)
139            .filter(|v| !v.is_empty())
140            .ok_or_else(|| {
141                ::zeroclaw_log::record!(
142                    WARN,
143                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
144                        .with_outcome(::zeroclaw_log::EventOutcome::Failure)
145                        .with_attrs(::serde_json::json!({"param": "message"})),
146                    "tool argument validation failed"
147                );
148
149                anyhow::Error::msg("Missing or empty 'message' parameter")
150            })?
151            .to_string();
152
153        let fallback_channel_type = channel.split_once('.').map(|(t, _)| t);
154        let resolved = resolve_peer_set(&self.config, &self.sender_alias);
155
156        if !resolved.is_known_peer(&channel, &target)
157            && !fallback_channel_type
158                .is_some_and(|channel_type| resolved.is_known_peer(channel_type, &target))
159        {
160            return Ok(ToolResult {
161                success: false,
162                output: String::new(),
163                error: Some(format!(
164                    "target {target:?} is not on agent {alias:?}'s resolved peer set for channel {channel:?}; \
165                     add a [peer_groups.<name>] entry that lists both this agent and the target before sending",
166                    alias = self.sender_alias,
167                )),
168            });
169        }
170
171        // The agent must itself listen on the channel — the target may
172        // be reachable on it via a peer group, but a sender can't
173        // dispatch on a channel it isn't configured for.
174        let agent_listens_on_channel = self
175            .config
176            .agents
177            .get(&self.sender_alias)
178            .map(|a| a.channels.iter().any(|c| c.as_str() == channel.as_str()))
179            .unwrap_or(false);
180        if !agent_listens_on_channel {
181            return Ok(ToolResult {
182                success: false,
183                output: String::new(),
184                error: Some(format!(
185                    "agent {alias:?} does not list channel {channel:?} on its `channels`; \
186                     add the channel ref to [agents.{alias}.channels] before sending",
187                    alias = self.sender_alias,
188                )),
189            });
190        }
191
192        // Agent-alias targets route in-process. The channel's bot
193        // identity is shared between alpha and beta, so an outbound
194        // to the channel would loop right back into inbound and the
195        // self-loop guard would drop it. Agent-to-agent messaging is
196        // process-internal by design; the channel registry only sees
197        // sends with external recipients.
198        let target_norm = target.trim_start_matches('@').to_ascii_lowercase();
199        let target_is_agent = self
200            .config
201            .agents
202            .keys()
203            .any(|alias| alias.to_ascii_lowercase() == target_norm);
204
205        if target_is_agent {
206            // The target's resolved alias may differ in case from the
207            // raw input ("@Beta" -> "beta"). Look up the canonical
208            // alias once so the agent loop's `agent_alias` field
209            // matches the [agents.<alias>] config key.
210            let canonical = self
211                .config
212                .agents
213                .keys()
214                .find(|alias| alias.to_ascii_lowercase() == target_norm)
215                .cloned()
216                .unwrap_or_else(|| target.clone());
217
218            // Fire-and-forget: agent-to-agent peer messages do not
219            // synchronously block the sender on the recipient's full
220            // turn (that's what the SubAgent surface is for). The
221            // recipient processes on its own event loop and surfaces
222            // its result via its own observability.
223            let cfg = (*self.config).clone();
224            let sender = self.sender_alias.clone();
225            let recipient_alias = canonical.clone();
226            let body = message.clone();
227            zeroclaw_spawn::spawn!(async move {
228                if let Err(e) =
229                    crate::agent::loop_::process_message(cfg, &recipient_alias, &body, None).await
230                {
231                    ::zeroclaw_log::record!(WARN, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_outcome(::zeroclaw_log::EventOutcome::Unknown).with_attrs(::serde_json::json!({"sender": sender, "recipient": recipient_alias, "error": format!("{}", e)})), "peer-message in-process delivery failed");
232                }
233            });
234
235            return Ok(ToolResult {
236                success: true,
237                output: format!(
238                    "accepted for in-process delivery to peer agent {canonical:?} (recipient runs detached; observe its agent loop for the actual outcome)"
239                ),
240                error: None,
241            });
242        }
243
244        match deliver_announcement(&self.config, &channel, &target, None, &message).await {
245            Ok(()) => Ok(ToolResult {
246                success: true,
247                output: format!("delivered to external peer {target:?} on {channel}"),
248                error: None,
249            }),
250            Err(e) => Ok(ToolResult {
251                success: false,
252                output: String::new(),
253                error: Some(format!("delivery failed: {e:#}")),
254            }),
255        }
256    }
257}

259fn build_description() -> String {
260    String::from(
261        "Send a message to a peer agent or external peer (human, external bot) \
262         on a shared channel. The target must be a member of a peer group both \
263         this agent and the target agree on (or an external peer listed on the \
264         shared group's `external_peers`). Cross-agent sends to non-peers are \
265         rejected at the tool boundary; the channel send only happens after \
266         the peer-set check passes. Use the current channel ref unless the user \
267         explicitly names another allowed channel. Do not pass peer group names \
268         as the `channel` parameter.",
269    )
270}

272const MAX_PEER_MAP_CHANNELS: usize = 6;
273const MAX_PEER_MAP_ITEMS: usize = 8;
274const MAX_PEER_MAP_CHARS: usize = 4_000;
275const MAX_PROMPT_VALUE_CHARS: usize = 80;
276
277/// Render current-channel peer guidance for the channel system prompt.
278///
279/// Source of truth is [`Config`]: this materializes a per-turn view from the
280/// inbound channel ref rather than storing channel-scoped state in the static
281/// tool registry.
282#[must_use]
283pub fn render_sender_peer_map_for_channel(
284    config: &Config,
285    sender_alias: &str,
286    current_channel_ref: &str,
287) -> String {
288    let Some(agent) = config.agents.get(sender_alias) else {
289        return String::new();
290    };
291
292    let resolved = resolve_peer_set(config, sender_alias);
293    let agent_channels: Vec<&str> = agent.channels.iter().map(|c| c.as_str()).collect();
294    let channel_keys: std::collections::BTreeSet<String> = resolved
295        .agent_peers
296        .keys()
297        .chain(resolved.external_peers.keys())
298        .filter(|channel_key| channel_matches_group(current_channel_ref, channel_key))
299        .cloned()
300        .collect();
301    if channel_keys.is_empty() {
302        return String::new();
303    }
304
305    let total_channels = channel_keys.len();
306    let shown_channels: Vec<String> = channel_keys
307        .iter()
308        .take(MAX_PEER_MAP_CHANNELS)
309        .cloned()
310        .collect();
311
312    let mut rows = Vec::new();
313    for channel_key in shown_channels {
314        let channel_refs =
315            matching_current_channel_refs(&agent_channels, &channel_key, current_channel_ref);
316        let channel_hint = if channel_refs.is_empty() {
317            format!(
318                "current channel ref {} is not listed in this agent's configured channels",
319                prompt_value(current_channel_ref)
320            )
321        } else {
322            format!("use channel ref {}", format_prompt_list(&channel_refs))
323        };
324
325        let group_names = group_names_for_channel(config, sender_alias, &channel_key);
326        let agent_peers = agent_peers_for_current_channel(
327            config,
328            resolved.agent_peers.get(&channel_key),
329            current_channel_ref,
330        );
331        let external_peers = set_values(resolved.external_peers.get(&channel_key));
332
333        rows.push(format!(
334            "- Channel scope {}: {channel_hint}; peer groups: {}; agent peers: {}; external peers: {}.",
335            prompt_value(&channel_key),
336            format_prompt_list(&group_names),
337            format_prompt_list(&agent_peers),
338            format_prompt_list(&external_peers)
339        ));
340    }
341
342    let mut omitted_channels = total_channels.saturating_sub(MAX_PEER_MAP_CHANNELS);
343    let mut peer_map = format_peer_map(sender_alias, &rows, omitted_channels);
344    while peer_map.len() > MAX_PEER_MAP_CHARS && !rows.is_empty() {
345        rows.pop();
346        omitted_channels += 1;
347        peer_map = format_peer_map(sender_alias, &rows, omitted_channels);
348    }
349
350    peer_map
351}
352
353fn format_peer_map(sender_alias: &str, rows: &[String], omitted_channels: usize) -> String {
354    let mut row_text = rows.join("\n");
355    if omitted_channels > 0 {
356        if !row_text.is_empty() {
357            row_text.push('\n');
358        }
359        row_text.push_str(&format!(
360            "- {omitted_channels} more channel scopes omitted from this prompt map."
361        ));
362    }
363
364    format!(
365        "Current-channel peer map for agent {}:\n{}\n\
366         For collective wording like \"the whole team\" or a peer group name, send only to the listed agent peers unless the user explicitly names external peers. \
367         Do not pass peer group names as the `channel` parameter; pass one of the listed channel refs.",
368        prompt_value(sender_alias),
369        row_text
370    )
371}

373fn group_names_for_channel(config: &Config, sender_alias: &str, channel_key: &str) -> Vec<String> {
374    let mut names: Vec<String> = config
375        .peer_groups
376        .iter()
377        .filter(|(_, group)| {
378            group.channel == channel_key
379                && group
380                    .agents
381                    .iter()
382                    .any(|member| member.as_str() == sender_alias)
383        })
384        .map(|(name, _)| name.clone())
385        .collect();
386    names.sort();
387    names
388}

390fn set_values(set: Option<&std::collections::BTreeSet<String>>) -> Vec<String> {
391    set.map(|set| set.iter().cloned().collect())
392        .unwrap_or_default()
393}
394
395fn agent_peers_for_current_channel(
396    config: &Config,
397    set: Option<&std::collections::BTreeSet<String>>,
398    current_channel_ref: &str,
399) -> Vec<String> {
400    set.map(|set| {
401        set.iter()
402            .filter(|alias| agent_listens_on_channel_ref(config, alias, current_channel_ref))
403            .cloned()
404            .collect()
405    })
406    .unwrap_or_default()
407}
408
409fn agent_listens_on_channel_ref(config: &Config, alias: &str, channel_ref: &str) -> bool {
410    config
411        .agents
412        .iter()
413        .find(|(configured_alias, _)| configured_alias.eq_ignore_ascii_case(alias))
414        .is_some_and(|(_, agent)| agent.channels.iter().any(|channel| channel == channel_ref))
415}
416
417fn matching_current_channel_refs(
418    agent_channels: &[&str],
419    group_channel: &str,
420    current_channel_ref: &str,
421) -> Vec<String> {
422    if !channel_matches_group(current_channel_ref, group_channel) {
423        return Vec::new();
424    }
425
426    if agent_channels.contains(&current_channel_ref) {
427        return vec![current_channel_ref.to_string()];
428    }
429
430    Vec::new()
431}
432
433fn channel_matches_group(channel_ref: &str, group_channel: &str) -> bool {
434    if group_channel.contains('.') {
435        return channel_ref == group_channel;
436    }
437    channel_ref == group_channel
438        || channel_ref
439            .split_once('.')
440            .map(|(channel_type, _)| channel_type == group_channel)
441            .unwrap_or(false)
442}
443
444fn prompt_value(value: &str) -> String {
445    let mut chars = value.chars();
446    let mut value: String = chars.by_ref().take(MAX_PROMPT_VALUE_CHARS).collect();
447    if chars.next().is_some() {
448        value.push_str("...");
449    }
450    serde_json::to_string(&value).unwrap_or_else(|_| "\"<invalid>\"".to_string())
451}
452
453fn format_prompt_list(values: &[String]) -> String {
454    if values.is_empty() {
455        return "none".to_string();
456    }
457
458    let mut parts: Vec<String> = values
459        .iter()
460        .take(MAX_PEER_MAP_ITEMS)
461        .map(|value| prompt_value(value))
462        .collect();
463    if values.len() > MAX_PEER_MAP_ITEMS {
464        parts.push(format!("and {} more", values.len() - MAX_PEER_MAP_ITEMS));
465    }
466    parts.join(", ")
467}
