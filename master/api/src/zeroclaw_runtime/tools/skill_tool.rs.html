1//! Shell-based tool derived from a skill's `[[tools]]` section.
2//!
3//! Each `SkillTool` with `kind = "shell"` or `kind = "script"` is converted
4//! into a `SkillShellTool` that implements the `Tool` trait. The tool name is
5//! prefixed with the skill name (e.g. `my_skill__run_lint`) to avoid collisions
6//! with built-in tools. The `__` separator matches the MCP server prefix
7//! convention and keeps names valid under OpenAI-compatible function-name
8//! rules (`^[a-zA-Z0-9_-]+$`), which reject `.`.
9
10use crate::security::SecurityPolicy;
11use async_trait::async_trait;
12use std::collections::HashMap;
13use std::sync::Arc;
14use std::time::Duration;
15use zeroclaw_api::tool::{Tool, ToolResult};

17/// Default execution time for a skill shell command when the manifest does not
18/// set `timeout_secs` (seconds). A skill may raise this via `timeout_secs` in
19/// its SKILL.toml `[[tools]]` entry.
20const SKILL_SHELL_TIMEOUT_SECS: u64 = 60;
21/// Maximum output size in bytes (1 MB).
22const MAX_OUTPUT_BYTES: usize = 1_048_576;
23
24/// Maximum provider function-name length. Anthropic's current client-tool
25/// contract is the strictest we rely on: `name` must match
26/// `^[a-zA-Z0-9_-]{1,64}$`. The server error captured in #6678 mentioned
27/// `{1,128}`, but the published provider contract is 64, so we target the
28/// stricter bound rather than baking the looser observed string into the
29/// runtime.
30const MAX_TOOL_NAME_LEN: usize = 64;
31
32fn is_name_char(c: char) -> bool {
33    c.is_ascii_alphanumeric() || c == '_' || c == '-'
34}
35
36/// Dependency-free, build-stable 64-bit FNV-1a hash, rendered as 16 hex chars.
37/// Used only to disambiguate names that had to be altered. A bounded 64-bit
38/// hash reduces accidental collisions among sanitized names; it is not a
39/// uniqueness proof and cannot make the mapping injective.
40fn short_hash(s: &str) -> String {
41    let mut h: u64 = 0xcbf2_9ce4_8422_2325;
42    for b in s.bytes() {
43        h ^= b as u64;
44        h = h.wrapping_mul(0x0000_0100_0000_01b3);
45    }
46    format!("{h:016x}")
47}

49/// Compose a skill tool's provider-visible name (`skill__tool`) and route it
50/// through the single [`sanitize_tool_name`] rule. Every registration path
51/// (shell, script, builtin, HTTP) and the skills prompt must call this so the
52/// advertised tool spec and the name the model is told to invoke stay
53/// identical. Do not re-derive the `{skill}__{tool}` string anywhere else.
54///
55/// This guarantees provider-*validity*, not global *uniqueness*: the hash
56/// suffix only reduces accidental collisions, so registration must still treat
57/// two skills' composed names as potentially equal and resolve duplicates
58/// itself rather than assuming this function makes them distinct.
59pub(crate) fn composed_tool_name(skill_name: &str, tool_name: &str) -> String {
60    sanitize_tool_name(&format!("{skill_name}__{tool_name}"))
61}
62
63/// Sanitize a composed skill tool name so it satisfies provider function-name
64/// rules (`^[a-zA-Z0-9_-]{1,64}$`). The `__` separator is already safe, but a
65/// skill or tool name can itself contain illegal characters: dots, spaces, or
66/// colons (plugin-namespaced skills such as `pr-review-toolkit:code-reviewer`),
67/// or non-ASCII. Anthropic rejects non-conforming names outright (issue #6678).
68///
69/// Names that are already valid and within length are returned unchanged. Any
70/// name that must be altered (illegal characters or over-length) gets every
71/// disallowed character mapped to `_` and a short stable hash of the original
72/// composed name appended within the 64-char budget. The hash disambiguates
73/// common collisions: distinct inputs that would otherwise collapse to the same
74/// string, such as `a.b__run` vs `a:b__run`, or two tools under one skill name
75/// longer than 64 chars whose suffix would be truncated away, stay distinct. It
76/// is a strong reducer of accidental collisions, not a guarantee of injectivity.
77fn sanitize_tool_name(raw: &str) -> String {
78    let already_valid =
79        !raw.is_empty() && raw.len() <= MAX_TOOL_NAME_LEN && raw.chars().all(is_name_char);
80    if already_valid {
81        return raw.to_string();
82    }
83
84    let mapped: String = raw
85        .chars()
86        .map(|c| if is_name_char(c) { c } else { '_' })
87        .collect();
88
89    // Reserve room for `_<16 hex>` so the disambiguating hash always survives.
90    let suffix = format!("_{}", short_hash(raw));
91    let budget = MAX_TOOL_NAME_LEN - suffix.len();
92    let head: String = mapped.chars().take(budget).collect();
93    format!("{head}{suffix}")
94}

96/// A tool derived from a skill's `[[tools]]` section that executes shell commands.
97pub struct SkillShellTool {
98    tool_name: String,
99    tool_description: String,
100    command_template: String,
101    args: HashMap<String, String>,
102    security: Arc<SecurityPolicy>,
103    /// Resolved per-command timeout in seconds (manifest `timeout_secs`, or the
104    /// `SKILL_SHELL_TIMEOUT_SECS` default), clamped to a minimum of 1.
105    timeout_secs: u64,
106}
107
108impl SkillShellTool {
109    /// Create a new skill shell tool.
110    ///
111    /// The tool name is prefixed with the skill name (`skill_name__tool_name`)
112    /// to prevent collisions with built-in tools.
113    pub fn new(
114        skill_name: &str,
115        tool: &crate::skills::SkillTool,
116        security: Arc<SecurityPolicy>,
117    ) -> Self {
118        Self {
119            tool_name: composed_tool_name(skill_name, &tool.name),
120            tool_description: tool.description.clone(),
121            command_template: tool.command.clone(),
122            args: tool.args.clone(),
123            security,
124            timeout_secs: tool.timeout_secs.unwrap_or(SKILL_SHELL_TIMEOUT_SECS).max(1),
125        }
126    }
127
128    fn build_parameters_schema(&self) -> serde_json::Value {
129        let mut properties = serde_json::Map::new();
130        let mut required = Vec::new();
131
132        for (name, description) in &self.args {
133            properties.insert(
134                name.clone(),
135                serde_json::json!({
136                    "type": "string",
137                    "description": description
138                }),
139            );
140            required.push(serde_json::Value::String(name.clone()));
141        }
142
143        serde_json::json!({
144            "type": "object",
145            "properties": properties,
146            "required": required
147        })
148    }
149
150    /// Substitute `{{arg_name}}` placeholders in the command template with
151    /// the provided argument values. Unknown placeholders are left as-is.
152    fn substitute_args(&self, args: &serde_json::Value) -> String {
153        let mut command = self.command_template.clone();
154        if let Some(obj) = args.as_object() {
155            for (key, value) in obj {
156                let placeholder = format!("{{{{{}}}}}", key);
157                let replacement = value.as_str().unwrap_or_default();
158                command = command.replace(&placeholder, replacement);
159            }
160        }
161        command
162    }
163}

165#[async_trait]
166impl Tool for SkillShellTool {
167    fn name(&self) -> &str {
168        &self.tool_name
169    }
170
171    fn description(&self) -> &str {
172        &self.tool_description
173    }
174
175    fn parameters_schema(&self) -> serde_json::Value {
176        self.build_parameters_schema()
177    }
178
179    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
180        let command = self.substitute_args(&args);
181
182        // Rate limiting is applied by the RateLimitedTool wrapper at
183        // registration time (see zeroclaw-runtime::tools::mod). The
184        // PathGuardedTool wrapper cannot inspect the substituted command
185        // built by substitute_args, so the forbidden_path_argument check
186        // below remains tool-local.
187
188        // Security validation — always requires explicit approval (approved=true)
189        // since skill tools are user-defined and should be treated as medium-risk.
190        match self.security.validate_command_execution(&command, true) {
191            Ok(_) => {}
192            Err(reason) => {
193                return Ok(ToolResult {
194                    success: false,
195                    output: String::new(),
196                    error: Some(reason),
197                });
198            }
199        }
200
201        if let Some(path) = self.security.forbidden_path_argument(&command) {
202            return Ok(ToolResult {
203                success: false,
204                output: String::new(),
205                error: Some(format!("Path blocked by security policy: {path}")),
206            });
207        }
208
209        // Build and execute the command
210        let mut cmd = tokio::process::Command::new("sh");
211        cmd.arg("-c").arg(&command);
212        cmd.current_dir(&self.security.workspace_dir);
213        cmd.env_clear();
214
215        // Only pass safe environment variables
216        for var in &[
217            "PATH", "HOME", "TERM", "LANG", "LC_ALL", "USER", "SHELL", "TMPDIR",
218        ] {
219            if let Ok(val) = std::env::var(var) {
220                cmd.env(var, val);
221            }
222        }
223
224        let result =
225            tokio::time::timeout(Duration::from_secs(self.timeout_secs), cmd.output()).await;
226
227        match result {
228            Ok(Ok(output)) => {
229                let mut stdout = String::from_utf8_lossy(&output.stdout).to_string();
230                let mut stderr = String::from_utf8_lossy(&output.stderr).to_string();
231
232                if stdout.len() > MAX_OUTPUT_BYTES {
233                    let mut b = MAX_OUTPUT_BYTES.min(stdout.len());
234                    while b > 0 && !stdout.is_char_boundary(b) {
235                        b -= 1;
236                    }
237                    stdout.truncate(b);
238                    stdout.push_str("\n... [output truncated at 1MB]");
239                }
240                if stderr.len() > MAX_OUTPUT_BYTES {
241                    let mut b = MAX_OUTPUT_BYTES.min(stderr.len());
242                    while b > 0 && !stderr.is_char_boundary(b) {
243                        b -= 1;
244                    }
245                    stderr.truncate(b);
246                    stderr.push_str("\n... [stderr truncated at 1MB]");
247                }
248
249                Ok(ToolResult {
250                    success: output.status.success(),
251                    output: stdout,
252                    error: if stderr.is_empty() {
253                        None
254                    } else {
255                        Some(stderr)
256                    },
257                })
258            }
259            Ok(Err(e)) => Ok(ToolResult {
260                success: false,
261                output: String::new(),
262                error: Some(format!("Failed to execute command: {e}")),
263            }),
264            Err(_) => Ok(ToolResult {
265                success: false,
266                output: String::new(),
267                error: Some(format!(
268                    "Command timed out after {}s and was killed",
269                    self.timeout_secs
270                )),
271            }),
272        }
273    }
274}

278/// A skill tool that delegates execution to another tool resolved from the
279/// resolution registry — either a built-in (`kind = "builtin"`) or an MCP tool
280/// (`kind = "mcp"`). This is the skill-scoped tool elevation mechanism: a
281/// policy blocking `shell` by name (or deferred MCP tools hidden from the
282/// model) does not block `my_skill__use_shell`, because the wrapper is
283/// registered under the prefixed name `{skill}__{tool}` and delegates to the
284/// resolved target.
285///
286/// `locked_args` are arguments fixed by the manifest. They are applied **on top
287/// of** the caller-supplied args (the caller cannot override them) and are
288/// stripped from the advertised parameter schema, so the model can neither see
289/// nor change them. This is what scopes a delegated tool — e.g.
290/// `target = "composio"` + `locked_args = { action_name = "TEXT_TO_PDF" }`
291/// exposes exactly one action, and `target = "images__generate"` exposes a
292/// single MCP capability.
293pub struct SkillBuiltinTool {
294    tool_name: String,
295    tool_description: String,
296    target_tool: Arc<dyn zeroclaw_api::tool::Tool>,
297    locked_args: serde_json::Map<String, serde_json::Value>,
298    /// Target schema with the locked keys removed (precomputed at construction).
299    advertised_schema: serde_json::Value,
300}
301
302impl SkillBuiltinTool {
303    /// Create a new skill elevation tool delegating to `target_tool`.
304    ///
305    /// `target_tool` is the resolved built-in or MCP tool (looked up from the
306    /// resolution registry at registration time). `locked_args` are fixed by
307    /// the manifest: applied over caller args (non-overridable) and hidden from
308    /// the advertised schema.
309    pub fn new(
310        skill_name: &str,
311        tool: &crate::skills::SkillTool,
312        target_tool: Arc<dyn zeroclaw_api::tool::Tool>,
313        locked_args: HashMap<String, String>,
314    ) -> Self {
315        let locked: serde_json::Map<String, serde_json::Value> = locked_args
316            .into_iter()
317            .map(|(k, v)| (k, serde_json::Value::String(v)))
318            .collect();
319        let advertised_schema = narrow_schema(target_tool.parameters_schema(), &locked);
320        Self {
321            tool_name: composed_tool_name(skill_name, &tool.name),
322            tool_description: tool.description.clone(),
323            target_tool,
324            locked_args: locked,
325            advertised_schema,
326        }
327    }
328}
329
330/// Merge caller args with manifest `locked` args. Locked args ALWAYS win — the
331/// caller cannot override a scope key — but the caller may add other keys.
332fn merge_locked_args(
333    locked: &serde_json::Map<String, serde_json::Value>,
334    caller: serde_json::Value,
335) -> serde_json::Value {
336    if locked.is_empty() {
337        return caller;
338    }
339    let mut merged = match caller {
340        serde_json::Value::Object(map) => map,
341        _ => serde_json::Map::new(),
342    };
343    for (k, v) in locked {
344        merged.insert(k.clone(), v.clone());
345    }
346    serde_json::Value::Object(merged)
347}
348
349/// Remove `locked` keys from an advertised JSON-schema object so the model
350/// neither sees nor tries to set keys the manifest fixes. Non-object schemas
351/// (or those without `properties`) pass through unchanged.
352fn narrow_schema(
353    schema: serde_json::Value,
354    locked: &serde_json::Map<String, serde_json::Value>,
355) -> serde_json::Value {
356    if locked.is_empty() {
357        return schema;
358    }
359    let serde_json::Value::Object(mut obj) = schema else {
360        return schema;
361    };
362    if let Some(serde_json::Value::Object(props)) = obj.get_mut("properties") {
363        for k in locked.keys() {
364            props.remove(k);
365        }
366    }
367    if let Some(serde_json::Value::Array(required)) = obj.get_mut("required") {
368        required.retain(|v| v.as_str().is_none_or(|s| !locked.contains_key(s)));
369    }
370    serde_json::Value::Object(obj)
371}

373#[async_trait]
374impl Tool for SkillBuiltinTool {
375    fn name(&self) -> &str {
376        &self.tool_name
377    }
378
379    fn description(&self) -> &str {
380        &self.tool_description
381    }
382
383    fn parameters_schema(&self) -> serde_json::Value {
384        self.advertised_schema.clone()
385    }
386
387    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
388        // Audit: elevated skill tools delegate to a target that may be blocked
389        // by SecurityPolicy or hidden from the model. Record every invocation
390        // at INFO with the delegation target and the locked scope keys.
391        ::zeroclaw_log::record!(
392            INFO,
393            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Invoke)
394                .with_category(::zeroclaw_log::EventCategory::Tool)
395                .with_attrs(::serde_json::json!({
396                    "skill_tool": self.tool_name,
397                    "delegates_to": self.target_tool.name(),
398                    "locked_keys": self.locked_args.keys().collect::<Vec<_>>(),
399                })),
400            "skill-scoped elevated tool invoked"
401        );
402        let merged = merge_locked_args(&self.locked_args, args);
403        self.target_tool.execute(merged).await
404    }
405}

1032    #[test]
1033    fn elevation_wrapper_survives_policy_filter_that_blocks_raw_target() {
1034        // The trust-boundary contract (#6915): a SecurityPolicy blocking the
1035        // raw tool by name must keep it out of the model-visible registry,
1036        // while the skill's scoped wrapper — registered under the prefixed
1037        // name — remains the only callable path to that capability.
1038        use crate::skills::{Skill, SkillTool};
1039
1040        let shell: Arc<dyn Tool> = Arc::new(MockBuiltinTool::new("shell"));
1041        let file_read: Arc<dyn Tool> = Arc::new(MockBuiltinTool::new("file_read"));
1042        // The resolution registry retains the raw tool so the wrapper can
1043        // delegate to it even after the policy filter removes it below.
1044        let resolution: Vec<Arc<dyn Tool>> = vec![Arc::clone(&shell), Arc::clone(&file_read)];
1045
1046        let mut registry: Vec<Box<dyn Tool>> = vec![
1047            Box::new(crate::tools::ArcToolRef(Arc::clone(&shell))),
1048            Box::new(crate::tools::ArcToolRef(Arc::clone(&file_read))),
1049        ];
1050        let policy = SecurityPolicy {
1051            excluded_tools: Some(vec!["shell".to_string()]),
1052            workspace_dir: std::env::temp_dir(),
1053            ..SecurityPolicy::default()
1054        };
1055        crate::agent::loop_::apply_policy_tool_filter(&mut registry, Some(&policy), None);
1056        assert!(
1057            !registry.iter().any(|t| t.name() == "shell"),
1058            "raw shell must be blocked by the policy filter"
1059        );
1060
1061        let skill = Skill {
1062            name: "ops".to_string(),
1063            description: "d".to_string(),
1064            version: "1".to_string(),
1065            author: None,
1066            tags: vec![],
1067            tools: vec![SkillTool {
1068                name: "use_shell".to_string(),
1069                description: "scoped shell".to_string(),
1070                kind: "builtin".to_string(),
1071                command: String::new(),
1072                args: HashMap::new(),
1073                target: Some("shell".to_string()),
1074                locked_args: HashMap::new(),
1075                timeout_secs: None,
1076            }],
1077            prompts: vec![],
1078            location: None,
1079        };
1080        crate::tools::register_skill_tools_with_context(
1081            &mut registry,
1082            &[skill],
1083            test_security(),
1084            &resolution,
1085        );
1086
1087        assert!(
1088            !registry.iter().any(|t| t.name() == "shell"),
1089            "raw shell must STILL be unavailable after skill registration"
1090        );
1091        assert!(
1092            registry.iter().any(|t| t.name() == "ops__use_shell"),
1093            "the scoped elevation wrapper must be the only callable path"
1094        );
