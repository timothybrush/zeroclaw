1//! Docker tag matrix generator. Emits the canonical `[[tags]]` table the
2//! docker-publish workflow consumes. `flags` carries the resolved cargo flag
3//! string and is the authoritative build input; `features` is human-readable
4//! only. Fully generated.
5
6use super::spec::{self, Selection};
7use std::path::Path;
8
9/// A published image tag bound to a selection and the dockerfile that builds it.
10struct TagSpec {
11    stem: &'static str,
12    selection: Selection,
13    dockerfile: &'static str,
14}
15
16/// The canonical tag set. Each selection maps to a stem + the dockerfile that
17/// realizes it. Driven by selections, not literal tag strings.
18fn tag_specs() -> Vec<TagSpec> {
19    vec![
20        TagSpec {
21            stem: "minimal",
22            selection: Selection::Minimal,
23            dockerfile: "Dockerfile",
24        },
25        TagSpec {
26            stem: "default-features",
27            selection: Selection::Full,
28            dockerfile: "Dockerfile",
29        },
30        TagSpec {
31            stem: "dist",
32            selection: Selection::Dist,
33            dockerfile: "Dockerfile",
34        },
35        TagSpec {
36            stem: "all-features",
37            selection: Selection::All,
38            dockerfile: "Containerfile",
39        },
40    ]
41}
42
43/// Render the docker tag matrix TOML. `flags` is the authoritative build
44/// input; `features` is human-readable only.
45pub fn render(root: &Path) -> anyhow::Result<String> {
46    let version = spec::resolve_version(root)?;
47    let mut out = String::new();
48    out.push_str("# Generated by `cargo generate installers` - do not edit.\n");
49    out.push_str("# Canonical docker tag matrix. Regenerate after a version or\n");
50    out.push_str("# feature change; CI fails on drift.\n");
51    out.push_str(&format!("version = \"{version}\"\n\n"));
52    for ts in tag_specs() {
53        let features = spec::resolve_feature_list(root, &ts.selection)?.join(",");
54        let flags = spec::resolve_flags(root, &ts.selection)?;
55        out.push_str("[[tags]]\n");
56        out.push_str(&format!("stem = \"{}\"\n", ts.stem));
57        out.push_str(&format!("pinned_tag = \"{}-v{version}\"\n", ts.stem));
58        out.push_str(&format!("floating_tag = \"{}\"\n", ts.stem));
59        out.push_str(&format!("dockerfile = \"{}\"\n", ts.dockerfile));
60        out.push_str(&format!("flags = \"{flags}\"\n"));
61        out.push_str(&format!("features = \"{features}\"\n\n"));
62    }
63    while out.ends_with('\n') {
64        out.pop();
65    }
66    out.push('\n');
67    Ok(out)
68}
69
70/// Render-or-check: the matrix is a fully generated file (no hand-written glue),
71/// so the renderer ignores prior content and emits the whole document.
72pub fn render_file(root: &Path, _current: &str) -> anyhow::Result<String> {
73    render(root)
74}

76#[cfg(test)]
77mod tests {
78    use super::*;
79    use std::path::PathBuf;
80
81    fn root() -> PathBuf {
82        PathBuf::from(env!("CARGO_MANIFEST_DIR"))
83            .parent()
84            .unwrap()
85            .to_path_buf()
86    }
87
88    #[test]
89    fn matrix_is_valid_toml_with_pinned_and_floating() {
90        let s = render(&root()).unwrap();
91        let v: toml::Value = toml::from_str(&s).expect("valid TOML");
92        let tags = v["tags"].as_array().unwrap();
93        assert_eq!(tags.len(), tag_specs().len());
94        let ver = v["version"].as_str().unwrap();
95        let dist = tags
96            .iter()
97            .find(|t| t["stem"].as_str() == Some("dist"))
98            .unwrap();
99        assert_eq!(dist["pinned_tag"].as_str().unwrap(), format!("dist-v{ver}"));
100        assert_eq!(dist["floating_tag"].as_str().unwrap(), "dist");
101    }
102
103    #[test]
104    fn minimal_carries_no_default_flag_default_features_does_not() {
105        let s = render(&root()).unwrap();
106        let v: toml::Value = toml::from_str(&s).unwrap();
107        let tags = v["tags"].as_array().unwrap();
108        let flags = |stem: &str| {
109            tags.iter()
110                .find(|t| t["stem"].as_str() == Some(stem))
111                .unwrap()["flags"]
112                .as_str()
113                .unwrap()
114                .to_string()
115        };
116        assert_eq!(flags("minimal"), "--no-default-features");
117        assert_eq!(flags("default-features"), "");
118        assert!(flags("dist").starts_with("--no-default-features --features"));
119    }
120
121    #[test]
122    fn dist_tag_ships_channels_all_tag_ships_heavyweight() {
123        let s = render(&root()).unwrap();
124        let v: toml::Value = toml::from_str(&s).unwrap();
125        let tags = v["tags"].as_array().unwrap();
126        let dist = tags
127            .iter()
128            .find(|t| t["stem"].as_str() == Some("dist"))
129            .unwrap();
130        let all = tags
131            .iter()
132            .find(|t| t["stem"].as_str() == Some("all-features"))
133            .unwrap();
134        assert!(
135            dist["features"]
136                .as_str()
137                .unwrap()
138                .contains("channel-discord")
139        );
140        assert!(!dist["features"].as_str().unwrap().contains("hardware"));
141        assert!(all["features"].as_str().unwrap().contains("hardware"));
142    }
