1//! Shared renderer for container/build surfaces (Containerfile, Dockerfile,
2//! Dockerfile.debian) that inject a cargo `--features` line per build stage.
3//! Each injection point is a sentinel-delimited zone naming the `Selection` it
4//! renders; the feature list is resolved from the canonical spec, never typed.
5//!
6//! Hash-comment syntax (`#`) for Dockerfile/Containerfile.
7
8use super::spec::{self, Selection};
9use std::path::Path;
10
11fn begin(zone: &str) -> String {
12    format!("# >>> generated:{zone} by `cargo generate installers` - do not edit <<<")
13}
14fn end(zone: &str) -> String {
15    format!("# >>> end generated:{zone} <<<")
16}
17
18/// Render the feature-arg body for a zone: a `ZEROCLAW_FEATURES="X,Y"`
19/// assignment the surrounding `cargo build` references as
20/// `--features "${ZEROCLAW_FEATURES}"`. Using a variable (rather than injecting
21/// a `--features` line mid backslash-continuation) keeps the generated zone a
22/// standalone statement, so sentinel comments never sit inside a continued
23/// command - which would break the shell parse and the StageX `--frozen` build.
24pub fn render_features(
25    manifest_dir: &Path,
26    selection: &Selection,
27    indent: &str,
28) -> anyhow::Result<String> {
29    let list = spec::resolve_feature_list(manifest_dir, selection)?;
30    Ok(format!("{indent}ZEROCLAW_FEATURES=\"{}\"", list.join(",")))
31}
32
33/// Render an `ARG ZEROCLAW_CARGO_FLAGS="..."` default line from a selection.
34/// The flag string (`--no-default-features [--features ...]` or empty for the
35/// Cargo default) is the only form that distinguishes `minimal` from the
36/// default-features build. Build-time overridable; only its default is
37/// canonical. Resolved, never typed.
38pub fn render_features_arg(manifest_dir: &Path, selection: &Selection) -> anyhow::Result<String> {
39    let flags = spec::resolve_flags(manifest_dir, selection)?;
40    Ok(format!("ARG ZEROCLAW_CARGO_FLAGS=\"{flags}\""))
41}
42
43/// Splice a named zone's body into `current`, preserving everything else.
44pub fn splice(current: &str, zone: &str, body: &str) -> anyhow::Result<String> {
45    let b = begin(zone);
46    let e = end(zone);
47    let begin_at = current
48        .find(&b)
49        .ok_or_else(|| anyhow::Error::msg(format!("missing generated:{zone} BEGIN sentinel")))?;
50    let after_begin = begin_at + b.len();
51    let end_rel = current[after_begin..]
52        .find(&e)
53        .ok_or_else(|| anyhow::Error::msg(format!("missing generated:{zone} END sentinel")))?;
54    let end_at = after_begin + end_rel;
55    let mut out = String::new();
56    out.push_str(&current[..after_begin]);
57    out.push('\n');
58    out.push_str(body);
59    out.push('\n');
60    out.push_str(&current[end_at..]);
61    Ok(out)
62}
63
64/// A container surface = a file plus its named feature zones, each bound to a
65/// selection. Drives both render and check generically.
66pub struct ContainerSurface {
67    pub file: &'static str,
68    /// (zone name, selection, indent) per injection point.
69    pub zones: Vec<(&'static str, Selection, &'static str)>,
70}
71
72impl ContainerSurface {
73    pub fn render(&self, manifest_dir: &Path, current: &str) -> anyhow::Result<String> {
74        let mut out = current.to_string();
75        for (zone, sel, indent) in &self.zones {
76            let body = render_features(manifest_dir, sel, indent)?;
77            out = splice(&out, zone, &body)?;
78        }
79        Ok(out)
80    }
81}

83#[cfg(test)]
84mod tests {
85    use super::*;
86    use std::path::PathBuf;
87
88    fn root() -> PathBuf {
89        PathBuf::from(env!("CARGO_MANIFEST_DIR"))
90            .parent()
91            .unwrap()
92            .to_path_buf()
93    }
94
95    #[test]
96    fn full_renders_explicit_default_leaves() {
97        let b = render_features(&root(), &Selection::Full, "    ").unwrap();
98        // Full emits the explicit resolved default leaves as a ZEROCLAW_FEATURES
99        // assignment (drift-checkable), not a bare comment.
100        assert!(b.contains("ZEROCLAW_FEATURES="));
101        assert!(b.contains("gateway"), "default includes gateway");
102    }
103
104    #[test]
105    fn dist_renders_all_channels() {
106        let b = render_features(&root(), &Selection::Dist, "        ").unwrap();
107        assert!(b.contains("channel-discord") && !b.contains("hardware"));
108    }
109
110    #[test]
111    fn all_renders_kitchen_sink() {
112        let b = render_features(&root(), &Selection::All, "        ").unwrap();
113        assert!(b.contains("hardware") && b.contains("channel-matrix"));
114    }
115
116    #[test]
117    fn splice_named_zone() {
118        let cur = format!("X\n{}\nOLD\n{}\nY\n", begin("z1"), end("z1"));
119        let out = splice(&cur, "z1", "NEW").unwrap();
120        assert!(out.contains("NEW") && !out.contains("OLD"));
121    }
122}
