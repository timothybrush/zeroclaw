1//! install.sh renderer. install.sh@HEAD is the behavioral reference and stays
2//! hand-authored; we generate one sentinel zone for the source-build
3//! cargo-install step so its dry-run narration and real command come from the
4//! canonical spec and cannot drift. All other install.sh logic (arg parsing,
5//! toolchain bootstrap, web-dist copy, apps, summary) stays outside the
6//! sentinels, unchanged.
7
8use super::spec::{self, Selection};
9use std::path::Path;
10
11fn begin(zone: &str) -> String {
12    format!("  # >>> generated:{zone} by `cargo generate installers` - do not edit <<<")
13}
14fn end(zone: &str) -> String {
15    format!("  # >>> end generated:{zone} <<<")
16}
17
18const ZONE_CARGO_INSTALL: &str = "source-cargo-install";
19
20/// Render the source-build cargo-install step dispatcher, matching install.sh's
21/// style (2-space indent, shellcheck pragmas), with the command derived from the
22/// canonical spec. Dry-run narration and execute action are one source.
23fn render_cargo_install(root: &Path) -> anyhow::Result<String> {
24    // Resolve to confirm the spec is readable; the cargo flags are interpolated
25    // at runtime via $CARGO_FLAGS, not baked, so the body is selection-stable.
26    let _ = spec::resolve_flags(root, &Selection::Full)?;
27    Ok([
28        "  if [ \"$DRY_RUN\" = true ]; then",
29        "    # shellcheck disable=SC2086",
30        "    info \"[dry-run] Would run: cargo install --path . --locked --force $CARGO_FLAGS\"",
31        "  else",
32        "    # shellcheck disable=SC2086",
33        "    cargo install --path . --locked --force $CARGO_FLAGS",
34        "  fi",
35    ]
36    .join("\n"))
37}
38
39/// Splice generated step zones into install.sh, leaving hand-written glue
40/// untouched.
41pub fn render_file(root: &Path, current: &str) -> anyhow::Result<String> {
42    splice(current, ZONE_CARGO_INSTALL, &render_cargo_install(root)?)
43}
44
45fn splice(current: &str, zone: &str, body: &str) -> anyhow::Result<String> {
46    let b = begin(zone);
47    let e = end(zone);
48    let begin_at = current.find(&b).ok_or_else(|| {
49        anyhow::Error::msg(format!(
50            "install.sh missing generated:{zone} BEGIN sentinel"
51        ))
52    })?;
53    let after_begin = begin_at + b.len();
54    let end_rel = current[after_begin..].find(&e).ok_or_else(|| {
55        anyhow::Error::msg(format!("install.sh missing generated:{zone} END sentinel"))
56    })?;
57    let end_at = after_begin + end_rel;
58    let mut out = String::new();
59    out.push_str(&current[..after_begin]);
60    out.push('\n');
61    out.push_str(body);
62    out.push('\n');
63    out.push_str(&current[end_at..]);
64    Ok(out)
65}

67#[cfg(test)]
68mod tests {
69    use super::*;
70    use std::path::PathBuf;
71
72    fn root() -> PathBuf {
73        PathBuf::from(env!("CARGO_MANIFEST_DIR"))
74            .parent()
75            .unwrap()
76            .to_path_buf()
77    }
78
79    #[test]
80    fn cargo_install_zone_has_dryrun_and_execute() {
81        let z = render_cargo_install(&root()).unwrap();
82        assert!(z.contains("[dry-run] Would run: cargo install"));
83        assert!(z.contains("cargo install --path . --locked --force $CARGO_FLAGS"));
84        assert!(z.contains("if [ \"$DRY_RUN\" = true ]; then"));
85    }
86
87    #[test]
88    fn render_file_is_idempotent_against_real_install_sh() {
89        let cur = std::fs::read_to_string(root().join("install.sh")).unwrap();
90        let once = render_file(&root(), &cur).unwrap();
91        let twice = render_file(&root(), &once).unwrap();
92        assert_eq!(once, twice, "render must be idempotent");
93    }
