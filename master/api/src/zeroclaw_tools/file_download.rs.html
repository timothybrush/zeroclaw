16pub struct FileDownloadTool {
17    security: Arc<SecurityPolicy>,
18    config: FileDownloadConfig,
19    /// Whether the downloaded file persists on the host filesystem. `false` on
20    /// an ephemeral runtime (Docker tmpfs / no volume mount), where the file is
21    /// written inside the container but invisible on the host and discarded at
22    /// session end. When `false`, a successful download carries a loud
23    /// ephemeral-workspace warning. Mirrors
24    /// [`super::file_write::FileWriteTool`]. See issue #4627.
25    persistent_writes: bool,
26}

52    /// Stream a response body into `temp_path`, treating `max_bytes` as a hard
53    /// ceiling so an unbounded or oversized body never fully buffers in memory.
54    /// Returns the number of bytes written, or an error message. The caller is
55    /// responsible for removing `temp_path` on any error.
56    async fn stream_to_temp(
57        response: reqwest::Response,
58        temp_path: &Path,
59        max_bytes: u64,
60    ) -> Result<u64, String> {
61        let mut file = tokio::fs::File::create(temp_path).await.map_err(|e| {
62            Self::tool_msg_with_args(
63                "tool-file-download-error-temp-create",
64                &[("err", &e.to_string())],
65            )
66        })?;
67
68        let mut stream = response.bytes_stream();
69        let mut written: u64 = 0;
70        while let Some(chunk) = stream.next().await {
71            let chunk = chunk.map_err(|e| {
72                Self::tool_msg_with_args(
73                    "tool-file-download-error-read-body",
74                    &[("err", &e.to_string())],
75                )
76            })?;
77            written = written.saturating_add(chunk.len() as u64);
78            if written > max_bytes {
79                let limit = max_bytes.to_string();
80                return Err(Self::tool_msg_with_args(
81                    "tool-file-download-error-too-large-stream",
82                    &[("limit", &limit)],
83                ));
84            }
85            file.write_all(&chunk).await.map_err(|e| {
86                Self::tool_msg_with_args(
87                    "tool-file-download-error-write-body",
88                    &[("err", &e.to_string())],
89                )
90            })?;
91        }
92
93        file.flush().await.map_err(|e| {
94            Self::tool_msg_with_args("tool-file-download-error-flush", &[("err", &e.to_string())])
95        })?;
96        Ok(written)
97    }

103    fn tool_msg_with_args(key: &str, args: &[(&str, &str)]) -> String {
104        crate::i18n::get_required_tool_string_with_args(key, args)
105    }

120    fn parameters_schema(&self) -> serde_json::Value {
121        json!({
122            "type": "object",
123            "properties": {
124                "document_id": {
125                    "type": "string",
126                    "description": Self::tool_msg("tool-file-download-param-document-id")
127                },
128                "dest_path": {
129                    "type": "string",
130                    "description": Self::tool_msg("tool-file-download-param-dest-path")
131                }
132            },
133            "required": ["document_id", "dest_path"]
134        })
135    }
136
137    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
138        let Some(url) = self
139            .config
140            .url
141            .as_deref()
142            .map(str::trim)
143            .filter(|s| !s.is_empty())
144        else {
145            return Ok(ToolResult {
146                success: false,
147                output: String::new(),
148                error: Some(Self::tool_msg("tool-file-download-error-disabled")),
149            });
150        };
151
152        if !self.security.can_act() {
153            return Ok(ToolResult {
154                success: false,
155                output: String::new(),
156                error: Some(Self::tool_msg("tool-file-download-error-read-only")),
157            });
158        }
159
160        if self.security.is_rate_limited() {
161            return Ok(ToolResult {
162                success: false,
163                output: String::new(),
164                error: Some(Self::tool_msg("tool-file-download-error-rate-limited-hour")),
165            });
166        }
167
168        let document_id = args
169            .get("document_id")
170            .and_then(|v| v.as_str())
171            .map(str::trim)
172            .filter(|s| !s.is_empty())
173            .ok_or_else(|| {
174                ::zeroclaw_log::record!(
175                    WARN,
176                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
177                        .with_outcome(::zeroclaw_log::EventOutcome::Failure)
178                        .with_attrs(::serde_json::json!({"param": "document_id"})),
179                    "file_download: missing document_id parameter"
180                );
181                anyhow::Error::msg(Self::tool_msg(
182                    "tool-file-download-error-missing-document-id",
183                ))
184            })?;
185
186        let dest_path = args
187            .get("dest_path")
188            .and_then(|v| v.as_str())
189            .map(str::trim)
190            .filter(|s| !s.is_empty())
191            .ok_or_else(|| {
192                ::zeroclaw_log::record!(
193                    WARN,
194                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
195                        .with_outcome(::zeroclaw_log::EventOutcome::Failure)
196                        .with_attrs(::serde_json::json!({"param": "dest_path"})),
197                    "file_download: missing dest_path parameter"
198                );
199                anyhow::Error::msg(Self::tool_msg("tool-file-download-error-missing-dest-path"))
200            })?;
201
202        // The downloaded bytes are attacker-influenceable, so the write target
203        // must resolve inside the workspace allowlist before any network call.
204        let full = self.security.resolve_tool_path(dest_path);
205
206        let file_name = match full.file_name().and_then(|s| s.to_str()) {
207            Some(name) if name != "." && name != ".." => name.to_string(),
208            _ => {
209                return Ok(ToolResult {
210                    success: false,
211                    output: String::new(),
212                    error: Some(Self::tool_msg_with_args(
213                        "tool-file-download-error-invalid-file-name",
214                        &[("dest_path", dest_path)],
215                    )),
216                });
217            }
218        };
219
220        let Some(parent) = full.parent() else {
221            return Ok(ToolResult {
222                success: false,
223                output: String::new(),
224                error: Some(Self::tool_msg_with_args(
225                    "tool-file-download-error-no-parent",
226                    &[("dest_path", dest_path)],
227                )),
228            });
229        };
230
231        // Canonicalize the parent (which must already exist) so a symlinked
232        // parent cannot redirect the write outside the workspace. `full` itself
233        // does not exist yet, so it is never canonicalized.
234        let canonical_parent = match tokio::fs::canonicalize(parent).await {
235            Ok(p) => p,
236            Err(e) => {
237                return Ok(ToolResult {
238                    success: false,
239                    output: String::new(),
240                    error: Some(Self::tool_msg_with_args(
241                        "tool-file-download-error-resolve-dir",
242                        &[("dest_path", dest_path), ("err", &e.to_string())],
243                    )),
244                });
245            }
246        };
247
248        if !self.security.is_resolved_path_allowed(&canonical_parent) {
249            return Ok(ToolResult {
250                success: false,
251                output: String::new(),
252                error: Some(
253                    self.security
254                        .resolved_path_violation_message(&canonical_parent),
255                ),
256            });
257        }
258
259        let dest = canonical_parent.join(&file_name);
260        if !self.security.is_resolved_path_allowed(&dest) {
261            return Ok(ToolResult {
262                success: false,
263                output: String::new(),
264                error: Some(self.security.resolved_path_violation_message(&dest)),
265            });
266        }
267
268        // Debit the action budget only once the request is validated, mirroring
269        // file_upload — right before the network call.
270        if !self.security.record_action() {
271            return Ok(ToolResult {
272                success: false,
273                output: String::new(),
274                error: Some(Self::tool_msg(
275                    "tool-file-download-error-rate-limited-budget",
276                )),
277            });
278        }
279
280        // Disable redirect-following: the configured `[file_download].url` is
281        // the operator-approved endpoint, so a 3xx response from it must surface
282        // as a non-success status rather than silently rehome the request.
283        let builder = reqwest::Client::builder()
284            .timeout(Duration::from_secs(self.config.timeout_secs))
285            .connect_timeout(Duration::from_secs(10))
286            .redirect(reqwest::redirect::Policy::none());
287        let builder =
288            zeroclaw_config::schema::apply_runtime_proxy_to_builder(builder, "tool.file_download");
289        let client = match builder.build() {
290            Ok(c) => c,
291            Err(e) => {
292                return Ok(ToolResult {
293                    success: false,
294                    output: String::new(),
295                    error: Some(Self::tool_msg_with_args(
296                        "tool-file-download-error-client-build",
297                        &[("err", &e.to_string())],
298                    )),
299                });
300            }
301        };
302
303        let mut request = client.get(url).query(&[("document_id", document_id)]);
304        for (k, v) in &self.config.headers {
305            request = request.header(k.as_str(), v.as_str());
306        }
307
308        let response = match request.send().await {
309            Ok(r) => r,
310            Err(e) => {
311                return Ok(ToolResult {
312                    success: false,
313                    output: String::new(),
314                    error: Some(Self::tool_msg_with_args(
315                        "tool-file-download-error-request",
316                        &[("err", &e.to_string())],
317                    )),
318                });
319            }
320        };
321
322        let status = response.status();
323
324        if !status.is_success() {
325            let raw_body = response.text().await.unwrap_or_default();
326            let truncated = if raw_body.len() > RESPONSE_BODY_LIMIT_BYTES {
327                // The body is attacker-influenceable, so split on a char boundary
328                // to avoid panicking when the byte cutoff lands inside a
329                // multi-byte UTF-8 sequence. floor_char_boundary is unstable, so
330                // walk down at most three bytes — a UTF-8 code point is at most
331                // four bytes wide, so a boundary is always within reach.
332                let mut cut = RESPONSE_BODY_LIMIT_BYTES;
333                while cut > 0 && !raw_body.is_char_boundary(cut) {
334                    cut -= 1;
335                }
336                format!(
337                    "{}... [truncated {} bytes]",
338                    &raw_body[..cut],
339                    raw_body.len() - cut
340                )
341            } else {
342                raw_body
343            };
344            return Ok(ToolResult {
345                success: false,
346                output: truncated,
347                error: Some(Self::tool_msg_with_args(
348                    "tool-file-download-error-status",
349                    &[("status", &status.to_string())],
350                )),
351            });
352        }
353
354        // Fast-reject when the endpoint advertises an oversized body, before
355        // opening the destination file at all.
356        if let Some(len) = response.content_length()
357            && len > self.config.max_file_size_bytes
358        {
359            return Ok(ToolResult {
360                success: false,
361                output: String::new(),
362                error: Some(Self::tool_msg_with_args(
363                    "tool-file-download-error-too-large-reported",
364                    &[
365                        ("len", &len.to_string()),
366                        ("limit", &self.config.max_file_size_bytes.to_string()),
367                    ],
368                )),
369            });
370        }
371
372        // Stream into a temp file in the destination directory so a failed or
373        // oversized transfer never leaves a partial artifact at `dest`; on
374        // success the rename is atomic within the same directory.
375        let nanos = SystemTime::now()
376            .duration_since(UNIX_EPOCH)
377            .map(|d| d.as_nanos())
378            .unwrap_or(0);
379        let temp_path = canonical_parent.join(format!(".{file_name}.part-{nanos}"));
380
381        match Self::stream_to_temp(response, &temp_path, self.config.max_file_size_bytes).await {
382            Ok(written) => match tokio::fs::rename(&temp_path, &dest).await {
383                Ok(()) => {
384                    let output = Self::tool_msg_with_args(
385                        "tool-file-download-success",
386                        &[
387                            ("written", &written.to_string()),
388                            ("dest_path", dest_path),
389                            ("status", &status.to_string()),
390                        ],
391                    );
392                    // The download landed in an ephemeral workspace and will not
393                    // reach the host — warn loudly rather than report a bare
394                    // success (issue #4627).
395                    let output = if self.persistent_writes {
396                        output
397                    } else {
398                        with_ephemeral_workspace_warning(&output)
399                    };
400                    Ok(ToolResult {
401                        success: true,
402                        output,
403                        error: None,
404                    })
405                }
406                Err(e) => {
407                    let _ = tokio::fs::remove_file(&temp_path).await;
408                    Ok(ToolResult {
409                        success: false,
410                        output: String::new(),
411                        error: Some(Self::tool_msg_with_args(
412                            "tool-file-download-error-move",
413                            &[("err", &e.to_string())],
414                        )),
415                    })
416                }
417            },
418            Err(msg) => {
419                let _ = tokio::fs::remove_file(&temp_path).await;
420                Ok(ToolResult {
421                    success: false,
422                    output: String::new(),
423                    error: Some(msg),
424                })
425            }
426        }
427    }

909    async fn execute_truncates_non_ascii_error_body_safely() {
910        let server = MockServer::start().await;
911        let tmp = TempDir::new().unwrap();
912
913        // Build a non-2xx body that is longer than RESPONSE_BODY_LIMIT_BYTES
914        // (4096) and where the byte at offset 4096 lands inside a multi-byte
915        // UTF-8 sequence. Pre-truncation pad — 4094 ASCII bytes — places the
916        // first byte of the next 3-byte character ("界") at offset 4094, so
917        // offset 4096 lies in the middle of that code point.
918        let mut body = "x".repeat(4094);
919        body.push_str("世界世界世界世界世界世界");
920        assert!(!body.is_char_boundary(4096));
921
922        Mock::given(method("GET"))
923            .and(path("/download"))
924            .respond_with(ResponseTemplate::new(500).set_body_string(body.clone()))
925            .expect(1)
926            .mount(&server)
927            .await;
928
929        let config = FileDownloadConfig {
930            url: Some(format!("{}/download", server.uri())),
931            ..FileDownloadConfig::default()
932        };
933        let tool = FileDownloadTool::new(
934            test_security(tmp.path().to_path_buf(), AutonomyLevel::Full),
935            config,
936        );
937
938        // Must not panic when slicing the body at a non-char-boundary byte
939        // index. The truncated output must still be valid UTF-8 and must
940        // include the "[truncated ...]" marker.
941        let result = tool
942            .execute(json!({ "document_id": "doc-1", "dest_path": "out.bin" }))
943            .await
944            .unwrap();
945
946        assert!(!result.success);
947        assert!(result.error.as_deref().unwrap_or("").contains("500"));
948        assert!(result.output.contains("[truncated"));
949        assert!(
950            result.output.len() < body.len(),
951            "expected the body to be shortened"
952        );
953        assert!(!tmp.path().join("out.bin").exists());
954    }
