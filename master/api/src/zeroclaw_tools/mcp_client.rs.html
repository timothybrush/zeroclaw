22/// Timeout for receiving a response from an MCP server during init/list.
23/// Prevents a hung server from blocking the daemon indefinitely.
24const RECV_TIMEOUT_SECS: u64 = 30;
25
26/// Default timeout for tool calls (seconds) when not configured per-server.
27const DEFAULT_TOOL_TIMEOUT_SECS: u64 = 180;
28
29/// Maximum allowed tool call timeout (seconds) — hard safety ceiling.
30const MAX_TOOL_TIMEOUT_SECS: u64 = 600;
31
32/// Maximum automatic reconnect attempts on a stale session or dropped
33/// transport before the tool-call error is surfaced to the caller.
34const MAX_RECONNECT_ATTEMPTS: u32 = 2;
35
36/// Fixed backoff between reconnect attempts (milliseconds).
37const RECONNECT_BACKOFF_MS: u64 = 500;
38
39/// Perform the MCP `initialize` + `notifications/initialized` handshake on a
40/// transport. Shared by the initial [`McpServer::connect`] and the
41/// reconnect-after-stale-session path in [`McpServer::call_tool`].
42async fn handshake(transport: &mut dyn McpTransportConn, server_name: &str) -> Result<()> {
43    let init_req = JsonRpcRequest::new(
44        1,
45        "initialize",
46        json!({
47            "protocolVersion": MCP_PROTOCOL_VERSION,
48            "capabilities": {},
49            "clientInfo": {
50                "name": "zeroclaw",
51                "version": env!("CARGO_PKG_VERSION")
52            }
53        }),
54    );
55
56    let init_resp = timeout(
57        Duration::from_secs(RECV_TIMEOUT_SECS),
58        transport.send_and_recv(&init_req),
59    )
60    .await
61    .with_context(|| {
62        format!(
63            "MCP server `{server_name}` timed out after {RECV_TIMEOUT_SECS}s waiting for initialize response"
64        )
65    })??;
66
67    if init_resp.error.is_some() {
68        bail!(
69            "MCP server `{server_name}` rejected initialize: {:?}",
70            init_resp.error
71        );
72    }
73
74    // Notify the server the client is initialized (notifications expect no
75    // response). Best effort — ignore errors.
76    let notif = JsonRpcRequest::notification("notifications/initialized", json!({}));
77    let _ = transport.send_and_recv(&notif).await;
78
79    Ok(())
80}

96/// A live connection to one MCP server (any transport).
97#[derive(Clone)]
98pub struct McpServer {
99    inner: Arc<Mutex<McpServerInner>>,
100}
101
102impl McpServer {
103    /// Connect to the server, perform the initialize handshake, and fetch the tool list.
104    pub async fn connect(config: McpServerConfig) -> Result<Self> {
105        // Create transport based on config
106        let mut transport = create_transport(&config).with_context(|| {
107            format!(
108                "failed to create transport for MCP server `{}`",
109                config.name
110            )
111        })?;
112
113        // Initialize handshake (initialize + initialized notification)
114        handshake(transport.as_mut(), &config.name).await?;
115
116        // Fetch available tools
117        let id = 2u64;
118        let list_req = JsonRpcRequest::new(id, "tools/list", json!({}));
119
120        let list_resp = timeout(
121            Duration::from_secs(RECV_TIMEOUT_SECS),
122            transport.send_and_recv(&list_req),
123        )
124        .await
125        .with_context(|| {
126            format!(
127                "MCP server `{}` timed out after {}s waiting for tools/list response",
128                config.name, RECV_TIMEOUT_SECS
129            )
130        })??;
131
132        let result = list_resp.result.ok_or_else(|| {
133            ::zeroclaw_log::record!(
134                ERROR,
135                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
136                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
137                    .with_attrs(::serde_json::json!({"mcp_server": &config.name})),
138                "mcp_client: tools/list returned no result"
139            );
140            anyhow::Error::msg(format!(
141                "tools/list returned no result from `{}`",
142                config.name
143            ))
144        })?;
145        let tool_list: McpToolsListResult = serde_json::from_value(result)
146            .with_context(|| format!("failed to parse tools/list from `{}`", config.name))?;
147
148        let tool_count = tool_list.tools.len();
149
150        let inner = McpServerInner {
151            config,
152            transport,
153            #[cfg(target_has_atomic = "64")]
154            next_id: AtomicU64::new(3), // Start at 3 since we used 1 and 2
155            #[cfg(not(target_has_atomic = "64"))]
156            next_id: AtomicU32::new(3), // Start at 3 since we used 1 and 2
157            tools: tool_list.tools,
158        };
159
160        ::zeroclaw_log::record!(
161            INFO,
162            ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note),
163            &format!(
164                "MCP server `{}` connected — {} tool(s) available",
165                inner.config.name, tool_count
166            )
167        );
168
169        Ok(Self {
170            inner: Arc::new(Mutex::new(inner)),
171        })
172    }
173
174    /// Tools advertised by this server.
175    pub async fn tools(&self) -> Vec<McpToolDef> {
176        self.inner.lock().await.tools.clone()
177    }
178
179    /// Server display name.
180    pub async fn name(&self) -> String {
181        self.inner.lock().await.config.name.clone()
182    }

184    /// Call a tool on this server. Returns the raw JSON result.
185    pub async fn call_tool(
186        &self,
187        tool_name: &str,
188        arguments: serde_json::Value,
189    ) -> Result<serde_json::Value> {
190        let mut inner = self.inner.lock().await;
191
192        // Use per-server tool timeout if configured, otherwise default.
193        // Cap at MAX_TOOL_TIMEOUT_SECS for safety.
194        let tool_timeout = inner
195            .config
196            .tool_timeout_secs
197            .unwrap_or(DEFAULT_TOOL_TIMEOUT_SECS)
198            .min(MAX_TOOL_TIMEOUT_SECS);
199
200        // Bounded reconnect loop: a stale session (server restart) or a dropped
201        // transport (SSE stream EOF) is recovered by resetting the session and
202        // re-running the handshake, then retrying the call. Genuine tool errors
203        // (including `isError`) and timeouts are surfaced immediately and never
204        // retried.
205        let mut attempt = 0u32;
206        let resp = loop {
207            let id = inner.next_id.fetch_add(1, Ordering::Relaxed);
208            let req = JsonRpcRequest::new(
209                id,
210                "tools/call",
211                json!({ "name": tool_name, "arguments": arguments }),
212            );
213
214            let send_result = timeout(
215                Duration::from_secs(tool_timeout),
216                inner.transport.send_and_recv(&req),
217            )
218            .await
219            .map_err(|_| {
220                ::zeroclaw_log::record!(
221                    WARN,
222                    ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Timeout)
223                        .with_outcome(::zeroclaw_log::EventOutcome::Failure)
224                        .with_attrs(::serde_json::json!({
225                            "mcp_server": &inner.config.name,
226                            "tool": tool_name,
227                            "timeout_secs": tool_timeout,
228                        })),
229                    "mcp_client: tool call timed out"
230                );
231                anyhow::Error::msg(format!(
232                    "MCP server `{}` timed out after {}s during tool call `{tool_name}`",
233                    inner.config.name, tool_timeout
234                ))
235            })?;
236
237            match send_result {
238                Ok(resp) => break resp,
239                Err(err) => {
240                    // Reconnect only on recoverable transport errors, within budget.
241                    let recoverable_reason = err
242                        .downcast_ref::<McpTransportError>()
243                        .map(|te| te.to_string());
244                    if let Some(reason) = recoverable_reason
245                        && attempt < MAX_RECONNECT_ATTEMPTS
246                    {
247                        attempt += 1;
248                        let server_name = inner.config.name.clone();
249                        ::zeroclaw_log::record!(
250                            WARN,
251                            ::zeroclaw_log::Event::new(
252                                module_path!(),
253                                ::zeroclaw_log::Action::Reconnect
254                            )
255                            .with_outcome(::zeroclaw_log::EventOutcome::Failure)
256                            .with_attrs(::serde_json::json!({
257                                "mcp_server": &server_name,
258                                "tool": tool_name,
259                                "attempt": attempt,
260                                "max_attempts": MAX_RECONNECT_ATTEMPTS,
261                                "reason": &reason,
262                            })),
263                            "mcp_client: reconnecting after transport error and retrying tool call"
264                        );
265                        tokio::time::sleep(Duration::from_millis(RECONNECT_BACKOFF_MS)).await;
266                        inner.transport.reset().await.with_context(|| {
267                            format!(
268                                "MCP server `{server_name}` failed to reset transport during reconnect"
269                            )
270                        })?;
271                        handshake(inner.transport.as_mut(), &server_name)
272                            .await
273                            .with_context(|| {
274                                format!(
275                                    "MCP server `{server_name}` failed to re-handshake during reconnect"
276                                )
277                            })?;
278                        continue;
279                    }
280                    return Err(err).with_context(|| {
281                        format!(
282                            "MCP server `{}` error during tool call `{tool_name}`",
283                            inner.config.name
284                        )
285                    });
286                }
287            }
288        };
289
290        if let Some(err) = resp.error {
291            bail!("MCP tool `{tool_name}` error {}: {}", err.code, err.message);
292        }
293
294        let result = resp.result.unwrap_or(serde_json::Value::Null);
295
296        // MCP servers signal *tool-execution* failures (as opposed to JSON-RPC
297        // protocol errors) with HTTP 200 + `result.isError: true` and the detail
298        // in `result.content[].text`, per the MCP spec. Without surfacing this,
299        // the error envelope is returned as a normal success — so the failure is
300        // invisible to the model and the daemon log, and callers only ever see a
301        // generic "error during tool call" with no detail.
302        if result.get("isError").and_then(serde_json::Value::as_bool) == Some(true) {
303            let detail = result
304                .get("content")
305                .and_then(|c| c.as_array())
306                .map(|arr| {
307                    arr.iter()
308                        .filter_map(|item| item.get("text").and_then(|t| t.as_str()))
309                        .collect::<Vec<_>>()
310                        .join("\n")
311                })
312                .filter(|s: &String| !s.is_empty())
313                .unwrap_or_else(|| "(no error detail returned by server)".to_string());
314            // Server-controlled text: scrub secrets (sk-/ghp_/…) and bound length
315            // (`sanitize_api_error` truncates to MAX_API_ERROR_CHARS) before it
316            // reaches the daemon log or the returned error.
317            let detail = zeroclaw_providers::sanitize_api_error(&detail);
318            ::zeroclaw_log::record!(
319                WARN,
320                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
321                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
322                    .with_attrs(::serde_json::json!({
323                        "mcp_server": &inner.config.name,
324                        "tool": tool_name,
325                        "detail": &detail,
326                    })),
327                "mcp_client: tool returned isError:true"
328            );
329            bail!(
330                "MCP tool `{tool_name}` (server `{}`) returned isError: {detail}",
331                inner.config.name
332            );
333        }
334
335        Ok(result)
336    }
337}

341/// Registry of all connected MCP servers, with a flat tool index.
342pub struct McpRegistry {
343    servers: Vec<McpServer>,
344    /// prefixed_name → (server_index, original_tool_name)
345    tool_index: HashMap<String, (usize, String)>,
346}
347
348impl McpRegistry {
349    /// Connect to all configured servers. Non-fatal: failures are logged and skipped.
350    pub async fn connect_all(configs: &[McpServerConfig]) -> Result<Self> {
351        let mut servers = Vec::new();
352        let mut tool_index = HashMap::new();
353
354        for config in configs {
355            match McpServer::connect(config.clone()).await {
356                Ok(server) => {
357                    let server_idx = servers.len();
358                    // Collect tools while holding the lock once, then release
359                    let tools = server.tools().await;
360                    for tool in &tools {
361                        // Prefix prevents name collisions across servers
362                        let prefixed = format!("{}__{}", config.name, tool.name);
363                        tool_index.insert(prefixed, (server_idx, tool.name.clone()));
364                    }
365                    servers.push(server);
366                }
367                // Non-fatal — log and continue with remaining servers
368                Err(e) => {
369                    ::zeroclaw_log::record!(
370                        ERROR,
371                        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Fail)
372                            .with_outcome(::zeroclaw_log::EventOutcome::Failure),
373                        &format!("Failed to connect to MCP server `{}`: {:#}", config.name, e)
374                    );
375                }
376            }
377        }
378
379        Ok(Self {
380            servers,
381            tool_index,
382        })
383    }
384
385    /// All prefixed tool names across all connected servers.
386    pub fn tool_names(&self) -> Vec<String> {
387        self.tool_index.keys().cloned().collect()
388    }
389
390    /// Tool definition for a given prefixed name (cloned).
391    pub async fn get_tool_def(&self, prefixed_name: &str) -> Option<McpToolDef> {
392        let (server_idx, original_name) = self.tool_index.get(prefixed_name)?;
393        let inner = self.servers[*server_idx].inner.lock().await;
394        inner
395            .tools
396            .iter()
397            .find(|t| &t.name == original_name)
398            .cloned()
399    }
400
401    /// Execute a tool by prefixed name.
402    pub async fn call_tool(
403        &self,
404        prefixed_name: &str,
405        arguments: serde_json::Value,
406    ) -> Result<String> {
407        let (server_idx, original_name) = self.tool_index.get(prefixed_name).ok_or_else(|| {
408            ::zeroclaw_log::record!(
409                WARN,
410                ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Reject)
411                    .with_outcome(::zeroclaw_log::EventOutcome::Failure)
412                    .with_attrs(::serde_json::json!({"tool": prefixed_name})),
413                "mcp_client: unknown MCP tool"
414            );
415            anyhow::Error::msg(format!("unknown MCP tool `{prefixed_name}`"))
416        })?;
417        let result = self.servers[*server_idx]
418            .call_tool(original_name, arguments)
419            .await?;
420        serde_json::to_string_pretty(&result)
421            .with_context(|| format!("failed to serialize result of MCP tool `{prefixed_name}`"))
422    }
423
424    pub fn is_empty(&self) -> bool {
425        self.servers.is_empty()
426    }
427
428    pub fn server_count(&self) -> usize {
429        self.servers.len()
430    }
431
432    pub fn tool_count(&self) -> usize {
433        self.tool_index.len()
434    }
435}

704    #[cfg(unix)]
705    #[tokio::test]
706    async fn dropping_stdio_registry_reaps_child_process() {
707        use std::io::Write;
708        use std::os::unix::fs::PermissionsExt;
709        use std::path::Path;
710        use tokio::time::{Duration, sleep};
711
712        fn process_is_alive(pid: u32) -> bool {
713            std::process::Command::new("kill")
714                .arg("-0")
715                .arg(pid.to_string())
716                .stdout(std::process::Stdio::null())
717                .stderr(std::process::Stdio::null())
718                .status()
719                .is_ok_and(|status| status.success())
720        }
721
722        async fn read_pid(path: &Path) -> u32 {
723            for _ in 0..50 {
724                if let Ok(raw) = tokio::fs::read_to_string(path).await
725                    && let Ok(pid) = raw.trim().parse()
726                {
727                    return pid;
728                }
729                sleep(Duration::from_millis(20)).await;
730            }
731            panic!("stdio MCP test server did not write its pid");
732        }
733
734        let temp = tempfile::tempdir().expect("tempdir");
735        let server_path = temp.path().join("echo-mcp.sh");
736        let pid_path = temp.path().join("echo-mcp.pid");
737        let mut script = std::fs::File::create(&server_path).expect("script");
738        script
739            .write_all(
740                br#"#!/bin/sh
741echo "$$" > "$1"
742while IFS= read -r line; do
743  case "$line" in
744    *'"method":"initialize"'*)
745      printf '%s\n' '{"jsonrpc":"2.0","id":1,"result":{"protocolVersion":"2024-11-05","capabilities":{"tools":{}},"serverInfo":{"name":"echo-mcp","version":"0.1.0"}}}'
746      ;;
747    *'"method":"tools/list"'*)
748      printf '%s\n' '{"jsonrpc":"2.0","id":2,"result":{"tools":[]}}'
749      exec tail -f /dev/null
750      ;;
751  esac
752done
753"#,
754            )
755            .expect("write script");
756        drop(script);
757        let mut perms = std::fs::metadata(&server_path)
758            .expect("metadata")
759            .permissions();
760        perms.set_mode(0o755);
761        std::fs::set_permissions(&server_path, perms).expect("chmod");
762
763        let config = McpServerConfig {
764            name: "echo".to_string(),
765            command: server_path.display().to_string(),
766            args: vec![pid_path.display().to_string()],
767            env: std::collections::HashMap::default(),
768            tool_timeout_secs: None,
769            transport: McpTransport::Stdio,
770            url: None,
771            headers: std::collections::HashMap::default(),
772        };
773
774        let registry = McpRegistry::connect_all(&[config])
775            .await
776            .expect("connect_all should not fail");
777        assert_eq!(registry.server_count(), 1);
778        assert_eq!(registry.tool_count(), 0);
779        let child_pid = read_pid(&pid_path).await;
780        assert!(
781            process_is_alive(child_pid),
782            "stdio MCP child should be alive while the registry is alive"
783        );
784
785        drop(registry);
786
787        for _ in 0..50 {
788            if !process_is_alive(child_pid) {
789                return;
790            }
791            sleep(Duration::from_millis(20)).await;
792        }
793        panic!("stdio MCP child process {child_pid} survived after registry drop");
794    }

808    async fn call_tool_reconnects_on_stale_session() {
809        use wiremock::matchers::{body_partial_json, method};
810        use wiremock::{Mock, MockServer, ResponseTemplate};
811
812        let server = MockServer::start().await;
813
814        // initialize → 200 + session header. Hit twice: initial connect plus the
815        // reconnect that follows the stale-session error.
816        Mock::given(method("POST"))
817            .and(body_partial_json(json!({"method": "initialize"})))
818            .respond_with(
819                ResponseTemplate::new(200)
820                    .insert_header("Mcp-Session-Id", "sess-1")
821                    .set_body_json(json!({"jsonrpc": "2.0", "id": 1, "result": {}})),
822            )
823            .expect(2)
824            .mount(&server)
825            .await;
826
827        Mock::given(method("POST"))
828            .and(body_partial_json(
829                json!({"method": "notifications/initialized"}),
830            ))
831            .respond_with(ResponseTemplate::new(202))
832            .mount(&server)
833            .await;
834
835        Mock::given(method("POST"))
836            .and(body_partial_json(json!({"method": "tools/list"})))
837            .respond_with(ResponseTemplate::new(200).set_body_json(json!({
838                "jsonrpc": "2.0",
839                "id": 2,
840                "result": {"tools": [{"name": "echo", "description": "d", "inputSchema": {"type": "object"}}]}
841            })))
842            .expect(1)
843            .mount(&server)
844            .await;
845
846        // First tools/call → 404 (stale session). Highest priority, single use,
847        // so after it is exhausted the success mock below takes over.
848        Mock::given(method("POST"))
849            .and(body_partial_json(json!({"method": "tools/call"})))
850            .respond_with(ResponseTemplate::new(404))
851            .up_to_n_times(1)
852            .with_priority(1)
853            .expect(1)
854            .mount(&server)
855            .await;
856
857        // Retried tools/call after reconnect → success.
858        Mock::given(method("POST"))
859            .and(body_partial_json(json!({"method": "tools/call"})))
860            .respond_with(ResponseTemplate::new(200).set_body_json(json!({
861                "jsonrpc": "2.0", "id": 3, "result": {"ok": true}
862            })))
863            .expect(1)
864            .mount(&server)
865            .await;
866
867        let srv = McpServer::connect(http_server_config(server.uri()))
868            .await
869            .expect("connect");
870        let result = srv
871            .call_tool("echo", json!({}))
872            .await
873            .expect("call_tool should succeed after reconnect");
874        assert_eq!(result, json!({"ok": true}));
875        server.verify().await;
876    }
