1//! Curated sections surface — a flat ordered set of [`Section`]s the
2//! operator walks (new install) or scans (returning user) to configure
3//! a working ZeroClaw deployment.
4//!
5//! Every fact about a section (its enum variant, its on-the-wire key,
6//! its UI shape, its display group, its help blurb, its canonical
7//! position) lives in ONE table — the `sections!` invocation below. The
8//! macro expands that table into the [`Section`] enum, every per-variant
9//! `match` helper, and the [`QUICKSTART_SECTIONS`] const, so adding a
10//! section is exactly one row, no hand-listed variant set anywhere else.
11//!
12//! Consumers (CLI runtime, gateway, dashboard) dispatch off this enum;
13//! drift is a compile error.
14
15use serde::{Deserialize, Serialize};

17/// UI rendering shape for a section. Drives picker / form dispatch on
18/// the `/config` curated section explorer and the Quickstart flow.
19#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
20#[cfg_attr(feature = "schema-export", derive(schemars::JsonSchema))]
21#[serde(rename_all = "snake_case")]
22pub enum SectionShape {
23    /// `<section>` renders a schema-driven form with no picker step.
24    DirectForm,
25    /// `<section>.<alias>` map of structured entries; the section page
26    /// shows an alias list with `+ Add` and clicking an alias opens its
27    /// schema form.
28    OneTierAliasMap,
29    /// `<section>.<type>.<alias>` two-tier map. Picker chooses `<type>`,
30    /// alias-list step chooses `<alias>`, then the schema form opens.
31    TypedFamilyMap,
32    /// Single non-alias choice (memory backend, tunnel provider). Picker
33    /// flips a top-level field, then the schema form for the chosen
34    /// backend/provider renders.
35    BackendPicker,
36}

38/// Display group for the Config menu. Every curated [`Section`] row
39/// names one in its `group:` cell; the long tail of schema-derived
40/// top-level roots (`gateway`, `observability`, …) that surface in the
41/// section explorer without a curated row resolves through
42/// [`section_group_for_key`]. [`SECTION_GROUPS`] fixes the display
43/// order across every surface.
44///
45/// `Other` is the catch-all: unknown keys land there so new schema
46/// additions still surface in the menu until someone curates them.
47#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
48#[cfg_attr(feature = "schema-export", derive(schemars::JsonSchema))]
49#[serde(rename_all = "snake_case")]
50pub enum SectionGroup {
51    /// Quickstart-walked essentials: providers, profiles, channels,
52    /// agents — the most-edited sections, surfaced first.
53    Foundation,
54    /// Agent loop, scheduling, and orchestration tuning.
55    Agent,
56    /// Multi-agent / delegation.
57    MultiAgent,
58    /// Tool integrations the agent can call.
59    Tools,
60    /// External services / vendor integrations.
61    Integrations,
62    /// Networking / multi-node infrastructure.
63    Network,
64    /// Storage, identity, secrets.
65    Storage,
66    /// Operations / monitoring / safety / cost.
67    Operations,
68    /// Catch-all for keys no one has curated yet.
69    Other,
70}

72impl SectionGroup {
73    /// UI label. These exact strings are what `ConfigSectionEntry.group`
74    /// carries on the wire and what the dashboard's `GROUP_ORDER`
75    /// (web/src/pages/Config.tsx) and the zerocode Config pane group
76    /// by — change one, change all of them together.
77    #[must_use]
78    pub const fn label(self) -> &'static str {
79        match self {
80            Self::Foundation => "Foundation",
81            Self::Agent => "Agent",
82            Self::MultiAgent => "Multi-agent",
83            Self::Tools => "Tools",
84            Self::Integrations => "Integrations",
85            Self::Network => "Network",
86            Self::Storage => "Storage",
87            Self::Operations => "Operations",
88            Self::Other => "Other",
89        }
90    }
91
92    /// Inverse of [`label`](Self::label): parse a group label string
93    /// back into the enum, or `None` for an unrecognized label. The
94    /// bridge for `#[group = "..."]` attributes resolved through the
95    /// schema — the derive emits the label as a string the macro can't
96    /// type-check, so this validates it on the way back in.
97    #[must_use]
98    pub fn from_label(label: &str) -> Option<Self> {
99        SECTION_GROUPS.iter().copied().find(|g| g.label() == label)
100    }
101}

104    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
105        f.write_str(self.label())
106    }

109/// Canonical display order for section groups across every Config
110/// surface (dashboard sidebar, zerocode Config pane). "All sections"
111/// is not a group — a surface that offers a flat default view renders
112/// that view itself; this list orders the grouped presentation.
113pub const SECTION_GROUPS: &[SectionGroup] = &[
114    SectionGroup::Foundation,
115    SectionGroup::Agent,
116    SectionGroup::MultiAgent,
117    SectionGroup::Tools,
118    SectionGroup::Integrations,
119    SectionGroup::Network,
120    SectionGroup::Storage,
121    SectionGroup::Operations,
122    SectionGroup::Other,
123];

125/// Display group for any section key. Two declarative sources, no
126/// hand-maintained table:
127///
128/// 1. Curated [`Section`]s read the `group:` cell of their `sections!`
129///    row.
130/// 2. The long tail of top-level `Config` roots the section explorer
131///    surfaces without a curated row declares its group inline via
132///    `#[group = "..."]`, harvested by the `Configurable` derive into
133///    [`crate::schema::Config::nested_section_group`]. This mirrors how
134///    [`section_help`] falls through to `Config::nested_section_help`.
135///
136/// Unknown keys — and roots whose `#[group]` label fails to parse — fall
137/// into [`SectionGroup::Other`] so new schema additions still surface;
138/// they just land in the catch-all bucket until someone annotates them.
139/// The `every_surfaced_root_has_a_group` test guards against a real root
140/// silently landing there.
141#[must_use]
142pub fn section_group_for_key(key: &str) -> SectionGroup {
143    if let Some(s) = Section::from_key(key) {
144        return s.group();
145    }
146    // `nested_section_group` arms are keyed by the Rust field ident,
147    // i.e. snake_case (the derive's `snake_to_kebab` is currently an
148    // identity passthrough). Section keys reaching here are snake too
149    // (the RPC/gateway pass `prop_fields()` first segments), so the
150    // direct lookup hits; tolerate a kebab-spelled caller by normalizing
151    // back to snake, matching `Section::from_key`'s leniency.
152    crate::schema::Config::nested_section_group(key)
153        .or_else(|| crate::schema::Config::nested_section_group(&key.replace('-', "_")))
154        .and_then(SectionGroup::from_label)
155        .unwrap_or(SectionGroup::Other)
156}

158/// Humanize a section wire key for display (`risk_profiles` → `Risk profiles`,
159/// `providers.models` → `Model providers`). Single source of truth for section
160/// labels across the gateway dashboard, zerocode Config pane, and docs. Specific
161/// wording overrides are listed explicitly; everything else is mechanically
162/// title-cased from the key.
163#[must_use]
164pub fn humanize_section_key(key: &str) -> String {
165    match key {
166        "providers.models" => return "Model providers".to_string(),
167        "providers.tts" => return "TTS providers".to_string(),
168        "providers.transcription" => return "Transcription providers".to_string(),
169        _ => {}
170    }
171    let mut s = key.replace(['_', '-', '.'], " ");
172    if let Some(c) = s.get_mut(0..1) {
173        c.make_ascii_uppercase();
174    }
175    s
176}

178/// Single source of truth for every pickable config section. Each row
179/// maps 1:1 to a dashboard `/config/<key>` page, a CLI
180/// `zeroclaw quickstart` flow and the gateway section picker handler.
181/// Adding/removing a section is one row here and every consumer's
182/// `match` either compiles cleanly or fails with an exhaustiveness
183/// error pointing at exactly what needs an arm.
184///
185/// Row order is the canonical order operators see in the dashboard
186/// and walk through in the CLI. It is dependency-correct: every
187/// downstream alias reference an Agent carries (model_provider,
188/// risk_profile, runtime_profile, channels, *_bundles) appears earlier
189/// in the list than [`Section::Agents`], so walking top-to-bottom
190/// never produces a dangling reference.
191macro_rules! sections {
192    (
193        $(
194            $var:ident => {
195                key:   $key:literal,
196                shape: $shape:ident,
197                group: $group:ident,
198                help:  $help:expr $(,)?
199            }
200        ),+ $(,)?
201    ) => {
202        /// One pickable section. The variant ordering follows the
203        /// `sections!` macro invocation.
204        ///
205        /// With the `clap` feature on, this enum doubles as the
206        /// `zeroclaw quickstart` and curated-section endpoints — no separate
207        /// mirror enum in the binary crate.
208        #[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
209        #[cfg_attr(feature = "schema-export", derive(schemars::JsonSchema))]
210        #[cfg_attr(feature = "clap", derive(clap::Subcommand))]
211        #[serde(rename_all = "snake_case")]
212        pub enum Section {
213            $(
214                // Both clap (`--help`) and our runtime `help()` method
215                // need the same blurb; emit it once as a doc comment so
216                // the two surfaces share a single string per variant.
217                #[doc = $help]
218                #[cfg_attr(feature = "clap", command(name = $key))]
219                $var,
220            )+
221        }
222
223        impl Section {
224            /// Stable on-the-wire key. Also serves as the TOML
225            /// top-level prefix (e.g. `providers.models.<type>.<alias>`),
226            /// the curated section URL segment, and the
227            /// `SectionInfo.key` field returned by the gateway.
228            #[must_use]
229            pub const fn as_str(self) -> &'static str {
230                match self {
231                    $( Self::$var => $key, )+
232                }
233            }
234
235            /// Editor shape — the dashboard and the CLI both
236            /// dispatch off this so the same component lights up for
237            /// the same section in both surfaces.
238            #[must_use]
239            pub const fn shape(self) -> SectionShape {
240                match self {
241                    $( Self::$var => SectionShape::$shape, )+
242                }
243            }
244
245            /// Display group for the Config menu — the bucket this
246            /// section renders under in the dashboard sidebar and the
247            /// zerocode Config pane. Read the `group:` cell of the
248            /// `sections!` row; the long tail of non-curated section
249            /// keys resolves through [`section_group_for_key`] instead.
250            #[must_use]
251            pub const fn group(self) -> SectionGroup {
252                match self {
253                    $( Self::$var => SectionGroup::$group, )+
254                }
255            }
256
257            /// Per-section help blurb — single source of truth for
258            /// the copy shown above the section's picker / form on
259            /// every surface (CLI `ui.note(...)`, TUI heading,
260            /// dashboard `SectionInfo.help`).
261            #[must_use]
262            pub const fn help(self) -> &'static str {
263                match self {
264                    $( Self::$var => $help, )+
265                }
266            }
267
268            /// Human-readable section label shown in every Config surface
269            /// (gateway dashboard sidebar, zerocode Config pane, docs).
270            /// Single source of truth — derived from the canonical wire key
271            /// so the gateway, runtime, and docs cannot disagree.
272            #[must_use]
273            pub fn label(self) -> String {
274                humanize_section_key(self.key())
275            }
276
277            /// The canonical wire key for this section.
278            #[must_use]
279            pub const fn key(self) -> &'static str {
280                match self {
281                    $( Self::$var => $key, )+
282                }
283            }
284
285            /// Parse a stable wire key, tolerating both the snake and
286            /// kebab spellings of any section. The schema mixes the two:
287            /// `model_providers` (snake) and `peer-groups` (kebab) are
288            /// both valid wire forms produced elsewhere in the codebase.
289            /// Callers (dashboard URL routing, gateway picker dispatch,
290            /// CLI clap subcommands) can pass either form; `from_key`
291            /// resolves to the same variant. Returns `None` for keys
292            /// outside the known section table. Named `from_key` rather
293            /// than `from_str` so clippy doesn't flag it as confusable
294            /// with `std::str::FromStr` (parse failure is `None`, not
295            /// `Err(_)`).
296            #[must_use]
297            pub fn from_key(s: &str) -> Option<Self> {
298                let try_match = |s: &str| -> Option<Self> {
299                    match s {
300                        $( $key => Some(Self::$var), )+
301                        _ => None,
302                    }
303                };
304                if let Some(v) = try_match(s) {
305                    return Some(v);
306                }
307                if s.contains('_')
308                    && let Some(v) = try_match(&s.replace('_', "-"))
309                {
310                    return Some(v);
311                }
312                if s.contains('-')
313                    && let Some(v) = try_match(&s.replace('-', "_"))
314                {
315                    return Some(v);
316                }
317                None
318            }
319        }
320
321        /// Canonical ordering of sections enumerated by
322        /// the Quickstart flow and the curated section explorer. The
323        /// dashboard renders Next/Finish navigation against this list.
324        /// Every consumer that needs section ordering reads from here.
325        pub const QUICKSTART_SECTIONS: &[Section] = &[ $( Section::$var ),+ ];
326    };
327}

529/// Canonical-order index of `section` in [`QUICKSTART_SECTIONS`].
530/// Always `Some` for any valid `Section` variant — the const includes
531/// every variant by construction. Returns `Option` for API symmetry
532/// with [`section_index_for_key`], which can fail on unknown keys.
533#[must_use]
534pub fn section_index(section: Section) -> Option<usize> {
535    QUICKSTART_SECTIONS.iter().position(|s| *s == section)
536}

538/// Canonical-order index for a wire key, or `None` if the key isn't a

541#[must_use]
542pub fn section_index_for_key(key: &str) -> Option<usize> {
543    Section::from_key(key).and_then(section_index)

552/// Help blurb for a section key, covering both `Section` variants and
553/// the long tail of top-level `Config` fields the dashboard / TUI config
554/// editor surface (gateway, scheduler, observability, …). Single source
555/// of truth shared by every surface — the gateway sidebar, the CLI
556/// Quickstart flow, and the future TUI config editor all call this rather
557/// than maintaining parallel tables.
558///
559/// Resolution order:
560/// 1. `Section` variants (curated `help` text next to the variant
561///    declaration in the `sections!` macro).
562/// 2. The `Config` struct's `#[nested]` field-level `///` docstring,
563///    harvested by the `Configurable` derive into
564///    `Config::nested_section_help`. This is what makes adding a new
565///    top-level section a one-line schema change with no parallel
566///    help table to update.
567///
568/// Returns `""` for keys without a docstring so callers can decide
569/// whether to omit the help row or show a fallback.
570#[must_use]
571pub fn section_help(key: &str) -> &'static str {
572    if let Some(s) = Section::from_key(key) {
573        return s.help();
574    }
575    crate::schema::Config::nested_section_help(key).unwrap_or("")
576}

586/// Does this section show any signal of having been touched on this
587/// install? Used by callers (RPC config-list filtering, lifecycle
588/// dispatch) to decide whether to surface a section as "untouched".
589///
590/// Each variant decides what counts as a real signal vs a default
591/// value that round-trips identically across a fresh install.
592pub fn section_has_signal(cfg: &crate::schema::Config, section: Section) -> bool {
593    match section {
594        Section::ModelProviders => !cfg.providers.models.is_empty(),
595        // `channels.cli: bool` is a default-true scalar that lives directly
596        // under `channels.*`, so a bare `starts_with("channels.")` check
597        // fires on every fresh install. Require a nested channel config
598        // (e.g. `channels.telegram.bot-token`) — anything with a second dot
599        // segment — to count as user-driven signal.
600        Section::Channels => cfg.prop_fields().iter().any(|f| {
601            f.name
602                .strip_prefix("channels.")
603                .is_some_and(|rest| rest.contains('.'))
604        }),
605        Section::Hardware => cfg.hardware.enabled,
606        // Servers' existence in the Vec is the signal — a fresh install
607        // has an empty `mcp.servers`, so any element at all (even one
608        // with no command set) counts as user intent to use MCP. The
609        // parent `Mcp` section stays marker-only because its top-level

952    /// Drift guard for the schema-attribute grouping: every top-level
953    /// config root the section explorer can surface must resolve to a
954    /// real [`SectionGroup`], not the `Other` catch-all — either through
955    /// the `sections!` table (curated rows) or a `#[group = "..."]`
956    /// annotation harvested into `Config::nested_section_group`.
957    ///
958    /// This is what makes grouping declarative-by-construction: add a
959    /// new top-level `#[nested]` section without a group and this test
960    /// fails, naming the root, rather than the section silently landing
961    /// in `Other`. A root that genuinely has no curated group yet goes
962    /// in `UNGROUPED` with intent; `HIDDEN` covers system roots the
963    /// explorer never shows.
964    #[test]
965    fn every_surfaced_root_has_a_group() {
966        use crate::schema::Config;
967        let cfg = Config::default();
968        let roots: std::collections::BTreeSet<String> = cfg
969            .prop_fields()
970            .iter()
971            .filter_map(|f| f.name.split('.').next().map(str::to_string))
972            .collect();
973
974        // System/bookkeeping root the explorer hides; resolved at
975        // runtime, never user-edited, so intentionally ungrouped.
976        const HIDDEN: &[&str] = &["schema_version"];
977
978        // Roots with no curated group yet: they render in the `Other`
979        // bucket. Pre-existing — ungrouped before grouping moved into
980        // the schema. Annotating one with `#[group]` (e.g. `microsoft365`
981        // and the `file_*` tool sections are obvious candidates) means
982        // removing it from this list. The assertions below keep the list
983        // honest: every entry must still be a real, still-ungrouped root.
984        const UNGROUPED: &[&str] = &[
985            "escalation",
986            "locale",
987            "microsoft365",
988            "model_routes",
989            "embedding_routes",
990            "file_upload",
991            "file_upload_bundle",
992            "file_download",
993            "wss",
994        ];
995
996        let violations: Vec<&String> = roots
997            .iter()
998            .filter(|r| !HIDDEN.contains(&r.as_str()) && !UNGROUPED.contains(&r.as_str()))
999            .filter(|r| section_group_for_key(r) == SectionGroup::Other)
1000            .collect();
1001        assert!(
1002            violations.is_empty(),
1003            "these surfaced config roots resolve to SectionGroup::Other — add a \
1004             `#[group = \"...\"]` to each in schema.rs (or, if intentionally \
1005             uncurated, to the UNGROUPED allowlist): {violations:?}",
1006        );
1007
1008        // Keep the allowlist from rotting: each entry must still be a
1009        // surfaced root and still ungrouped.
1010        for u in UNGROUPED {
1011            assert!(
1012                roots.contains(*u),
1013                "UNGROUPED lists `{u}` but it is no longer a surfaced root — remove it",
1014            );
1015            assert_eq!(
1016                section_group_for_key(u),
1017                SectionGroup::Other,
1018                "`{u}` now resolves to a real group — remove it from UNGROUPED",
1019            );
1020        }
1021    }
